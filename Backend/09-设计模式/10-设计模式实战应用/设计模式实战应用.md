## 1. 如何设计一个可扩展的通知系统？

**回答：**

```
需求：支持邮件/短信/推送/Webhook 等多渠道通知

使用模式：
  策略模式 → 不同发送渠道
  模板方法 → 统一发送流程
  观察者模式 → 事件触发通知
  装饰器 → 重试/日志/限流
```

```go
// 策略：通知渠道
type Channel interface {
    Send(ctx context.Context, msg *Message) error
    Type() string
}

type EmailChannel struct{ client SMTPClient }
type SMSChannel struct{ client SMSClient }
type WebhookChannel struct{ httpClient *http.Client }

// 通知服务
type NotificationService struct {
    channels map[string]Channel
    rules    []RoutingRule
}

type RoutingRule struct {
    EventType string
    Channels  []string
    Template  string
}

func (s *NotificationService) Notify(ctx context.Context, event Event) error {
    rules := s.matchRules(event.Type)
    
    var errs []error
    for _, rule := range rules {
        msg := s.renderTemplate(rule.Template, event.Data)
        for _, chName := range rule.Channels {
            ch, ok := s.channels[chName]
            if !ok {
                continue
            }
            if err := ch.Send(ctx, msg); err != nil {
                errs = append(errs, fmt.Errorf("%s: %w", chName, err))
            }
        }
    }
    return errors.Join(errs...)
}

// 路由配置：
// user.registered → [email, sms]
// order.shipped → [email, push, webhook]
// payment.failed → [sms, push]
```

## 2. 如何设计一个插件系统？

**回答：**

```
使用模式：
  工厂 + 注册表 → 插件注册
  策略模式 → 插件行为
  模板方法 → 插件生命周期
```

```go
// 插件接口
type Plugin interface {
    Name() string
    Init(config map[string]interface{}) error
    Execute(ctx context.Context, input interface{}) (interface{}, error)
    Close() error
}

// 插件注册表
type PluginRegistry struct {
    mu       sync.RWMutex
    factories map[string]func() Plugin
    instances map[string]Plugin
}

func NewPluginRegistry() *PluginRegistry {
    return &PluginRegistry{
        factories: make(map[string]func() Plugin),
        instances: make(map[string]Plugin),
    }
}

func (r *PluginRegistry) Register(name string, factory func() Plugin) {
    r.mu.Lock()
    defer r.mu.Unlock()
    r.factories[name] = factory
}

func (r *PluginRegistry) Load(name string, config map[string]interface{}) error {
    r.mu.Lock()
    defer r.mu.Unlock()
    
    factory, ok := r.factories[name]
    if !ok {
        return fmt.Errorf("plugin %s not registered", name)
    }
    
    plugin := factory()
    if err := plugin.Init(config); err != nil {
        return fmt.Errorf("init plugin %s: %w", name, err)
    }
    
    r.instances[name] = plugin
    return nil
}

func (r *PluginRegistry) Execute(ctx context.Context, name string, input interface{}) (interface{}, error) {
    r.mu.RLock()
    plugin, ok := r.instances[name]
    r.mu.RUnlock()
    
    if !ok {
        return nil, fmt.Errorf("plugin %s not loaded", name)
    }
    return plugin.Execute(ctx, input)
}

// 具体插件
type JSONValidator struct{}
func (p *JSONValidator) Name() string { return "json-validator" }
func (p *JSONValidator) Init(config map[string]interface{}) error { return nil }
func (p *JSONValidator) Execute(ctx context.Context, input interface{}) (interface{}, error) {
    data := input.([]byte)
    var result interface{}
    if err := json.Unmarshal(data, &result); err != nil {
        return nil, fmt.Errorf("invalid JSON: %w", err)
    }
    return result, nil
}
func (p *JSONValidator) Close() error { return nil }

// 注册
registry.Register("json-validator", func() Plugin { return &JSONValidator{} })
```

## 3. 如何实现一个规则引擎？

**回答：**

```
使用模式：
  策略模式 → 不同规则
  责任链 → 规则依次匹配
  组合模式 → 复合条件（AND/OR）
```

```go
// 规则接口
type Rule interface {
    Evaluate(ctx context.Context, facts map[string]interface{}) bool
}

type Action interface {
    Execute(ctx context.Context, facts map[string]interface{}) error
}

// 组合条件
type AndRule struct {
    rules []Rule
}

func (r *AndRule) Evaluate(ctx context.Context, facts map[string]interface{}) bool {
    for _, rule := range r.rules {
        if !rule.Evaluate(ctx, facts) {
            return false
        }
    }
    return true
}

type OrRule struct {
    rules []Rule
}

func (r *OrRule) Evaluate(ctx context.Context, facts map[string]interface{}) bool {
    for _, rule := range r.rules {
        if rule.Evaluate(ctx, facts) {
            return true
        }
    }
    return false
}

// 比较规则
type CompareRule struct {
    Field    string
    Operator string
    Value    interface{}
}

func (r *CompareRule) Evaluate(ctx context.Context, facts map[string]interface{}) bool {
    val, ok := facts[r.Field]
    if !ok {
        return false
    }
    switch r.Operator {
    case "==":
        return val == r.Value
    case ">":
        return toFloat64(val) > toFloat64(r.Value)
    case "<":
        return toFloat64(val) < toFloat64(r.Value)
    }
    return false
}

// 规则引擎
type RuleEngine struct {
    rules []struct {
        condition Rule
        action    Action
    }
}

func (e *RuleEngine) AddRule(condition Rule, action Action) {
    e.rules = append(e.rules, struct {
        condition Rule
        action    Action
    }{condition, action})
}

func (e *RuleEngine) Execute(ctx context.Context, facts map[string]interface{}) error {
    for _, r := range e.rules {
        if r.condition.Evaluate(ctx, facts) {
            if err := r.action.Execute(ctx, facts); err != nil {
                return err
            }
        }
    }
    return nil
}

// 使用：VIP 折扣规则
engine.AddRule(
    &AndRule{rules: []Rule{
        &CompareRule{Field: "user_level", Operator: "==", Value: "vip"},
        &CompareRule{Field: "order_amount", Operator: ">", Value: 100.0},
    }},
    &DiscountAction{Rate: 0.9},
)
```

## 4. 如何设计限流器？

**回答：**

```go
// 策略模式 + 工厂

type Limiter interface {
    Allow(key string) bool
    AllowN(key string, n int) bool
}

// 令牌桶
type TokenBucket struct {
    mu       sync.Mutex
    rate     float64
    capacity int
    tokens   float64
    lastTime time.Time
}

func NewTokenBucket(rate float64, capacity int) *TokenBucket {
    return &TokenBucket{
        rate:     rate,
        capacity: capacity,
        tokens:   float64(capacity),
        lastTime: time.Now(),
    }
}

func (tb *TokenBucket) Allow(key string) bool {
    return tb.AllowN(key, 1)
}

func (tb *TokenBucket) AllowN(key string, n int) bool {
    tb.mu.Lock()
    defer tb.mu.Unlock()
    
    now := time.Now()
    elapsed := now.Sub(tb.lastTime).Seconds()
    tb.tokens += elapsed * tb.rate
    if tb.tokens > float64(tb.capacity) {
        tb.tokens = float64(tb.capacity)
    }
    tb.lastTime = now
    
    if tb.tokens >= float64(n) {
        tb.tokens -= float64(n)
        return true
    }
    return false
}

// 滑动窗口
type SlidingWindow struct {
    mu       sync.Mutex
    limit    int
    window   time.Duration
    requests []time.Time
}

func (sw *SlidingWindow) Allow(key string) bool {
    sw.mu.Lock()
    defer sw.mu.Unlock()
    
    now := time.Now()
    windowStart := now.Add(-sw.window)
    
    // 移除过期请求
    valid := sw.requests[:0]
    for _, t := range sw.requests {
        if t.After(windowStart) {
            valid = append(valid, t)
        }
    }
    sw.requests = valid
    
    if len(sw.requests) >= sw.limit {
        return false
    }
    sw.requests = append(sw.requests, now)
    return true
}

// 工厂
func NewLimiter(strategy string, rate float64, capacity int) Limiter {
    switch strategy {
    case "token_bucket":
        return NewTokenBucket(rate, capacity)
    case "sliding_window":
        return &SlidingWindow{limit: capacity, window: time.Second}
    default:
        return NewTokenBucket(rate, capacity)
    }
}
```

## 5. 如何设计配置中心客户端？

**回答：**

```go
// 使用模式：观察者(配置变更) + 策略(多来源) + 单例

type ConfigSource interface {
    Load(ctx context.Context) (map[string]string, error)
    Watch(ctx context.Context, onChange func(map[string]string))
}

type ConfigManager struct {
    mu       sync.RWMutex
    config   map[string]string
    sources  []ConfigSource
    watchers []func(key, oldVal, newVal string)
}

func NewConfigManager(sources ...ConfigSource) *ConfigManager {
    cm := &ConfigManager{
        config:  make(map[string]string),
        sources: sources,
    }
    return cm
}

func (cm *ConfigManager) Load(ctx context.Context) error {
    merged := make(map[string]string)
    
    // 按优先级加载（后面覆盖前面）
    for _, source := range cm.sources {
        cfg, err := source.Load(ctx)
        if err != nil {
            return err
        }
        for k, v := range cfg {
            merged[k] = v
        }
    }
    
    cm.mu.Lock()
    cm.config = merged
    cm.mu.Unlock()
    return nil
}

func (cm *ConfigManager) Get(key string) string {
    cm.mu.RLock()
    defer cm.mu.RUnlock()
    return cm.config[key]
}

func (cm *ConfigManager) OnChange(fn func(key, oldVal, newVal string)) {
    cm.mu.Lock()
    cm.watchers = append(cm.watchers, fn)
    cm.mu.Unlock()
}

// 配置来源：环境变量
type EnvSource struct{ prefix string }
func (s *EnvSource) Load(ctx context.Context) (map[string]string, error) {
    result := make(map[string]string)
    for _, env := range os.Environ() {
        parts := strings.SplitN(env, "=", 2)
        if strings.HasPrefix(parts[0], s.prefix) {
            result[parts[0]] = parts[1]
        }
    }
    return result, nil
}

// 配置来源：文件
type FileSource struct{ path string }

// 配置来源：远程（etcd/consul）
type EtcdSource struct{ client *clientv3.Client }
```

## 6. 如何设计缓存层？

**回答：**

```go
// 装饰器模式：多级缓存

type Cache interface {
    Get(ctx context.Context, key string) ([]byte, error)
    Set(ctx context.Context, key string, value []byte, ttl time.Duration) error
    Delete(ctx context.Context, key string) error
}

// L1: 本地缓存
type LocalCache struct {
    data sync.Map
}

// L2: Redis 缓存
type RedisCache struct {
    client *redis.Client
}

// 多级缓存（装饰器组合）
type MultiLevelCache struct {
    levels []Cache
}

func NewMultiLevelCache(levels ...Cache) *MultiLevelCache {
    return &MultiLevelCache{levels: levels}
}

func (c *MultiLevelCache) Get(ctx context.Context, key string) ([]byte, error) {
    for i, level := range c.levels {
        data, err := level.Get(ctx, key)
        if err == nil {
            // 回填上层缓存
            for j := 0; j < i; j++ {
                c.levels[j].Set(ctx, key, data, 0)
            }
            return data, nil
        }
    }
    return nil, ErrCacheMiss
}

func (c *MultiLevelCache) Set(ctx context.Context, key string, value []byte, ttl time.Duration) error {
    for _, level := range c.levels {
        if err := level.Set(ctx, key, value, ttl); err != nil {
            return err
        }
    }
    return nil
}

// 使用
cache := NewMultiLevelCache(
    &LocalCache{},               // L1
    &RedisCache{client: rdb},    // L2
)
```

## 7. 日志系统如何设计？

**回答：**

```go
// 策略模式 + 装饰器 + 建造者

type Logger interface {
    Debug(msg string, fields ...Field)
    Info(msg string, fields ...Field)
    Warn(msg string, fields ...Field)
    Error(msg string, fields ...Field)
}

type Field struct {
    Key   string
    Value interface{}
}

// 日志装饰器：添加上下文
type ContextLogger struct {
    base   Logger
    fields []Field
}

func (l *ContextLogger) With(fields ...Field) *ContextLogger {
    return &ContextLogger{
        base:   l.base,
        fields: append(l.fields, fields...),
    }
}

func (l *ContextLogger) Info(msg string, fields ...Field) {
    allFields := append(l.fields, fields...)
    l.base.Info(msg, allFields...)
}

// 使用：请求维度的日志
func HandleRequest(ctx context.Context, req *Request) {
    logger := baseLogger.With(
        Field{"request_id", req.ID},
        Field{"user_id", req.UserID},
    )
    
    logger.Info("processing request")
    // 后续所有日志自动带 request_id 和 user_id
    
    result, err := processOrder(ctx, req)
    if err != nil {
        logger.Error("process failed", Field{"error", err})
        return
    }
    logger.Info("request completed", Field{"result", result})
}
```

## 8. 任务调度器怎么设计？

**回答：**

```go
// 命令模式 + Worker Pool + 优先队列

type Task struct {
    ID       string
    Priority int
    Fn       func(ctx context.Context) error
    Timeout  time.Duration
}

type Scheduler struct {
    tasks   chan *Task
    workers int
    wg      sync.WaitGroup
}

func NewScheduler(workers, queueSize int) *Scheduler {
    s := &Scheduler{
        tasks:   make(chan *Task, queueSize),
        workers: workers,
    }
    s.start()
    return s
}

func (s *Scheduler) start() {
    for i := 0; i < s.workers; i++ {
        s.wg.Add(1)
        go func(id int) {
            defer s.wg.Done()
            for task := range s.tasks {
                s.executeTask(id, task)
            }
        }(i)
    }
}

func (s *Scheduler) executeTask(workerID int, task *Task) {
    ctx, cancel := context.WithTimeout(context.Background(), task.Timeout)
    defer cancel()
    
    log.Printf("worker-%d executing task %s", workerID, task.ID)
    
    if err := task.Fn(ctx); err != nil {
        log.Printf("task %s failed: %v", task.ID, err)
    }
}

func (s *Scheduler) Submit(task *Task) {
    s.tasks <- task
}

func (s *Scheduler) Shutdown() {
    close(s.tasks)
    s.wg.Wait()
}

// 使用
scheduler := NewScheduler(10, 100)
defer scheduler.Shutdown()

scheduler.Submit(&Task{
    ID:      "send-email-001",
    Priority: 1,
    Timeout: 10 * time.Second,
    Fn: func(ctx context.Context) error {
        return sendEmail(ctx, "user@example.com", "Hello")
    },
})
```

## 9. 面试中设计类问题怎么回答？

**回答：**

```
回答方法论：

Step 1：明确需求
  功能需求：做什么
  非功能需求：性能/可用性/扩展性

Step 2：定义接口
  核心抽象是什么
  对外暴露什么 API

Step 3：选择模式
  创建对象 → 工厂/建造者
  行为可替换 → 策略
  动态增强 → 装饰器
  事件通知 → 观察者
  流程固定步骤可变 → 模板方法
  多对多解耦 → 中介者

Step 4：画出结构
  接口关系
  数据流向
  组件交互

Step 5：考虑扩展
  新增功能不改旧代码
  可测试（接口注入）
  并发安全

常见面试题：
  设计一个缓存系统 → 装饰器+策略
  设计一个限流器 → 策略+工厂
  设计一个通知系统 → 策略+观察者
  设计一个任务调度器 → 命令+Worker Pool
```

## 10. 设计模式实战面试速答？

**回答：**

```
Q: 通知系统用什么模式？
A: 策略(多渠道) + 观察者(事件触发)
   + 装饰器(重试/限流)

Q: 插件系统怎么设计？
A: 接口定义插件契约
   注册表管理插件实例
   工厂创建插件

Q: 规则引擎用什么模式？
A: 策略(不同规则) + 组合模式(AND/OR)
   + 责任链(规则链)

Q: 多级缓存怎么设计？
A: 装饰器模式层层包装
   L1(local) → L2(Redis) → DB
   上层miss回填

Q: 配置中心客户端？
A: 多来源策略(env/file/remote)
   观察者通知变更
   单例全局访问

Q: 日志系统设计？
A: With模式添加上下文字段
   装饰器增强(采样/过滤/格式化)

Q: 怎么避免过度设计？
A: 当前只有一种实现不建接口
   YAGNI 不预测未来需求
   先简单实现 有需求再重构
```
