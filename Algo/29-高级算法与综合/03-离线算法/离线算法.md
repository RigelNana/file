# 离线算法

> 离线算法可以提前读取所有查询，按最优顺序处理，常与排序、树状数组、并查集等结合实现高效求解。

---

## 1. 在线 vs 离线

```
在线算法 (Online):
  ┌───────┐     ┌────┐     ┌───────┐
  │ 查询1  │ ──→ │ 回答 │ ──→ │ 查询2  │ ──→ ...
  └───────┘     └────┘     └───────┘
  必须按顺序逐个处理，不能看到后面的查询

离线算法 (Offline):
  ┌───────┐ ┌───────┐ ┌───────┐
  │ 查询1  │ │ 查询2  │ │ 查询3  │  ← 先读取全部
  └───────┘ └───────┘ └───────┘
       │         │         │
       ▼ 排序/分组/分治     ▼
  ┌───────────────────────────┐
  │ 按最优顺序统一处理        │
  └───────────────────────────┘
       │         │         │
  ┌───────┐ ┌───────┐ ┌───────┐
  │ 答案1  │ │ 答案2  │ │ 答案3  │  ← 按原始顺序输出
  └───────┘ └───────┘ └───────┘

常见离线技巧:
  ┌────────────────────┬───────────────────────────────┐
  │ 技巧               │ 适用场景                      │
  ├────────────────────┼───────────────────────────────┤
  │ 排序 + 扫描        │ 按右端点/值排序，逐步添加     │
  │ 排序 + 树状数组    │ 区间内 ≤k 的元素个数         │
  │ 莫队               │ 无修改区间统计               │
  │ CDQ 分治           │ 多维偏序                     │
  │ 离线 + 并查集      │ 按边权排序，最小生成树变形    │
  │ 倒序处理           │ 删除操作 → 反转为添加操作     │
  └────────────────────┴───────────────────────────────┘
```

## 2. 排序 + 树状数组

### 模式一：按值排序，区间内 ≤ k 的元素个数

```python
class BIT:
    """树状数组（1-indexed）"""
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (n + 1)

    def update(self, i, delta=1):
        while i <= self.n:
            self.tree[i] += delta
            i += i & (-i)

    def query(self, i):
        s = 0
        while i > 0:
            s += self.tree[i]
            i -= i & (-i)
        return s

    def range_query(self, l, r):
        return self.query(r) - self.query(l - 1)


def offline_count_leq(nums, queries):
    """
    nums: 原数组 (0-indexed)
    queries: [(l, r, k), ...] 求 nums[l..r] 中 ≤ k 的元素个数
    返回: 每个查询的答案
    """
    n = len(nums)
    
    # ⚠️ 按 (值, 原下标) 排序原始元素
    indexed_nums = sorted(range(n), key=lambda i: nums[i])
    
    # ⚠️ 按 k 值排序查询
    sorted_queries = sorted(range(len(queries)), key=lambda qi: queries[qi][2])

    bit = BIT(n)
    ans = [0] * len(queries)
    j = 0  # 指向 indexed_nums 的指针

    for qi in sorted_queries:
        l, r, k = queries[qi]
        # 把所有 ≤ k 的元素添加到 BIT
        while j < n and nums[indexed_nums[j]] <= k:
            bit.update(indexed_nums[j] + 1, 1)  # ⚠️ BIT 是 1-indexed
            j += 1
        # 查询 [l, r] 范围内已添加的元素个数
        ans[qi] = bit.range_query(l + 1, r + 1)  # ⚠️ 转换为 1-indexed

    return ans
```

### ⚠️ 易错点

```
1. 索引转换:
   ⚠️ 原数组 0-indexed, BIT 是 1-indexed
   → update(idx + 1, ...) 和 range_query(l + 1, r + 1)

2. 排序后要保留原始索引:
   ⚠️ 不能只排序值，要知道每个值在原数组中的位置
   → 排序下标: sorted(range(n), key=lambda i: nums[i])

3. 查询排序后也要保留原始索引:
   ⚠️ 最终答案要按原始查询顺序输出
   → ans[qi] = ...
```

## 3. 模式二：按右端点排序

```python
# 经典问题: 求区间 [l, r] 内不同元素的个数
# 离线 + 按右端点排序 + BIT

def count_distinct_offline(nums, queries):
    """
    nums: 原数组 (0-indexed)
    queries: [(l, r), ...] 0-indexed 闭区间
    返回: 每个区间内不同元素的个数
    """
    n = len(nums)
    
    # 按右端点排序查询
    sorted_qi = sorted(range(len(queries)), key=lambda qi: queries[qi][1])

    bit = BIT(n)
    last_pos = {}  # 值 → 最后出现的位置 (1-indexed)
    ans = [0] * len(queries)
    j = 0  # 扫描到的位置

    for qi in sorted_qi:
        l, r = queries[qi]
        # 扫描到 r
        while j <= r:
            val = nums[j]
            if val in last_pos:
                # ⚠️ 同一个值之前出现过，删除旧位置的贡献
                bit.update(last_pos[val], -1)
            last_pos[val] = j + 1  # ⚠️ 1-indexed
            bit.update(j + 1, 1)
            j += 1
        # 查询 [l+1, r+1] (1-indexed)
        ans[qi] = bit.range_query(l + 1, r + 1)

    return ans
```

### 思路可视化

```
nums = [1, 2, 1, 3, 2]

扫描到 j=0: BIT 标记位置 1 (值1)   last_pos = {1: 1}
扫描到 j=1: BIT 标记位置 2 (值2)   last_pos = {1: 1, 2: 2}
扫描到 j=2: 值1 之前在位置1, 删除位置1, 标记位置3
             last_pos = {1: 3, 2: 2}
扫描到 j=3: BIT 标记位置 4 (值3)   last_pos = {1: 3, 2: 2, 3: 4}
扫描到 j=4: 值2 之前在位置2, 删除位置2, 标记位置5
             last_pos = {1: 3, 2: 5, 3: 4}

BIT 状态:  0  0  1  1  1   ← 只有每个值的"最后出现位置"被标记
位置:      1  2  3  4  5

查询 [0, 4] → range_query(1, 5) = 3 ✓ (1,2,3 三种值)
查询 [2, 4] → range_query(3, 5) = 3 ✓ (位置3是1, 位置4是3, 位置5是2)
查询 [0, 1] → ⚠️ 这个查询的 r=1, 必须在扫描到 j=1 时回答!
             所以按 r 排序很关键

关键洞察:
  ⚠️ 对于每个值，只保留最右边的出现位置
  → 查询 [l, r] 时，BIT 中 [l, r] 范围内的标记数
     恰好是区间内不同元素的个数
  → 因为每个值只在最右位置被计数一次
```

## 4. 模式三：倒序处理（删除变添加）

```python
# 问题: 每次删除一个节点，每次操作后求连通分量数
# 直接删除很难维护 → 反转为逆序添加，用并查集

class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
        self.components = 0

    def find(self, x):
        while self.parent[x] != x:
            self.parent[x] = self.parent[self.parent[x]]
            x = self.parent[x]
        return x

    def union(self, x, y):
        rx, ry = self.find(x), self.find(y)
        if rx == ry:
            return
        if self.rank[rx] < self.rank[ry]:
            rx, ry = ry, rx
        self.parent[ry] = rx
        if self.rank[rx] == self.rank[ry]:
            self.rank[rx] += 1
        self.components -= 1  # ⚠️ 合并成功，连通分量减 1

    def add_node(self):
        self.components += 1  # ⚠️ 新增一个孤立节点


def solve_delete_to_add(n, edges, delete_order):
    """
    n: 节点数
    edges: [(u, v), ...] 所有边
    delete_order: [node1, node2, ...] 删除顺序
    返回: 每次删除后的连通分量数 (倒序处理)
    """
    deleted = set(delete_order)
    # 找出不会被删除的节点和它们之间的边
    adj = [[] for _ in range(n)]
    for u, v in edges:
        adj[u].append(v)
        adj[v].append(u)

    uf = UnionFind(n)
    # 先添加不被删除的节点
    remaining = set(range(n)) - deleted
    for node in remaining:
        uf.add_node()
    for node in remaining:
        for nb in adj[node]:
            if nb in remaining:
                uf.union(node, nb)

    # ⚠️ 倒序添加被删除的节点
    results = []
    for node in reversed(delete_order):
        results.append(uf.components)
        uf.add_node()
        for nb in adj[node]:
            if nb not in deleted or delete_order.index(nb) > delete_order.index(node):
                # ⚠️ 简化: 用 set 跟踪已添加的节点
                pass
        # 实际实现中用 added 集合跟踪
    
    results.reverse()
    return results
```

### LeetCode 803 - 打砖块

```
题意: 
  grid m×n, 1=砖块 0=空
  按顺序击打 hits 中的砖块
  砖块不与顶部连通则会掉落
  求每次击打后掉落的砖块数

分析:
  正向删除很难维护连通性
  ⚠️ 关键技巧: 倒序处理！
    1. 先把所有要击打的砖块全部移除
    2. 从最后一次击打开始，逐个添加回来
    3. 每次添加后新增的"连通到顶部"的砖块数 = 本次掉落数
```

```python
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.size = [1] * n

    def find(self, x):
        while self.parent[x] != x:
            self.parent[x] = self.parent[self.parent[x]]
            x = self.parent[x]
        return x

    def union(self, x, y):
        rx, ry = self.find(x), self.find(y)
        if rx == ry:
            return
        if self.size[rx] < self.size[ry]:
            rx, ry = ry, rx
        self.parent[ry] = rx
        self.size[rx] += self.size[ry]


def hitBricks(grid, hits):
    m, n = len(grid), len(grid[0])
    ROOF = m * n  # 虚拟顶部节点
    uf = UnionFind(m * n + 1)

    # 标记被击打的位置
    grid_copy = [row[:] for row in grid]
    for r, c in hits:
        grid_copy[r][c] = 0  # ⚠️ 先全部移除

    # 对剩余砖块建立并查集
    def idx(r, c):
        return r * n + c

    for r in range(m):
        for c in range(n):
            if grid_copy[r][c] == 1:
                if r == 0:
                    uf.union(idx(r, c), ROOF)  # ⚠️ 第一行连接到虚拟顶部
                if r > 0 and grid_copy[r-1][c] == 1:
                    uf.union(idx(r, c), idx(r-1, c))
                if c > 0 and grid_copy[r][c-1] == 1:
                    uf.union(idx(r, c), idx(r, c-1))

    # ⚠️ 倒序添加砖块
    result = [0] * len(hits)
    dirs = [(0, 1), (0, -1), (1, 0), (-1, 0)]

    for i in range(len(hits) - 1, -1, -1):
        r, c = hits[i]
        if grid[r][c] == 0:  # ⚠️ 原来就没有砖块
            continue

        prev_roof = uf.size[uf.find(ROOF)]  # 添加前连通到顶部的砖块数
        grid_copy[r][c] = 1

        if r == 0:
            uf.union(idx(r, c), ROOF)
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < m and 0 <= nc < n and grid_copy[nr][nc] == 1:
                uf.union(idx(r, c), idx(nr, nc))

        curr_roof = uf.size[uf.find(ROOF)]
        # ⚠️ 新增连通到顶部的数量 - 1 (减去自己)
        result[i] = max(0, curr_roof - prev_roof - 1)

    return result
```

## 本节要点速查

```
✅ 离线 = 先读取所有查询，按最优顺序处理
✅ 按值排序: 逐步添加元素到 BIT，添加到 k 后立即回答
✅ 按右端点: 扫描到 r 时回答所有右端点为 r 的查询
✅ 区间不同元素: 每个值只保留最右位置，BIT 区间求和
✅ 倒序处理: 删除操作 → 反转为添加操作，用并查集维护
✅ 保留原索引: 排序后答案要按原始查询顺序输出
✅ BIT 1-indexed: 所有坐标 +1 转换
```
