# 线段树

> 线段树支持 O(log n) 的区间查询和区间修改，配合懒标记实现高效区间操作。

---

## 1. 核心思想

```
线段树结构:
  用完全二叉树表示区间, 每个节点对应一个子区间

              [1,8] sum=36
           /              \
      [1,4] sum=10       [5,8] sum=26
      /        \          /        \
   [1,2] s=3  [3,4] s=7  [5,6] s=5  [7,8] s=21
   / \         / \         / \         / \
  1   2       3   4       5   0       8  13

  存储: 用数组, 节点 o 的左子 = 2*o, 右子 = 2*o+1
  
  ⚠️ 数组大小: 4*n (最坏情况)

查询 sum[3,7]:
  递归分解:
    [1,8] → 左[1,4] 右[5,8]
    [1,4] → 只查右半 [3,4]
    [3,4] → 完全包含, 直接返回 7
    [5,8] → 左[5,6] 完全包含返5, 右[7,8]→只查[7,7]
  合并: 7 + 5 + 8 = 20

  ⚠️ 任意区间最多分解为 O(log n) 个节点

懒标记 (Lazy Propagation):
  区间修改 [l,r] += val:
    不一层层更新到叶子
    把 val 暂存在区间节点的 lazy 标记上
    查询/修改子区间时再 push_down 下传

  ⚠️ push_down: 将当前节点的 lazy 下传给左右子节点
  ⚠️ push_up: 用左右子节点的值更新当前节点
```

---

## 2. 代码模板 (区间加 + 区间和)

```python
class SegTree:
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (4 * n)          # ⚠️ 4倍空间
        self.lazy = [0] * (4 * n)
    
    def _push_up(self, o):
        """用子节点更新当前节点"""
        self.tree[o] = self.tree[2*o] + self.tree[2*o+1]
    
    def _push_down(self, o, l, r):
        """下传懒标记"""
        if self.lazy[o]:
            mid = (l + r) // 2
            lz = self.lazy[o]
            # 更新左子
            self.tree[2*o] += lz * (mid - l + 1)  # ⚠️ 乘以区间长度
            self.lazy[2*o] += lz
            # 更新右子
            self.tree[2*o+1] += lz * (r - mid)
            self.lazy[2*o+1] += lz
            # 清除当前懒标记
            self.lazy[o] = 0               # ⚠️ 必须清零!
    
    def build(self, nums, o=1, l=1, r=None):
        """从 nums 构建线段树"""
        if r is None:
            r = self.n
        if l == r:
            self.tree[o] = nums[l - 1]    # ⚠️ nums 是 0-indexed
            return
        mid = (l + r) // 2
        self.build(nums, 2*o, l, mid)
        self.build(nums, 2*o+1, mid+1, r)
        self._push_up(o)
    
    def update(self, ql, qr, val, o=1, l=1, r=None):
        """[ql, qr] 区间加 val"""
        if r is None:
            r = self.n
        if ql <= l and r <= qr:            # ⚠️ 完全包含
            self.tree[o] += val * (r - l + 1)
            self.lazy[o] += val
            return
        self._push_down(o, l, r)           # ⚠️ 先下传再递归!
        mid = (l + r) // 2
        if ql <= mid:                      # ⚠️ 与左子有交集
            self.update(ql, qr, val, 2*o, l, mid)
        if qr > mid:                       # ⚠️ 与右子有交集 (不用 elif!)
            self.update(ql, qr, val, 2*o+1, mid+1, r)
        self._push_up(o)                   # ⚠️ 递归后更新当前节点
    
    def query(self, ql, qr, o=1, l=1, r=None):
        """查询 [ql, qr] 区间和"""
        if r is None:
            r = self.n
        if ql <= l and r <= qr:            # ⚠️ 完全包含, 直接返回
            return self.tree[o]
        self._push_down(o, l, r)           # ⚠️ 先下传!
        mid = (l + r) // 2
        res = 0
        if ql <= mid:
            res += self.query(ql, qr, 2*o, l, mid)
        if qr > mid:
            res += self.query(ql, qr, 2*o+1, mid+1, r)
        return res

# ⚠️ 常见错误:
#  1. 忘记 push_down → 查询结果不含未下传的修改
#  2. 忘记 push_up → 父节点信息过期
#  3. ql <= mid 和 qr > mid 不用 elif → 区间可能跨 mid
#  4. 数组大小用 4*n, 不是 2*n
```

## 3. 使用示例

```python
# 创建大小为 5 的线段树
nums = [1, 3, 5, 7, 9]
seg = SegTree(5)
seg.build(nums)

# 查询 [2, 4] 的和
print(seg.query(2, 4))    # → 3+5+7 = 15

# 区间 [1, 3] 加 2
seg.update(1, 3, 2)

# 再次查询 [2, 4]
print(seg.query(2, 4))    # → 5+7+7 = 19  (3+2=5, 5+2=7, 7不变)

# ⚠️ 线段树用 1-indexed
#    nums[0]=1 对应 位置 1
```

## 4. 区间最值线段树

```python
class SegTreeMax:
    """区间最大值 + 单点修改"""
    def __init__(self, n):
        self.n = n
        self.tree = [0] * (4 * n)
    
    def _push_up(self, o):
        self.tree[o] = max(self.tree[2*o], self.tree[2*o+1])
    
    def build(self, nums, o=1, l=1, r=None):
        if r is None: r = self.n
        if l == r:
            self.tree[o] = nums[l-1]
            return
        mid = (l+r)//2
        self.build(nums, 2*o, l, mid)
        self.build(nums, 2*o+1, mid+1, r)
        self._push_up(o)
    
    def update(self, pos, val, o=1, l=1, r=None):
        """单点修改 a[pos] = val"""
        if r is None: r = self.n
        if l == r:
            self.tree[o] = val
            return
        mid = (l+r)//2
        if pos <= mid:
            self.update(pos, val, 2*o, l, mid)
        else:
            self.update(pos, val, 2*o+1, mid+1, r)
        self._push_up(o)
    
    def query(self, ql, qr, o=1, l=1, r=None):
        """查询 [ql, qr] 最大值"""
        if r is None: r = self.n
        if ql <= l and r <= qr:
            return self.tree[o]
        mid = (l+r)//2
        res = float('-inf')
        if ql <= mid:
            res = max(res, self.query(ql, qr, 2*o, l, mid))
        if qr > mid:
            res = max(res, self.query(ql, qr, 2*o+1, mid+1, r))
        return res

# ⚠️ 单点修改不需要 lazy 标记
#    区间修改最值需要 "Segment Tree Beats" (进阶)
```

## 5. LeetCode 307: 区域和检索 - 数组可修改

```python
class NumArray:
    def __init__(self, nums):
        self.n = len(nums)
        self.seg = SegTree(self.n)
        self.seg.build(nums)
    
    def update(self, index, val):
        # ⚠️ 单点赋值 → 先算差值, 再区间加
        # 或者写一个单点赋值的线段树函数
        old = self.seg.query(index+1, index+1)
        self.seg.update(index+1, index+1, val - old)
    
    def sumRange(self, left, right):
        return self.seg.query(left+1, right+1)
        # ⚠️ 0-indexed → 1-indexed: +1
```

## 本节要点速查

```
✅ 数组大小 4*n, 节点 o 的子节点: 2*o, 2*o+1
✅ push_up: 子→父 (递归回来后)
✅ push_down: 父→子 (递归进去前)
✅ 完全包含 → 直接操作, 否则递归左右子
✅ ql<=mid 和 qr>mid 不用 elif (区间可能跨界)
✅ 懒标记必须清零, 否则重复下传
✅ 1-indexed, 题目 0-indexed 要 +1
```
