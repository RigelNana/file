# 树与树形算法

> 树是无环连通图，广泛用于层级结构表示。本章覆盖一般树上的 DFS/BFS、树形 DP、LCA 等。

---

## 1. 一般树的 DFS ⭐

### 理解与可视化

```
一般树（非二叉树）: 每个节点可以有任意多个子节点
通常用邻接表表示, 通过 parent 避免往回走

  例: 以 0 为根
       0
      / \
     1   2
    / \   \
   3   4   5

  邻接表:
    0 → [1, 2]
    1 → [0, 3, 4]
    2 → [0, 5]
    3 → [1]
    4 → [1]
    5 → [2]

  DFS 遍历（跳过 parent）:
    dfs(0, -1) → dfs(1, 0) → dfs(3, 1), dfs(4, 1)
                  dfs(2, 0) → dfs(5, 2)
```

### 代码模板

```python
from collections import defaultdict

# 一般树的 DFS 模板（邻接表 + parent 去重）
def tree_dfs(graph, root):
    def dfs(node, parent):
        # 处理当前节点
        for child in graph[node]:
            if child != parent:
                dfs(child, parent=node)
    dfs(root, -1)

# 求树的直径（最长路径）
def tree_diameter(n, edges):
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v)
        graph[v].append(u)

    # 两次 BFS
    from collections import deque
    def bfs(start):
        dist = [-1] * n
        dist[start] = 0
        queue = deque([start])
        farthest = start
        while queue:
            node = queue.popleft()
            for nei in graph[node]:
                if dist[nei] == -1:
                    dist[nei] = dist[node] + 1
                    queue.append(nei)
                    if dist[nei] > dist[farthest]:
                        farthest = nei
        return farthest, dist[farthest]

    far1, _ = bfs(0)
    far2, diameter = bfs(far1)
    return diameter

# 或者用 DFS 求直径
def tree_diameter_dfs(n, edges):
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v)
        graph[v].append(u)

    ans = 0
    def dfs(node, parent):
        nonlocal ans
        max1 = max2 = 0  # 最长 & 次长子树深度
        for child in graph[node]:
            if child == parent:
                continue
            d = dfs(child, node) + 1
            if d >= max1:
                max2, max1 = max1, d
            elif d > max2:
                max2 = d
        ans = max(ans, max1 + max2)
        return max1
    dfs(0, -1)
    return ans
```

## 2. 树形 DP

### 理解与可视化

```
树形 DP: 在树上做 DP, 利用子树的答案推出父节点的答案

典型模式:
  1. 自底向上: dfs 返回子树信息, 在回溯时合并
  2. 换根 DP: 先求某一根的答案, 再 O(n) 换根

自底向上:
  dfs(node) → 遍历所有子节点 → 合并返回值 → return 结果

换根 DP:
  第一遍 dfs: 以 0 为根, 求每个节点的子树答案
  第二遍 dfs: 从 0 出发, 将答案从父转移到子
```

### 代码模板

```python
# 打家劫舍III（树形DP）
class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

def rob_tree(root):
    def dfs(node):
        if not node:
            return 0, 0  # (选当前节点, 不选当前节点)
        l_rob, l_skip = dfs(node.left)
        r_rob, r_skip = dfs(node.right)
        rob = node.val + l_skip + r_skip
        skip = max(l_rob, l_skip) + max(r_rob, r_skip)
        return rob, skip
    return max(dfs(root))

# 换根 DP 模板（求所有节点到其他所有节点的距离之和）
def sum_of_distances(n, edges):
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v)
        graph[v].append(u)

    count = [1] * n   # 子树大小
    dist = [0] * n    # 子树内距离和

    # 第一遍: 以 0 为根, 求子树信息
    def dfs1(node, parent):
        for child in graph[node]:
            if child != parent:
                dfs1(child, node)
                count[node] += count[child]
                dist[node] += dist[child] + count[child]

    # 第二遍: 换根
    def dfs2(node, parent):
        for child in graph[node]:
            if child != parent:
                # 从 node 换到 child:
                # child 子树内的点距离各减 1 (共 count[child] 个)
                # 子树外的点距离各加 1 (共 n - count[child] 个)
                dist[child] = dist[node] + n - 2 * count[child]
                dfs2(child, node)

    dfs1(0, -1)
    dfs2(0, -1)
    return dist
```

## 3. LCA（最近公共祖先）

### 代码模板

```python
# 倍增法 LCA (预处理 O(n log n), 查询 O(log n))
import math

def build_lca(n, edges, root=0):
    LOG = max(1, int(math.log2(n)) + 1)
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v)
        graph[v].append(u)

    depth = [0] * n
    parent = [[-1] * n for _ in range(LOG)]

    # BFS 预处理
    from collections import deque
    queue = deque([root])
    visited = [False] * n
    visited[root] = True
    while queue:
        node = queue.popleft()
        for nei in graph[node]:
            if not visited[nei]:
                visited[nei] = True
                depth[nei] = depth[node] + 1
                parent[0][nei] = node
                queue.append(nei)

    # 倍增
    for k in range(1, LOG):
        for v in range(n):
            if parent[k-1][v] != -1:
                parent[k][v] = parent[k-1][parent[k-1][v]]

    def lca(u, v):
        if depth[u] < depth[v]:
            u, v = v, u
        diff = depth[u] - depth[v]
        for k in range(LOG):
            if (diff >> k) & 1:
                u = parent[k][u]
        if u == v:
            return u
        for k in range(LOG - 1, -1, -1):
            if parent[k][u] != parent[k][v]:
                u = parent[k][u]
                v = parent[k][v]
        return parent[0][u]

    return lca
```
