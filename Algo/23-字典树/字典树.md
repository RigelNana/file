# 字典树

> 字典树（Trie）是专门处理字符串前缀匹配的树形数据结构，在自动补全、拼写检查、IP 路由等场景广泛使用。

---

## 1. 字典树基础 ⭐

### 理解与可视化

```
Trie: 每个节点代表一个字符前缀

  插入: "app", "apple", "bat"

            root
           /    \
          a      b
          |      |
          p      a
          |      |
          p*     t*
          |
          l
          |
          e*

  * 标记单词结尾

  特点:
    - 公共前缀只存一次 → 节省空间
    - 查找/插入 O(L), L = 字符串长度
    - 前缀匹配极快
```

### 代码模板

```python
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word):
        node = self.root
        for ch in word:
            if ch not in node.children:
                node.children[ch] = TrieNode()
            node = node.children[ch]
        node.is_end = True

    def search(self, word):
        node = self._find(word)
        return node is not None and node.is_end

    def starts_with(self, prefix):
        return self._find(prefix) is not None

    def _find(self, prefix):
        node = self.root
        for ch in prefix:
            if ch not in node.children:
                return None
            node = node.children[ch]
        return node
```

## 2. 字典树进阶

### 代码模板

```python
# 单词搜索 II (Trie + 回溯)
def find_words(board, words):
    trie = Trie()
    for w in words:
        trie.insert(w)

    m, n = len(board), len(board[0])
    result = set()

    def dfs(i, j, node, path):
        if node.is_end:
            result.add(path)
            node.is_end = False  # 剪枝: 找到后不重复
        if i < 0 or i >= m or j < 0 or j >= n:
            return
        ch = board[i][j]
        if ch not in node.children:
            return
        board[i][j] = '#'  # 标记访问
        for di, dj in [(-1,0),(1,0),(0,-1),(0,1)]:
            dfs(i+di, j+dj, node.children[ch], path + ch)
        board[i][j] = ch   # 恢复

    for i in range(m):
        for j in range(n):
            dfs(i, j, trie.root, "")
    return list(result)

# 数组中两个数的最大异或值 (0-1 Trie)
class BitTrie:
    def __init__(self):
        self.root = {}

    def insert(self, num):
        node = self.root
        for i in range(30, -1, -1):
            bit = (num >> i) & 1
            if bit not in node:
                node[bit] = {}
            node = node[bit]

    def query(self, num):
        """查找与 num 异或最大的值"""
        node = self.root
        xor_val = 0
        for i in range(30, -1, -1):
            bit = (num >> i) & 1
            want = 1 - bit
            if want in node:
                xor_val |= (1 << i)
                node = node[want]
            else:
                node = node[bit]
        return xor_val

def find_maximum_xor(nums):
    trie = BitTrie()
    for x in nums:
        trie.insert(x)
    return max(trie.query(x) for x in nums)
```
