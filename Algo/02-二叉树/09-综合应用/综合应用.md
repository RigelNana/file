## 1. 二叉树展开为链表 ⭐

### LeetCode 114

```
题意: 把二叉树按前序遍历顺序展开为"右倾链表"
  
       1              1
      / \              \
     2   5     →        2
    / \   \              \
   3   4   6              3
                           \
                            4
                             \
                              5
                               \
                                6
```

```python
# ━━━━ 方法 1: 后序处理（技巧性）━━━━
def flatten(root):
    prev = None                       # ⚠️ 反前序(右左根), prev 记上一个
    def dfs(node):
        nonlocal prev
        if not node:
            return
        dfs(node.right)               # ⚠️ 先处理右!
        dfs(node.left)                # 再处理左
        node.right = prev             # 当前的 right 指向 prev
        node.left = None              # 左边置空
        prev = node                   # 更新 prev
    dfs(root)

# ⚠️ 遍历顺序: 右→左→根 (前序的完全反转)
#   6→5→4→3→2→1 的顺序处理
#   每个节点的 right 指向它在前序中的下一个（prev）
#   巧妙避免了丢失右子树的问题

# ━━━━ 方法 2: 前序遍历 + 拉平 ━━━━
def flatten_v2(root):
    cur = root
    while cur:
        if cur.left:
            # 找左子树的最右节点
            rightmost = cur.left
            while rightmost.right:
                rightmost = rightmost.right
            # 把右子树接到左子树的最右节点
            rightmost.right = cur.right
            # 左子树变成右子树
            cur.right = cur.left
            cur.left = None
        cur = cur.right               # 移到下一个
```

## 2. 路径总和 III

### LeetCode 437

```
题意: 找出二叉树中路径和等于 targetSum 的路径数量
     路径不需要从根开始, 也不需要到叶子结束
     但方向必须从上到下
```

```python
# ━━━━ 方法: 前缀和 ━━━━
from collections import defaultdict

def pathSum(root, targetSum):
    prefix = defaultdict(int)
    prefix[0] = 1                     # ⚠️ 空路径前缀和=0, 计数1
    ans = 0
    
    def dfs(node, cur_sum):
        nonlocal ans
        if not node:
            return
        cur_sum += node.val
        ans += prefix[cur_sum - targetSum]  # ⚠️ 找是否存在前缀和差=target
        prefix[cur_sum] += 1                # 记录当前前缀和
        dfs(node.left, cur_sum)
        dfs(node.right, cur_sum)
        prefix[cur_sum] -= 1                # ⚠️ 回溯! 离开时减掉
    
    dfs(root, 0)
    return ans

# ⚠️ 前缀和思想:
#   从根到当前节点的路径和 = cur_sum
#   如果之前某个祖先的前缀和 = cur_sum - target
#   那么从那个祖先到当前节点的路径和 = target
#
# ⚠️ 回溯 prefix[cur_sum] -= 1 很重要!
#   离开当前节点时, 当前前缀和不再有效
#   如果不回溯, 其他分支会错误地使用这个前缀和
#
# ⚠️ prefix[0] = 1 的含义:
#   从根到当前节点的整条路径的和 = target 时
#   cur_sum - target = 0, 需要 prefix[0]=1 才能计数
```

## 3. 打家劫舍 III

### LeetCode 337

```
题意: 小偷不能同时偷直接相连的节点, 求最大偷窃金额

思路: 对每个节点, 两种选择:
  选: 偷当前节点 + 不偷左右孩子(取孩子的"不选"值)
  不选: 不偷当前节点 + 左右孩子各自取最大值
```

```python
def rob(root):
    def dfs(node):
        if not node:
            return 0, 0              # (选, 不选) 都是 0
        
        l_rob, l_skip = dfs(node.left)
        r_rob, r_skip = dfs(node.right)
        
        rob_this = node.val + l_skip + r_skip   # ⚠️ 选当前 + 不选孩子
        skip_this = max(l_rob, l_skip) + max(r_rob, r_skip)  # ⚠️ 不选当前, 孩子随便
        
        return rob_this, skip_this
    
    return max(dfs(root))

# ⚠️ 返回元组 (选, 不选) 同时返回两种状态
# ⚠️ skip_this 取 max(rob, skip): 不选当前时, 孩子可选可不选
# ⚠️ 时间 O(n), 每个节点只访问一次
```

## 4. 完全二叉树的节点个数

### LeetCode 222

```
完全二叉树特性: 最后一层可能不满, 但从左到右连续

利用: 完全二叉树要么是满二叉树, 要么左或右子树是满的
     满二叉树节点数 = 2^h - 1
```

```python
def countNodes(root):
    if not root:
        return 0
    
    left_h = right_h = 0
    left, right = root, root
    while left:                       # 左边高度
        left_h += 1
        left = left.left
    while right:                      # 沿右边走的高度
        right_h += 1
        right = right.right
    
    if left_h == right_h:             # ⚠️ 是满二叉树
        return (1 << left_h) - 1      # 2^h - 1
    
    return 1 + countNodes(root.left) + countNodes(root.right)

# ⚠️ 时间 O(log²n):
#   每层递归计算高度 O(log n)
#   最多递归 log n 层（每次只有一侧需要递归）
#   总共 O(log²n), 比朴素 O(n) 快很多
#
# ⚠️ 1 << left_h = 2^left_h
```

## 5. 二叉树的最大宽度

### LeetCode 662

```
题意: 每层的宽度定义为最左和最右非空节点之间的长度(包括空节点)

思路: BFS + 给每个节点编号
     根=1, 左孩子=2*i, 右孩子=2*i+1
     每层宽度 = 最右编号 - 最左编号 + 1
```

```python
from collections import deque

def widthOfBinaryTree(root):
    if not root:
        return 0
    queue = deque([(root, 0)])        # (节点, 编号)
    max_width = 0
    while queue:
        size = len(queue)
        _, first_idx = queue[0]       # ⚠️ 当前层第一个编号
        for _ in range(size):
            node, idx = queue.popleft()
            idx -= first_idx          # ⚠️ 防溢出: 减去当前层起始编号
            if node.left:
                queue.append((node.left, 2 * idx))
            if node.right:
                queue.append((node.right, 2 * idx + 1))
        max_width = max(max_width, idx - 0 + 1)
                                      # ⚠️ idx 已经减过 first_idx
                                      # 最后一个的相对编号 + 1 = 宽度
    return max_width

# ⚠️ 防溢出: idx -= first_idx
#   完全二叉树深度大时, 编号可能非常大(2^n)
#   每层减去起始编号, 保持数值较小
```

## 6. 二叉树中所有距离为 K 的节点

### LeetCode 863

```
思路: 
  1. 先用 DFS 建立父指针(parent map)
  2. 再从 target 出发 BFS, 走 K 步
     BFS 时可以往左、右、父三个方向走
```

```python
from collections import deque

def distanceK(root, target, k):
    # 1. 建立父节点映射
    parent = {}
    def dfs(node, par):
        if not node:
            return
        parent[node] = par
        dfs(node.left, node)
        dfs(node.right, node)
    dfs(root, None)
    
    # 2. 从 target 开始 BFS
    queue = deque([target])
    visited = {target}
    dist = 0
    while queue:
        if dist == k:
            return [node.val for node in queue]
        dist += 1
        for _ in range(len(queue)):
            node = queue.popleft()
            for neighbor in [node.left, node.right, parent[node]]:
                if neighbor and neighbor not in visited:
                    visited.add(neighbor)
                    queue.append(neighbor)
    return []
```
