# 网格图

> 网格图 (Grid) 是二维数组上的搜索问题，常用 DFS 和 BFS 解决。岛屿问题、迷宫最短路等都是经典题型。

---

## 1. 网格图 DFS ⭐

### 理解与可视化

```
网格图就是二维矩阵，每个格子可以向上下左右四个方向移动

  1 1 0 0 0
  1 1 0 0 0       1 表示陆地，0 表示水
  0 0 1 0 0       相连的 1 组成一个岛屿
  0 0 0 1 1       这个图有 3 个岛屿

DFS 洪水填充法（Flood Fill）:
  从一个 1 出发，把所有相连的 1 都标记为已访问
  每次启动一次新的 DFS = 发现一个新岛屿

  方向数组: [(0,1), (0,-1), (1,0), (-1,0)]
            右      左      下      上
```

### 代码模板

```python
# 岛屿数量（⭐ 必会）
def num_islands(grid):
    if not grid:
        return 0
    rows, cols = len(grid), len(grid[0])
    count = 0

    def dfs(r, c):
        if r < 0 or r >= rows or c < 0 or c >= cols or grid[r][c] != '1':
            return
        grid[r][c] = '0'  # 标记已访问
        dfs(r + 1, c)
        dfs(r - 1, c)
        dfs(r, c + 1)
        dfs(r, c - 1)

    for i in range(rows):
        for j in range(cols):
            if grid[i][j] == '1':
                dfs(i, j)
                count += 1
    return count

# 岛屿的最大面积
def max_area_of_island(grid):
    rows, cols = len(grid), len(grid[0])

    def dfs(r, c):
        if r < 0 or r >= rows or c < 0 or c >= cols or grid[r][c] != 1:
            return 0
        grid[r][c] = 0
        return 1 + dfs(r+1,c) + dfs(r-1,c) + dfs(r,c+1) + dfs(r,c-1)

    return max((dfs(i, j) for i in range(rows) for j in range(cols) if grid[i][j] == 1), default=0)

# 被围绕的区域（边界 O 不变，内部 O 变 X）
def solve(board):
    if not board:
        return
    rows, cols = len(board), len(board[0])

    def dfs(r, c):
        if r < 0 or r >= rows or c < 0 or c >= cols or board[r][c] != 'O':
            return
        board[r][c] = 'S'  # 标记为安全
        dfs(r+1,c); dfs(r-1,c); dfs(r,c+1); dfs(r,c-1)

    # 从边界的 O 开始 DFS
    for i in range(rows):
        dfs(i, 0); dfs(i, cols - 1)
    for j in range(cols):
        dfs(0, j); dfs(rows - 1, j)

    for i in range(rows):
        for j in range(cols):
            if board[i][j] == 'O':
                board[i][j] = 'X'
            elif board[i][j] == 'S':
                board[i][j] = 'O'
```

## 2. 网格图 BFS ⭐

### 理解与可视化

```
BFS 在网格图中的核心用途: 求最短路径

  从起点开始，一层一层向外扩展（像水波纹）
  第一次到达终点时经过的层数 = 最短距离

  S . . . .          S 0 1 2 3
  # # . # .    →     # # 2 # 4     数字=到 S 的距离
  . . . . .          4 3 3 4 5
  . # # # .          5 # # # 6
  . . . . E          6 7 8 7 E=8

BFS 保证第一次到达的就是最短路径（边权为1时）
```

### 代码模板

```python
from collections import deque

# 网格图 BFS 最短路模板
def shortest_path(grid, start, end):
    rows, cols = len(grid), len(grid[0])
    sr, sc = start
    er, ec = end
    queue = deque([(sr, sc, 0)])  # (row, col, distance)
    visited = {(sr, sc)}
    dirs = [(0,1), (0,-1), (1,0), (-1,0)]

    while queue:
        r, c, d = queue.popleft()
        if (r, c) == (er, ec):
            return d
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols and (nr, nc) not in visited and grid[nr][nc] != 1:
                visited.add((nr, nc))
                queue.append((nr, nc, d + 1))
    return -1

# 腐烂的橘子（多源 BFS）
def oranges_rotting(grid):
    rows, cols = len(grid), len(grid[0])
    queue = deque()
    fresh = 0

    for i in range(rows):
        for j in range(cols):
            if grid[i][j] == 2:
                queue.append((i, j))  # 所有腐烂橘子同时开始
            elif grid[i][j] == 1:
                fresh += 1

    if fresh == 0:
        return 0

    minutes = 0
    dirs = [(0,1), (0,-1), (1,0), (-1,0)]
    while queue:
        minutes += 1
        for _ in range(len(queue)):
            r, c = queue.popleft()
            for dr, dc in dirs:
                nr, nc = r + dr, c + dc
                if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == 1:
                    grid[nr][nc] = 2
                    fresh -= 1
                    queue.append((nr, nc))
    return minutes - 1 if fresh == 0 else -1

# 01 矩阵（每个 1 到最近 0 的距离）
def update_matrix(mat):
    rows, cols = len(mat), len(mat[0])
    queue = deque()
    for i in range(rows):
        for j in range(cols):
            if mat[i][j] == 0:
                queue.append((i, j))
            else:
                mat[i][j] = float('inf')

    dirs = [(0,1), (0,-1), (1,0), (-1,0)]
    while queue:
        r, c = queue.popleft()
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols and mat[nr][nc] > mat[r][c] + 1:
                mat[nr][nc] = mat[r][c] + 1
                queue.append((nr, nc))
    return mat
```

## 3. 网格图 0-1 BFS

### 理解与可视化

```
0-1 BFS: 边权只有 0 和 1 的最短路

  普通 BFS 假设所有边权为 1
  当有些边权为 0 时，用双端队列:
    权为 0 的边 → 加入队首 (appendleft)
    权为 1 的边 → 加入队尾 (append)

  这样保证队列始终按距离有序，O(V+E) 复杂度
```

### 代码模板

```python
from collections import deque

# 0-1 BFS 模板
def zero_one_bfs(grid):
    rows, cols = len(grid), len(grid[0])
    dist = [[float('inf')] * cols for _ in range(rows)]
    dist[0][0] = 0
    dq = deque([(0, 0)])
    dirs = [(0,1), (0,-1), (1,0), (-1,0)]

    while dq:
        r, c = dq.popleft()
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols:
                w = grid[nr][nc]  # 0 或 1
                if dist[r][c] + w < dist[nr][nc]:
                    dist[nr][nc] = dist[r][c] + w
                    if w == 0:
                        dq.appendleft((nr, nc))
                    else:
                        dq.append((nr, nc))
    return dist[rows-1][cols-1]
```

## 4. 网格图总结

```
选择哪种方法？

  ┌──────────────────────┬──────────────────────┐
  │ 场景                  │ 方法                  │
  ├──────────────────────┼──────────────────────┤
  │ 连通块/岛屿计数       │ DFS（洪水填充）       │
  │ 最短路径（无权）       │ BFS                  │
  │ 多源最短路            │ 多源 BFS              │
  │ 0-1 权边最短路        │ 0-1 BFS（双端队列）   │
  │ 带权最短路            │ Dijkstra             │
  │ 路径是否存在          │ DFS 或 BFS           │
  └──────────────────────┴──────────────────────┘
```
