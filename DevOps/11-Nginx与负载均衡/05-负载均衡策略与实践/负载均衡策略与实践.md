# 负载均衡策略与实践

---

## 1. 负载均衡核心概念？

**回答：**

```
负载均衡 (Load Balancing):
  将请求分发到多个后端服务器, 提高可用性和吞吐量

分层:
  ┌──────────┬──────────────┬───────────────────────┐
  │ 层次      │ 工作层级      │ 代表工具               │
  ├──────────┼──────────────┼───────────────────────┤
  │ DNS      │ DNS 解析层    │ Route53, CloudFlare    │
  │ L4 (四层) │ 传输层 TCP   │ LVS, HAProxy, NLB      │
  │ L7 (七层) │ 应用层 HTTP  │ Nginx, Envoy, ALB      │
  └──────────┴──────────────┴───────────────────────┘

  DNS 轮询: 简单但不灵活, 无健康检查
  L4 负载: 高性能, 只看 IP/Port, 无法按 URL 路由
  L7 负载: 功能丰富, 按 URL/Header 路由, 但性能略低
```

## 2. Nginx upstream 所有负载均衡算法详解？

**回答：**

```nginx
# ============ 1. 轮询 Round Robin (默认) ============
upstream backend {
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
    server 10.0.1.103:8080;
}
# 按顺序逐个分发: 101 → 102 → 103 → 101 → ...
# 适合: 后端服务器性能相近

# ============ 2. 加权轮询 Weighted Round Robin ============
upstream backend {
    server 10.0.1.101:8080 weight=5;    # 5/8 请求
    server 10.0.1.102:8080 weight=2;    # 2/8 请求
    server 10.0.1.103:8080 weight=1;    # 1/8 请求
}
# 适合: 服务器性能不同 (高配机器权重大)

# ============ 3. IP Hash ============
upstream backend {
    ip_hash;
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
    server 10.0.1.103:8080;
}
# 同一客户端 IP 总是分到同一台后端
# 适合: Session 保持 (但不如真正的 sticky session)
# 缺点: NAT 后多用户共享 IP → 负载不均

# ============ 4. 最少连接 Least Connections ============
upstream backend {
    least_conn;
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
}
# 分发到当前活跃连接最少的服务器
# 适合: 请求处理时间差异大 (长短请求混合)
# 可结合 weight: 加权最少连接

# ============ 5. 通用 Hash ============
upstream backend {
    hash $request_uri consistent;
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
}
# 自定义 hash key, 相同 key → 同一服务器
# consistent: 一致性哈希 (增减节点只影响部分请求)
# 适合: 缓存场景 (相同 URL → 同一后端, 提高缓存命中)

# ============ 6. Random ============
upstream backend {
    random two least_conn;
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
    server 10.0.1.103:8080;
}
# random: 随机选择
# random two least_conn: 随机选 2 台, 取连接少的 (Power of Two Choices)
# 适合: 多层负载均衡 (避免同步问题)
```

## 3. upstream 服务器参数详解？

**回答：**

```nginx
upstream backend {
    # ============ 基础参数 ============
    server 10.0.1.101:8080 weight=3;
    # weight: 权重 (默认 1)

    # ============ 健康检查参数 ============
    server 10.0.1.102:8080 max_fails=3 fail_timeout=30s;
    # max_fails:    在 fail_timeout 时间内最大失败次数
    # fail_timeout: 两个作用:
    #   1. 统计失败次数的时间窗口
    #   2. 标记不可用后的暂停时间

    # ============ 备用服务器 ============
    server 10.0.1.103:8080 backup;
    # 只有当所有非 backup 服务器不可用时才启用
    # 注意: ip_hash 不支持 backup

    # ============ 标记不可用 ============
    server 10.0.1.104:8080 down;
    # 永久标记为不可用 (维护时用)

    # ============ 连接限制 ============
    server 10.0.1.105:8080 max_conns=100;
    # 到该服务器的最大并发连接数
    # 超过限制 → 请求排队或分发到其他服务器

    # ============ 慢启动 (Nginx Plus) ============
    # server 10.0.1.106:8080 slow_start=30s;
    # 新加入/恢复的服务器权重从 0 逐渐增到正常值
    # 防止瞬间大量请求打到刚恢复的服务器

    # ============ DNS 解析 ============
    server backend.example.com resolve;
    # 动态 DNS 解析 (需要 resolver 指令)
    # 适合: 服务发现, 后端 IP 会变化

    resolver 10.0.0.2 valid=30s;
    # DNS 服务器地址, 缓存有效期
}

# ============ 完整示例 ============
upstream production {
    least_conn;

    server 10.0.1.101:8080 weight=5 max_fails=3 fail_timeout=30s max_conns=200;
    server 10.0.1.102:8080 weight=5 max_fails=3 fail_timeout=30s max_conns=200;
    server 10.0.1.103:8080 weight=3 max_fails=3 fail_timeout=30s max_conns=100;
    server 10.0.1.199:8080 backup;

    keepalive 32;     # 保持 32 个到后端的空闲连接
    keepalive_timeout 60s;
    keepalive_requests 1000;
}
```

## 4. 会话保持 (Session Persistence)？

**回答：**

```
问题: 轮询分发时, 用户 Session 在 Server A, 下次请求到 Server B → Session 丢失

解决方案:
  ┌──────────────────┬──────────────┬──────────────────┐
  │ 方案              │ 实现方式      │ 优缺点            │
  ├──────────────────┼──────────────┼──────────────────┤
  │ Session 共享      │ Redis 存储   │ ✓最佳方案, 推荐   │
  │ IP Hash          │ ip_hash      │ NAT 后负载不均    │
  │ Cookie Hash      │ hash $cookie │ 需要特定 cookie   │
  │ Sticky Cookie    │ Nginx Plus   │ 商业version      │
  │ URL Hash         │ hash $uri    │ 特定场景          │
  └──────────────────┴──────────────┴──────────────────┘
```

```nginx
# ============ 方案 1: Session 共享 (最佳) ============
# 后端使用 Redis/Memcached 存储 Session
# Nginx 不需要特殊配置, 任何负载均衡算法都行
upstream backend {
    least_conn;
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
}

# ============ 方案 2: IP Hash ============
upstream backend {
    ip_hash;
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
}
# 缺点: NAT/代理 后多用户 → 同一后端 → 不均衡

# ============ 方案 3: Cookie Hash ============
upstream backend {
    hash $cookie_jsessionid consistent;
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
}
# 根据 Session Cookie 值做 Hash
# consistent → 一致性哈希, 增减节点影响小

# ============ 方案 4: Sticky Cookie (Nginx Plus) ============
# upstream backend {
#     server 10.0.1.101:8080;
#     server 10.0.1.102:8080;
#     sticky cookie srv_id expires=1h domain=.example.com path=/;
# }
# Nginx 自动设置 Cookie 标识后端服务器

# ============ 方案 5: OpenResty Sticky ============
# 开源方案通过 Lua 实现 sticky session
# 或使用 nginx-sticky-module 第三方模块
```

## 5. 四层负载均衡 (Stream)？

**回答：**

```nginx
# ============ stream 模块: TCP/UDP 四层负载均衡 ============
# 需要编译 --with-stream (大多数发行版默认包含)

# 在 nginx.conf 中, 与 http 同级
stream {
    # ============ TCP 代理: MySQL ============
    upstream mysql_cluster {
        least_conn;
        server 10.0.1.101:3306 max_fails=3 fail_timeout=30s;
        server 10.0.1.102:3306 max_fails=3 fail_timeout=30s;
        server 10.0.1.103:3306 backup;
    }

    server {
        listen 3306;
        proxy_pass mysql_cluster;
        proxy_connect_timeout 3s;
        proxy_timeout 300s;  # 空闲超时
    }

    # ============ TCP 代理: Redis ============
    upstream redis_cluster {
        server 10.0.1.101:6379;
        server 10.0.1.102:6379;
    }

    server {
        listen 6379;
        proxy_pass redis_cluster;
        proxy_connect_timeout 1s;
        proxy_timeout 60s;
    }

    # ============ UDP 代理: DNS ============
    upstream dns_servers {
        server 10.0.1.10:53;
        server 10.0.1.11:53;
    }

    server {
        listen 53 udp;
        proxy_pass dns_servers;
        proxy_timeout 5s;
        proxy_responses 1;   # UDP: 期望的响应包数
    }

    # ============ TCP SSL 终止 ============
    server {
        listen 3307 ssl;
        ssl_certificate     /etc/ssl/mysql.pem;
        ssl_certificate_key /etc/ssl/mysql.key;
        proxy_pass mysql_cluster;
    }

    # ============ 访问日志 ============
    log_format stream_log '$remote_addr [$time_local] '
                          '$protocol $status $bytes_sent '
                          '$session_time "$upstream_addr"';
    access_log /var/log/nginx/stream.log stream_log;
}
```

## 6. 四层 vs 七层负载均衡对比？

**回答：**

```
四层 (L4) 负载均衡:
  ┌─────────────────────────────────┐
  │ 工作在传输层 (TCP/UDP)            │
  │ 基于: 源 IP, 目标 IP, 源端口,    │
  │       目标端口                    │
  │ 不解析应用层内容                  │
  └─────────────────────────────────┘

七层 (L7) 负载均衡:
  ┌─────────────────────────────────┐
  │ 工作在应用层 (HTTP/HTTPS)         │
  │ 基于: URL, Header, Cookie,       │
  │       请求方法, 请求体            │
  │ 需要解析 HTTP 协议                │
  └─────────────────────────────────┘

对比:
  ┌──────────────────┬─────────────┬─────────────────┐
  │ 维度              │ L4           │ L7               │
  ├──────────────────┼─────────────┼─────────────────┤
  │ 性能              │ 极高         │ 高               │
  │ 路由灵活性        │ 低 (IP/Port) │ 高 (URL/Header)  │
  │ SSL 终止         │ 可以         │ 可以             │
  │ 应用感知         │ 无           │ 完整             │
  │ 内容路由         │ ✗            │ ✓ (按 URL 路由)  │
  │ URL 重写         │ ✗            │ ✓                │
  │ 压缩/缓存        │ ✗            │ ✓                │
  │ WAF              │ ✗            │ ✓                │
  │ WebSocket        │ 直接转发     │ 需要配置 Upgrade │
  │ 适用场景         │ 高性能转发   │ Web/API 代理     │
  └──────────────────┴─────────────┴─────────────────┘

常见组合架构:
  Client → CDN → L4 LB (LVS/NLB)
                   └→ L7 LB (Nginx/Envoy)
                        ├→ Service A
                        └→ Service B
```

## 7. 健康检查机制？

**回答：**

```nginx
# ============ 被动健康检查 (OSS 版本) ============

upstream backend {
    server 10.0.1.101:8080 max_fails=3 fail_timeout=30s;
    server 10.0.1.102:8080 max_fails=3 fail_timeout=30s;
}

# 工作原理:
# 1. Nginx 转发请求到后端
# 2. 后端连接失败/超时 → 计入 fail 次数
# 3. 30s 内失败 3 次 → 标记不可用
# 4. 30s 后重新尝试 → 成功则恢复

# 配合 proxy_next_upstream 使用
location / {
    proxy_pass http://backend;
    proxy_next_upstream error timeout http_502 http_503;
}

# ============ 主动健康检查 (Nginx Plus) ============
# upstream backend {
#     zone backend_zone 64k;
#     server 10.0.1.101:8080;
#     server 10.0.1.102:8080;
# }
#
# match health_check {
#     status 200;
#     header Content-Type = application/json;
#     body ~ '"status": "ok"';
# }
#
# server {
#     location / {
#         proxy_pass http://backend;
#         health_check interval=5s fails=3 passes=2
#                      uri=/health match=health_check;
#     }
# }

# ============ 开源替代: 外部健康检查 ============

# 方案 1: nginx_upstream_check_module (Tengine/第三方)
# upstream backend {
#     check interval=3000 rise=2 fall=3 timeout=1000 type=http;
#     check_http_send "GET /health HTTP/1.0\r\n\r\n";
#     check_http_expect_alive http_2xx http_3xx;
#     server 10.0.1.101:8080;
# }

# 方案 2: 外部脚本 + 动态配置
# cron 定期检查后端健康 → 修改配置 → reload

# 方案 3: Consul/Eureka 服务发现 + 模板生成
# consul-template 监听服务变化 → 自动更新 upstream → reload
```

## 8. 灰度发布与流量拆分？

**回答：**

```nginx
# ============ 基于权重的流量拆分 ============

# 方法 1: split_clients (精确百分比)
split_clients $request_id $variant {
    10%     "canary";
    *       "stable";
}

upstream stable {
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
}

upstream canary {
    server 10.0.2.101:8080;
}

server {
    location / {
        proxy_pass http://$variant;
    }
}

# ============ 基于 Header 路由 ============
# 方法 2: map + Header
map $http_x_canary $target_upstream {
    "true"    canary;
    default   stable;
}

server {
    location / {
        proxy_pass http://$target_upstream;
    }
}

# ============ 基于 Cookie 路由 ============
# 方法 3: 内部测试用户
map $cookie_canary $target {
    "yes"     canary;
    default   stable;
}

# ============ 基于 IP 路由 ============
# 方法 4: 内部 IP 先灰度
geo $target_upstream {
    default         stable;
    10.0.0.0/8      canary;    # 内网走金丝雀
    192.168.0.0/16  canary;
}

server {
    location / {
        proxy_pass http://$target_upstream;
    }
}

# ============ A/B 测试 ============
split_clients "${remote_addr}${request_uri}" $ab_variant {
    50%   "A";
    50%   "B";
}

server {
    location / {
        add_header X-AB-Variant $ab_variant;
        if ($ab_variant = "A") {
            proxy_pass http://variant_a;
            break;
        }
        proxy_pass http://variant_b;
    }
}
```

## 9. Upstream Keepalive 连接池？

**回答：**

```nginx
# ============ 为什么需要 Keepalive ============
# 
# 没有 Keepalive:
#   每个请求 → 新建 TCP 连接 → 三次握手 → 请求 → 四次挥手
#   高并发下: 大量 TIME_WAIT, 性能下降
#
# 有 Keepalive:
#   连接复用 → 省去握手/挥手 → 减少延迟 → 减少端口占用

upstream backend {
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;

    # 每个 Worker 保持的空闲连接数
    keepalive 32;
    
    # 每个连接最大请求数 (超过后关闭重建)
    keepalive_requests 1000;
    
    # 空闲连接超时
    keepalive_timeout 60s;
    
    # 等待空闲连接的最大时间 (Nginx 1.25.3+)
    # keepalive_time 1h;
}

server {
    location / {
        proxy_pass http://backend;
        
        # 必须! 使用 HTTP/1.1
        proxy_http_version 1.1;
        
        # 必须! 清空 Connection 头 (默认是 "close")
        proxy_set_header Connection "";
    }
}

# ============ 常见错误 ============
# 
# 错误 1: 忘记设置 proxy_http_version 1.1
#   → 默认 HTTP/1.0, 不支持 keepalive
#
# 错误 2: 忘记清空 Connection 头
#   → Connection: close 会关闭连接
#
# 错误 3: keepalive 数值设太大
#   → 空闲连接占用过多资源
#   → 建议: 并发量 / Worker 数 (留余量)

# ============ 监控 Keepalive 效果 ============
# 查看 TIME_WAIT 连接数
# ss -s | grep TIME-WAIT
# 
# keepalive 生效后 TIME_WAIT 应大幅减少
```

## 10. 负载均衡实战案例？

**回答：**

```nginx
# ============ 电商网站多层负载均衡 ============

# 1. 前端静态资源
upstream frontend {
    server 10.0.1.1:80;
    server 10.0.1.2:80;
    keepalive 16;
}

# 2. API 服务 (最少连接)
upstream api_service {
    least_conn;
    server 10.0.2.1:8080 weight=5 max_fails=3 fail_timeout=30s;
    server 10.0.2.2:8080 weight=5 max_fails=3 fail_timeout=30s;
    server 10.0.2.3:8080 weight=3 max_fails=3 fail_timeout=30s;
    server 10.0.2.99:8080 backup;
    keepalive 64;
}

# 3. 搜索服务 (一致性 Hash, 缓存友好)
upstream search_service {
    hash $arg_q consistent;
    server 10.0.3.1:9200;
    server 10.0.3.2:9200;
    server 10.0.3.3:9200;
    keepalive 32;
}

# 4. WebSocket 服务 (IP Hash 会话保持)
upstream ws_service {
    ip_hash;
    server 10.0.4.1:8080;
    server 10.0.4.2:8080;
}

# 限流
limit_req_zone $binary_remote_addr zone=api_rate:10m rate=20r/s;

server {
    listen 443 ssl http2;
    server_name shop.example.com;

    # 前端 SPA
    location / {
        proxy_pass http://frontend;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
    }

    # API
    location /api/ {
        limit_req zone=api_rate burst=40 nodelay;
        proxy_pass http://api_service;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        include snippets/proxy-headers.conf;
        proxy_next_upstream error timeout http_502 http_503;
        proxy_next_upstream_timeout 5s;
        proxy_next_upstream_tries 2;
    }

    # 搜索
    location /api/search {
        proxy_pass http://search_service;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_read_timeout 10s;
    }

    # WebSocket
    location /ws/ {
        proxy_pass http://ws_service;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_read_timeout 3600s;
    }
}
```
