## 1. Nginx 性能优化总览？

**回答：**

```
性能优化层次:

  ┌──────────────────────────────────────┐
  │ 1. 系统层 (OS Kernel)                 │
  │    文件描述符, TCP 参数, 内存          │
  ├──────────────────────────────────────┤
  │ 2. Nginx 进程层                       │
  │    Worker 数, 连接数, CPU 亲和        │
  ├──────────────────────────────────────┤
  │ 3. 网络传输层                         │
  │    sendfile, tcp_nopush, keepalive    │
  ├──────────────────────────────────────┤
  │ 4. 应用层                             │
  │    Gzip, 缓存, 限流, 缓冲            │
  ├──────────────────────────────────────┤
  │ 5. 后端代理层                         │
  │    连接池, 超时, 重试策略             │
  └──────────────────────────────────────┘
```

## 2. 系统内核参数优化？

**回答：**

```bash
# /etc/sysctl.conf

# ============ TCP 连接优化 ============

# 最大监听队列 (listen 的 backlog)
net.core.somaxconn = 65535

# 网络设备接收队列
net.core.netdev_max_backlog = 65535

# TCP SYN 半连接队列
net.ipv4.tcp_max_syn_backlog = 65535

# 端口范围 (默认 32768-60999)
net.ipv4.ip_local_port_range = 1024 65535

# ============ TIME_WAIT 优化 ============

# TIME_WAIT 端口重用 (反向代理必备)
net.ipv4.tcp_tw_reuse = 1

# FIN_WAIT2 超时 (默认 60s)
net.ipv4.tcp_fin_timeout = 15

# TIME_WAIT 最大数量
net.ipv4.tcp_max_tw_buckets = 200000

# ============ TCP 性能 ============

# TCP 窗口缩放 (大带宽传输)
net.ipv4.tcp_window_scaling = 1

# 发送/接收缓冲区
net.core.rmem_max = 16777216
net.core.wmem_max = 16777216
net.ipv4.tcp_rmem = 4096 87380 16777216
net.ipv4.tcp_wmem = 4096 65536 16777216

# TCP 快速打开 (减少握手延迟)
net.ipv4.tcp_fastopen = 3

# ============ 文件描述符 ============
# /etc/security/limits.conf
# nginx soft nofile 65535
# nginx hard nofile 65535
#
# 或系统级:
fs.file-max = 2097152

# 应用: sysctl -p
```

## 3. Worker 进程优化？

**回答：**

```nginx
# ============ Worker 进程数 ============
worker_processes auto;
# auto = CPU 核心数 (推荐)
# CPU 密集型 (SSL, 压缩) → 等于 CPU 核心数
# I/O 密集型 → 可适当增加

# ============ CPU 亲和性 ============
worker_cpu_affinity auto;
# 绑定 Worker 到特定 CPU 核心
# 减少上下文切换和 CPU 缓存失效
# auto: 自动均匀绑定

# 手动绑定示例 (4 核):
# worker_cpu_affinity 0001 0010 0100 1000;

# ============ Worker 优先级 ============
worker_priority -5;
# Nice 值 -20 到 19, 越小优先级越高
# -5 比普通进程优先级高

# ============ 文件描述符 ============
worker_rlimit_nofile 65535;
# 每个 Worker 的最大文件描述符数
# 应 >= worker_connections × 2

# ============ 事件模型 ============
events {
    worker_connections 65535;   # 每个 Worker 最大连接数
    use epoll;                  # Linux 高性能 I/O
    multi_accept on;            # 一次接受多个连接
    accept_mutex off;           # 高并发关闭互斥锁
}

# ============ 连接能力计算 ============
# 
# 静态文件: 最大并发 = worker_processes × worker_connections
#   例: 4 × 65535 = 262,140
#
# 反向代理: 最大并发 = worker_processes × worker_connections / 2
#   例: 4 × 65535 / 2 = 131,070
#
# 注意: 还受限于系统 文件描述符、端口范围、内存等
```

## 4. sendfile 与零拷贝？

**回答：**

```
传统文件发送 (sendfile off):
  1. 磁盘 → 内核缓冲区 (read)
  2. 内核缓冲区 → 用户空间 (Nginx 进程)
  3. 用户空间 → Socket 内核缓冲区 (write)
  4. Socket 缓冲区 → 网卡

  4 次数据拷贝 + 4 次上下文切换

零拷贝 (sendfile on):
  1. 磁盘 → 内核缓冲区
  2. 内核缓冲区 → 网卡 (DMA, 无 CPU 参与)

  2 次数据拷贝 + 2 次上下文切换
  数据不经过用户空间 → 极大提升静态文件性能
```

```nginx
# ============ 文件传输优化 ============

sendfile on;
# 启用零拷贝, 静态文件性能提升巨大

tcp_nopush on;
# 配合 sendfile 使用
# 积累数据包, 一次发送完整数据 (减少小包)
# 适合大文件传输

tcp_nodelay on;
# 禁用 Nagle 算法
# 小数据包立即发送, 减少延迟
# 适合实时交互, API 响应

# tcp_nopush + tcp_nodelay 同时开启:
# Nginx 会先用 tcp_nopush 凑满包
# 最后一个包用 tcp_nodelay 立即发送
# 兼顾吞吐量和延迟

# ============ AIO (异步 I/O) ============
# 对大文件读取有帮助
location /large-files/ {
    aio on;
    directio 512k;      # > 512k 的文件使用 directio
    sendfile on;         # < 512k 的文件使用 sendfile
    output_buffers 1 128k;
}

# directio: 绕过内核页面缓存, 直接从磁盘读取
# 适合大文件 (避免占满页面缓存)
# 不适合小文件 (页面缓存更快)
```

## 5. Gzip 压缩优化？

**回答：**

```nginx
# ============ Gzip 配置 ============

gzip on;                           # 启用压缩
gzip_vary on;                      # 添加 Vary: Accept-Encoding
gzip_proxied any;                  # 代理请求也压缩
gzip_comp_level 6;                 # 压缩级别 1-9 (6 性价比最高)
gzip_min_length 1000;              # 小于 1000 字节不压缩
gzip_types                         # 压缩的 MIME 类型
    text/plain
    text/css
    text/xml
    text/javascript
    application/json
    application/javascript
    application/xml
    application/rss+xml
    application/atom+xml
    image/svg+xml;

# 注意: 不要压缩图片/视频/压缩文件 (已压缩, 压缩效果差且浪费 CPU)
# 不需要列出 text/html, 默认总是压缩
```

```
压缩级别对比:
  ┌────────────┬───────────────┬───────────────┐
  │ Level       │ 压缩率 (大致)  │ CPU 消耗       │
  ├────────────┼───────────────┼───────────────┤
  │ 1           │ ~30%          │ 低             │
  │ 4-6 (推荐)  │ ~60-70%       │ 中             │
  │ 9           │ ~70-75%       │ 高             │
  └────────────┴───────────────┴───────────────┘

  Level 1→6: 压缩率提升明显
  Level 6→9: 压缩率提升很小, CPU 消耗增大
  推荐: Level 6 (最佳性价比)

Brotli 压缩 (比 Gzip 更好):
  需要第三方模块: ngx_brotli

  brotli on;
  brotli_comp_level 6;
  brotli_types text/plain text/css application/json application/javascript;
  
  Brotli vs Gzip:
    Brotli 压缩率高 15-20%
    所有现代浏览器支持
    只在 HTTPS 下生效
```

## 6. Keep-Alive 连接优化？

**回答：**

```nginx
# ============ 客户端 Keep-Alive ============

keepalive_timeout 65;          # 保持连接超时 (秒)
keepalive_requests 1000;       # 单连接最大请求数

# keepalive_timeout 65 含义:
#   客户端最后一次请求后, 连接保持 65 秒
#   65 秒内无新请求 → 关闭连接
#
# keepalive_requests 1000:
#   单个连接处理 1000 个请求后强制关闭
#   防止内存泄漏, 保证资源回收

# 调优建议:
#   高流量站点: keepalive_timeout 30; keepalive_requests 10000;
#   低流量站点: keepalive_timeout 65; keepalive_requests 100;

# ============ 后端 Keep-Alive ============
upstream backend {
    server 10.0.1.101:8080;
    keepalive 32;              # 保持 32 个空闲连接
    keepalive_requests 1000;   # 每个连接最大请求数
    keepalive_timeout 60s;     # 空闲连接超时
}

location / {
    proxy_pass http://backend;
    proxy_http_version 1.1;     # 必须! HTTP/1.0 不支持 keepalive
    proxy_set_header Connection "";  # 必须! 清空默认的 "close"
}

# ============ 效果验证 ============
# 查看 TIME_WAIT 连接数 (应明显减少)
# ss -s | grep TIME-WAIT
#
# 查看连接复用情况
# 日志中 $upstream_connect_time 应接近 0 (复用连接无需握手)
```

## 7. 超时参数调优？

**回答：**

```nginx
# ============ 客户端超时 ============
client_header_timeout 10s;    # 读取请求头超时
client_body_timeout 10s;      # 读取请求体超时 (两个连续读操作之间)
send_timeout 10s;             # 发送响应超时 (两个连续写操作之间)
lingering_timeout 5s;         # 延迟关闭超时

# ============ 后端代理超时 ============
proxy_connect_timeout 5s;     # 与后端建立连接超时
proxy_send_timeout 60s;       # 向后端发送请求超时
proxy_read_timeout 60s;       # 从后端读取响应超时

# ============ 各场景建议 ============

# API 服务 (快速响应)
location /api/ {
    proxy_connect_timeout 3s;
    proxy_send_timeout 10s;
    proxy_read_timeout 10s;
}

# 上传文件 (大体积, 需要长超时)
location /upload/ {
    client_body_timeout 300s;
    proxy_read_timeout 300s;
    client_max_body_size 5g;
}

# WebSocket (长连接)
location /ws/ {
    proxy_read_timeout 3600s;   # 1 小时
    proxy_send_timeout 3600s;
}

# 报表/导出 (耗时操作)
location /api/export/ {
    proxy_read_timeout 300s;    # 5 分钟
}

# ============ 超时过短的问题 ============
# 504 Gateway Timeout: proxy_read_timeout 太短
# 408 Request Timeout: client_header/body_timeout 太短
# 
# ============ 超时过长的问题 ============
# 连接不释放 → Worker 连接数耗尽
# 慢速攻击 (Slowloris) → 占满所有连接
```

## 8. 限流参数调优？

**回答：**

```nginx
# ============ 限流参数深入理解 ============

# limit_req_zone 的 rate 是 "漏桶速率"
# rate=10r/s → 每 100ms 放行 1 个请求

limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;

location /api/ {
    limit_req zone=api burst=20 nodelay;
}

# ============ burst 的含义 ============
#
# 无 burst:
#   rate=10r/s → 每 100ms 一个, 多的直接 503
#   突发流量全部被拒绝
#
# burst=20 (无 nodelay):
#   20 个桶位, 超过 rate 的请求进入桶排队
#   按 rate 速率逐个释放
#   效果: 请求不被拒, 但有延迟
#
# burst=20 nodelay:
#   突发 20 个立即处理 (不排队)
#   之后按 rate 补充桶位
#   超过 burst 的请求直接拒绝
#   效果: 突发流量快速处理, 持续过载才拒绝

# ============ 多维度限流 ============

# 全局 + 按 IP
limit_req_zone $binary_remote_addr zone=per_ip:10m rate=10r/s;
limit_req_zone $server_name zone=per_server:10m rate=1000r/s;

location /api/ {
    limit_req zone=per_ip burst=20 nodelay;
    limit_req zone=per_server burst=100 nodelay;
    # 两个限流同时生效, 任一触发都会限制
}

# 按 API Key 限流
map $http_x_api_key $api_key_zone {
    default $binary_remote_addr;
    ~.+     $http_x_api_key;
}
limit_req_zone $api_key_zone zone=api_key:10m rate=100r/s;

# ============ 共享内存估算 ============
# $binary_remote_addr → 4 bytes (IPv4) 或 16 bytes (IPv6)
# 每个记录 ~128 bytes
# 10m → 约 80,000 个 IP
# 一般 10m 足够
```

## 9. 日志性能优化？

**回答：**

```nginx
# 日志是 I/O 密集操作, 高并发下影响性能

# ============ 日志缓冲 ============
access_log /var/log/nginx/access.log main buffer=64k flush=5s;
# buffer=64k:  缓冲 64k 数据后再写磁盘
# flush=5s:    最多 5s 刷新一次
# 减少磁盘 I/O 次数

# ============ 条件日志 (排除无用请求) ============
map $request_uri $loggable {
    ~*\.(js|css|png|jpg|gif|ico|woff2?)$  0;  # 静态资源不记日志
    ~*healthz                               0;  # 健康检查不记
    ~*readyz                                0;
    default                                 1;
}

access_log /var/log/nginx/access.log main if=$loggable;

# ============ 关闭特定 location 日志 ============
location = /health {
    access_log off;
    return 200 "OK";
}

location ~* \.(css|js|png|jpg)$ {
    access_log off;
    expires 30d;
}

# ============ 错误日志级别 ============
error_log /var/log/nginx/error.log warn;
# 生产环境使用 warn 或 error
# 不要使用 info 或 debug (日志量巨大)

# ============ syslog 远程日志 ============
access_log syslog:server=10.0.1.100:514,facility=local7,tag=nginx main;
# 直接发送到远程 syslog, 减少本地磁盘 I/O

# ============ 日志切割 ============
# logrotate 配置
# /etc/logrotate.d/nginx
# /var/log/nginx/*.log {
#     daily
#     missingok
#     rotate 30
#     compress
#     delaycompress
#     notifempty
#     create 0640 nginx nginx
#     sharedscripts
#     postrotate
#         [ -f /var/run/nginx.pid ] && kill -USR1 $(cat /var/run/nginx.pid)
#     endscript
# }
```

## 10. 性能监控与基准测试？

**回答：**

```nginx
# ============ stub_status 内置监控 ============
server {
    listen 8080;
    
    location /nginx_status {
        stub_status;
        allow 10.0.0.0/8;
        deny all;
    }
}

# 输出:
# Active connections: 291
# server accepts handled requests
#  16630948 16630948 31070465
# Reading: 6 Writing: 179 Waiting: 106
```

```bash
# ============ Prometheus 监控 ============
# nginx-prometheus-exporter 采集 stub_status
# docker run nginx/nginx-prometheus-exporter \
#    --nginx.scrape-uri=http://nginx:8080/nginx_status

# 关键指标:
#   nginx_connections_active    → 活跃连接数
#   nginx_connections_reading   → 正在读请求的连接
#   nginx_connections_writing   → 正在写响应的连接
#   nginx_connections_waiting   → Keep-Alive 空闲连接
#   nginx_http_requests_total   → 请求总数

# VTS 模块 (更丰富的指标):
# nginx-module-vts → 按 server/upstream/filter 统计
# 包含: 请求数, 状态码分布, 流量, 响应时间

# ============ 基准测试工具 ============

# wrk (推荐, 高性能)
# wrk -t4 -c1000 -d30s http://localhost/
# -t4:    4 线程
# -c1000: 1000 并发连接
# -d30s:  持续 30 秒

# ab (Apache Bench, 简单)
# ab -n 100000 -c 500 http://localhost/

# k6 (现代 HTTP 性能测试)
# k6 run script.js

# siege
# siege -c 200 -t 60s http://localhost/

# ============ 调优迭代 ============
# 1. 建立基线: wrk 测试默认配置
# 2. 调整参数 (一次一个)
# 3. 重新测试, 对比结果
# 4. 记录最优配置
# 5. 监控生产指标, 持续优化
```
