# 缓存与静态资源

---

## 1. Nginx 代理缓存 (proxy_cache) 原理？

**回答：**

```
代理缓存工作流程:

  Client → Nginx [Cache]
              │
              ├── 缓存命中 (HIT) → 直接返回缓存内容
              │                     (不访问后端)
              │
              └── 缓存未命中 (MISS) → 转发到后端
                                       ↓
                              后端返回响应
                                       ↓
                              存入缓存 + 返回给客户端

缓存存储结构:
  /var/cache/nginx/
    ├── 0/           # levels=1:2 → 两级目录
    │   ├── 00/
    │   │   ├── abc123def456...   # 缓存文件 (MD5 key)
    │   │   └── ...
    │   └── 01/
    └── 1/

缓存文件内容:
  [缓存元数据 (key, 过期时间, 响应头)]
  [响应头]
  [响应体]
```

## 2. proxy_cache 完整配置？

**回答：**

```nginx
# ============ HTTP 块: 定义缓存区域 ============
http {
    # 定义缓存路径和参数
    proxy_cache_path /var/cache/nginx/api_cache
        levels=1:2              # 目录层级 (1字符/2字符)
        keys_zone=api_cache:10m # 缓存键共享内存 (10MB ≈ 80K keys)
        max_size=1g             # 磁盘缓存最大 1GB
        inactive=60m            # 60 分钟未访问则删除
        use_temp_path=off;      # 不用临时目录 (减少 I/O)

    proxy_cache_path /var/cache/nginx/static_cache
        levels=1:2
        keys_zone=static_cache:20m
        max_size=10g
        inactive=7d;
}

# ============ Server 块: 使用缓存 ============
server {
    location /api/ {
        proxy_pass http://backend;
        
        # 启用缓存区域
        proxy_cache api_cache;
        
        # 缓存键 (决定缓存的唯一标识)
        proxy_cache_key "$scheme$request_method$host$request_uri";
        
        # 缓存有效期 (按状态码)
        proxy_cache_valid 200 301 302 10m;  # 成功响应缓存 10 分钟
        proxy_cache_valid 404 1m;           # 404 缓存 1 分钟
        proxy_cache_valid any 5m;           # 其他响应缓存 5 分钟
        
        # 缓存条件
        proxy_cache_min_uses 2;     # 请求 2 次后才缓存
        proxy_cache_methods GET HEAD; # 只缓存 GET 和 HEAD
        
        # 添加缓存状态头 (便于调试)
        add_header X-Cache-Status $upstream_cache_status;
        
        # 后端不可用时使用过期缓存
        proxy_cache_use_stale error timeout updating http_500 http_502 http_503;
        
        # 后台更新缓存 (用户不等待)
        proxy_cache_background_update on;
        
        # 防止缓存穿透 (多个相同请求同时到达时, 只有一个去后端)
        proxy_cache_lock on;
        proxy_cache_lock_timeout 5s;
    }
}
```

## 3. 缓存状态与调试？

**回答：**

```nginx
# $upstream_cache_status 的值:

# MISS    → 缓存未命中, 从后端获取
# HIT     → 缓存命中, 直接返回
# EXPIRED → 缓存已过期, 正在从后端更新
# STALE   → 使用了过期缓存 (后端不可用时)
# UPDATING → 正在后台更新, 返回旧缓存
# REVALIDATED → 条件请求验证后仍有效
# BYPASS  → 绕过缓存 (proxy_cache_bypass 触发)

# 添加响应头 (便于调试)
add_header X-Cache-Status $upstream_cache_status always;

# ============ 手动清除缓存 ============

# 方法 1: 删除缓存目录
# rm -rf /var/cache/nginx/api_cache/*
# 注意: 不需要 reload

# 方法 2: proxy_cache_purge (Nginx Plus 或第三方模块)
# location ~ /purge(/.*) {
#     allow 10.0.0.0/8;
#     deny all;
#     proxy_cache_purge api_cache "$scheme$request_method$host$1";
# }
# 使用: curl -X PURGE https://example.com/purge/api/users

# 方法 3: 缓存键加版本号
# proxy_cache_key "$scheme$request_method$host$request_uri$arg_v";
# 参数 ?v=2 变化 → 新缓存键 → 旧缓存自然过期
```

```nginx
# ============ 绕过缓存 ============

# proxy_cache_bypass: 请求绕过缓存 (但响应仍会被缓存)
# proxy_no_cache:     响应不写入缓存

# 带特定 Header 时不使用缓存
proxy_cache_bypass $http_cache_control;
proxy_no_cache $http_cache_control;

# 带 Cookie 时不缓存 (登录用户)
map $http_cookie $no_cache {
    default 0;
    ~*session_id 1;
    ~*auth_token 1;
}

proxy_cache_bypass $no_cache;
proxy_no_cache $no_cache;

# POST 请求不缓存 (默认只缓存 GET)
# proxy_cache_methods 已限制
```

## 4. 浏览器缓存控制 (expires)？

**回答：**

```nginx
# ============ expires 指令 ============

# 静态资源: 长期缓存
location ~* \.(jpg|jpeg|png|gif|ico|svg|webp)$ {
    expires 30d;
    # 等价于添加:
    #   Cache-Control: max-age=2592000
    #   Expires: <30天后的日期>
}

# CSS/JS (带版本号时可长缓存)
location ~* \.(css|js)$ {
    expires 7d;
    add_header Cache-Control "public, immutable";
    # immutable: 告诉浏览器在有效期内不要重新验证
}

# 带 hash 的资源 (永久缓存)
location ~* \.[0-9a-f]{8,}\.(css|js|png|jpg)$ {
    expires max;     # 10 年
    add_header Cache-Control "public, immutable";
}

# HTML (不缓存或短缓存)
location ~* \.html$ {
    expires -1;      # 不缓存
    add_header Cache-Control "no-cache, no-store, must-revalidate";
}

# API 响应 (不缓存)
location /api/ {
    add_header Cache-Control "no-store, no-cache, must-revalidate";
    add_header Pragma "no-cache";
    proxy_pass http://backend;
}

# ============ Cache-Control 指令含义 ============
# public:       可被任何缓存 (CDN, 浏览器) 缓存
# private:      只能被浏览器缓存 (不能被 CDN 缓存)
# no-cache:     可缓存, 但每次使用前必须重新验证
# no-store:     完全不缓存
# must-revalidate: 过期后必须重新验证
# immutable:    有效期内不必重新验证
# max-age=N:    缓存有效 N 秒
# s-maxage=N:   CDN/代理缓存有效 N 秒
```

## 5. ETag 与条件请求？

**回答：**

```
条件请求流程:

  第一次请求:
    Client → GET /style.css
    Server → 200 OK
              ETag: "abc123"
              Last-Modified: Tue, 01 Jan 2025 00:00:00 GMT

  后续请求:
    Client → GET /style.css
              If-None-Match: "abc123"
              If-Modified-Since: Tue, 01 Jan 2025 00:00:00 GMT
    
    文件未变:
    Server → 304 Not Modified (不发送文件体, 节省带宽)
    
    文件已变:
    Server → 200 OK + 新文件 + 新 ETag
```

```nginx
# ============ ETag 配置 ============

etag on;   # 默认开启

# ETag 的生成: 基于 文件大小 + 最后修改时间
# 格式: "文件大小16进制-修改时间16进制"
# 例: "5f4-60a2b3c4"

# ============ 何时关闭 ETag ============

# 多服务器场景: 同一文件的 ETag 可能不同
# (inode 不同 → ETag 不同)
# 解决: 使用 Last-Modified 或统一部署

# gzip 压缩后: 弱 ETag (W/"...")
# Nginx 1.7.3+ 自动处理

# ============ SPA 应用缓存策略 ============

# index.html: 不缓存 (每次检查更新)
location = /index.html {
    add_header Cache-Control "no-cache";
    # 浏览器会发条件请求 → 文件没变返回 304
}

# 带 hash 的资源: 长期缓存
# /assets/app.3a4b5c.js
location /assets/ {
    expires max;
    add_header Cache-Control "public, immutable";
}
```

## 6. CDN 与 Nginx 配合？

**回答：**

```
CDN 架构:

  Client → CDN Edge Node (就近节点)
              │
              ├── CDN 缓存命中 → 直接返回
              │
              └── CDN 缓存未命中 → Nginx (源站)
                                     └→ Backend

Nginx 作为 CDN 源站的配置:
```

```nginx
server {
    listen 443 ssl http2;
    server_name origin.example.com;

    # CDN 回源 Header
    # CDN 请求会带 X-Forwarded-For, 需要信任 CDN IP
    set_real_ip_from 103.21.244.0/22;   # Cloudflare
    set_real_ip_from 104.16.0.0/13;
    real_ip_header X-Forwarded-For;
    real_ip_recursive on;

    # 静态资源: 告诉 CDN 缓存
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|woff2?)$ {
        expires 30d;
        add_header Cache-Control "public, s-maxage=2592000";
        # s-maxage: CDN 专用过期时间
        add_header CDN-Cache-Control "max-age=2592000";
    }

    # API: 告诉 CDN 不缓存
    location /api/ {
        add_header Cache-Control "private, no-store";
        add_header CDN-Cache-Control "no-store";
        proxy_pass http://backend;
    }

    # HTML: CDN 短缓存 + 浏览器不缓存
    location ~* \.html$ {
        add_header Cache-Control "no-cache, s-maxage=60";
        # 浏览器每次检查 (no-cache)
        # CDN 缓存 60 秒 (s-maxage)
    }
}

# ============ CDN 缓存清理 ============
# 通过 CDN API 或控制台操作:
#   • Purge by URL: 清除特定 URL 缓存
#   • Purge by Tag: 清除特定标签的缓存
#   • Purge All:    清除所有缓存
#
# 源站配合:
#   Surrogate-Key: product-123 category-5
#   → CDN 支持按 Key 清除相关缓存
```

## 7. 静态资源服务配置？

**回答：**

```nginx
# ============ 基础静态文件服务 ============
server {
    listen 80;
    server_name static.example.com;
    root /var/www/static;

    # 自动索引 (列出目录文件, 按需开启)
    location /downloads/ {
        autoindex on;
        autoindex_exact_size off;   # 显示 MB 而非字节
        autoindex_localtime on;     # 本地时间
    }

    # 默认 index
    index index.html index.htm;

    # 静态文件缓存
    location ~* \.(jpg|jpeg|png|gif|ico|svg|webp)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
        access_log off;    # 静态文件不记日志 (减少 I/O)
    }

    location ~* \.(css|js)$ {
        expires 7d;
        add_header Cache-Control "public";
        access_log off;
    }

    location ~* \.(woff|woff2|ttf|otf|eot)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
        
        # 字体跨域 (常见需求)
        add_header Access-Control-Allow-Origin "*";
        access_log off;
    }

    # 禁止访问隐藏文件
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }
}
```

## 8. SPA 单页应用部署？

**回答：**

```nginx
# ============ React / Vue / Angular 部署 ============

server {
    listen 80;
    server_name app.example.com;
    root /var/www/frontend/dist;

    # ============ 核心: History Mode 路由 ============
    location / {
        try_files $uri $uri/ /index.html;
        # /about → 找文件 /about → 找目录 /about/ → 都没有 → /index.html
        # 让前端路由处理
    }

    # ============ HTML 不缓存 ============
    location = /index.html {
        add_header Cache-Control "no-cache";
        # 每次都检查更新, 确保用户获取最新版本
    }

    # ============ 静态资源长缓存 ============
    # Webpack/Vite 打包的带 hash 文件名资源
    location /assets/ {
        expires max;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    # ============ API 反向代理 ============
    location /api/ {
        proxy_pass http://backend:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # ============ Gzip 压缩 ============
    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml;
    gzip_min_length 1000;

    # 预压缩 (如果构建时生成了 .gz 文件)
    gzip_static on;
    # Nginx 优先使用 .gz 文件, 省去实时压缩的 CPU 开销
    # 需要构建时: vite-plugin-compression 等生成 .gz 文件
}

# ============ Docker 多阶段构建 ============
# Dockerfile
# FROM node:20-alpine AS build
# WORKDIR /app
# COPY package*.json ./
# RUN npm ci
# COPY . .
# RUN npm run build
#
# FROM nginx:1.25-alpine
# COPY --from=build /app/dist /usr/share/nginx/html
# COPY nginx.conf /etc/nginx/conf.d/default.conf
# EXPOSE 80
# CMD ["nginx", "-g", "daemon off;"]
```

## 9. 文件下载与防盗链？

**回答：**

```nginx
# ============ 文件下载配置 ============
location /download/ {
    alias /var/www/files/;
    
    # 限速下载
    limit_rate 1m;          # 每连接限速 1MB/s
    limit_rate_after 10m;   # 前 10MB 不限速

    # 限制并发下载
    limit_conn_zone $binary_remote_addr zone=download:10m;
    limit_conn download 3;   # 每 IP 最多 3 个并发下载
    
    # 强制下载 (不在浏览器中打开)
    add_header Content-Disposition "attachment";
}

# ============ X-Accel-Redirect (安全下载)  ============
# 后端鉴权 → Nginx 发送文件

location /api/download/ {
    proxy_pass http://backend;
}

location /internal-files/ {
    internal;                   # 只允许内部访问
    alias /var/www/protected/;
    
    # 后端返回:
    # X-Accel-Redirect: /internal-files/secret.pdf
    # Content-Type: application/pdf
}

# ============ 防盗链 ============

location ~* \.(jpg|jpeg|png|gif|mp4)$ {
    valid_referers none blocked server_names
                   *.example.com
                   example.com;
    
    if ($invalid_referer) {
        return 403;
        # 或返回防盗链图片:
        # rewrite ^/ /images/hotlink-denied.png break;
    }
}

# valid_referers 参数:
#   none:         没有 Referer 头 (直接访问)
#   blocked:      有 Referer 但被代理/防火墙删除
#   server_names: 当前 server_name
#   *.example.com: 允许的域名 (支持通配符)

# 更安全的防盗链: secure_link 模块
```

## 10. 缓存策略最佳实践总结？

**回答：**

```
缓存策略决策树:

  资源类型？
    │
    ├── HTML
    │   └── Cache-Control: no-cache
    │       (每次验证, 确保最新)
    │
    ├── 带 Hash 的 CSS/JS/图片
    │   └── Cache-Control: public, max-age=31536000, immutable
    │       (一年, 文件名变 = 新资源)
    │
    ├── 不带 Hash 的 CSS/JS
    │   └── Cache-Control: public, max-age=604800
    │       (7 天, 需要手动清缓存)
    │
    ├── API 响应
    │   ├── 用户相关 → no-store (不缓存)
    │   ├── 公共数据 → max-age=60, s-maxage=300
    │   └── 搜索结果 → max-age=300
    │
    └── 字体
        └── max-age=2592000, immutable
            + Access-Control-Allow-Origin: *

多层缓存架构:
  浏览器缓存 (expires, Cache-Control)
    ↓ 未命中
  CDN 缓存 (s-maxage, CDN-Cache-Control)
    ↓ 未命中
  Nginx proxy_cache
    ↓ 未命中
  后端应用 (Redis/Memcached)
    ↓ 未命中
  数据库

配置模板:
```

```nginx
# ============ 推荐缓存配置模板 ============

# 1. proxy_cache 定义
proxy_cache_path /var/cache/nginx
    levels=1:2
    keys_zone=main_cache:10m
    max_size=2g
    inactive=24h
    use_temp_path=off;

server {
    # 2. HTML: 协商缓存
    location ~* \.html$ {
        add_header Cache-Control "no-cache";
        etag on;
    }

    # 3. 带 hash 资源: 强缓存
    location ~* \.[0-9a-f]{8,}\.(css|js|png|jpg|svg|woff2?)$ {
        expires max;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    # 4. 其他静态资源: 中等缓存
    location ~* \.(css|js|png|jpg|gif|ico|svg|woff2?)$ {
        expires 7d;
        add_header Cache-Control "public";
        access_log off;
    }

    # 5. API: proxy_cache
    location /api/ {
        proxy_pass http://backend;
        proxy_cache main_cache;
        proxy_cache_valid 200 5m;
        proxy_cache_valid 404 1m;
        proxy_cache_use_stale error timeout http_500 http_502 http_503;
        proxy_cache_lock on;
        add_header X-Cache-Status $upstream_cache_status;
    }
}
```
