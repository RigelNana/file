# 反向代理深入

---

## 1. 正向代理与反向代理的区别？

**回答：**

```
正向代理 (Forward Proxy):
  客户端知道代理的存在, 代理代替客户端访问外部

  Client ──→ [Forward Proxy] ──→ Internet ──→ Server
  
  用途:
    • 翻墙 / 科学上网
    • 网络出口统一管控
    • 客户端缓存
    • 隐藏客户端 IP

反向代理 (Reverse Proxy):
  客户端不知道代理的存在, 以为直接访问了后端

  Client ──→ Internet ──→ [Reverse Proxy] ──→ Backend Servers
  
  用途:
    • 负载均衡
    • SSL 终止
    • 安全防护 (隐藏后端 IP)
    • 缓存加速
    • 统一入口 (API Gateway)

关键区别:
  ┌──────────────────┬──────────────┬──────────────┐
  │ 维度              │ 正向代理      │ 反向代理      │
  ├──────────────────┼──────────────┼──────────────┤
  │ 代理对象          │ 客户端        │ 服务端        │
  │ 客户端是否感知    │ 是            │ 否            │
  │ 配置方          │ 客户端设置    │ 服务端部署    │
  │ 隐藏的是          │ 客户端 IP     │ 服务端 IP     │
  │ 典型工具          │ Squid, V2Ray │ Nginx, Envoy │
  └──────────────────┴──────────────┴──────────────┘
```

## 2. proxy_pass 完整配置详解？

**回答：**

```nginx
server {
    listen 80;
    server_name api.example.com;

    location /api/ {
        # ============ 核心转发 ============
        proxy_pass http://backend;

        # ============ 请求头传递 ============
        # 传递真实客户端信息 (极其重要!)
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Host $host;
        proxy_set_header X-Forwarded-Port $server_port;

        # ============ HTTP 版本 ============
        proxy_http_version 1.1;
        # 使用 HTTP/1.1 → 支持 chunked transfer + keepalive

        # ============ 超时设置 ============
        proxy_connect_timeout 5s;     # 与后端建立连接超时
        proxy_send_timeout 60s;       # 发送请求到后端超时
        proxy_read_timeout 60s;       # 从后端读取响应超时

        # ============ 缓冲设置 ============
        proxy_buffering on;           # 开启响应缓冲 (默认 on)
        proxy_buffer_size 4k;         # 响应头缓冲大小
        proxy_buffers 8 4k;           # 响应体缓冲: 8 个 4k 缓冲
        proxy_busy_buffers_size 8k;   # 忙碌时可用的缓冲大小

        # ============ 请求体 ============
        proxy_request_buffering on;   # 缓冲请求体 (全部读完再转发)
        client_max_body_size 100m;    # 最大上传大小
    }
}
```

## 3. proxy_set_header 传递客户端真实 IP？

**回答：**

```
请求链路中的 IP 传递:

  Client (1.2.3.4)
    → CDN (5.6.7.8)
      → LB (10.0.0.1)
        → Nginx (10.0.1.1)
          → Backend

各层的 Header 变化:
  CDN 添加:   X-Forwarded-For: 1.2.3.4
  LB 添加:    X-Forwarded-For: 1.2.3.4, 5.6.7.8
  Nginx 添加: X-Forwarded-For: 1.2.3.4, 5.6.7.8, 10.0.0.1
```

```nginx
# ============ 标准头部设置 ============
proxy_set_header X-Real-IP $remote_addr;
proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
proxy_set_header X-Forwarded-Proto $scheme;

# $proxy_add_x_forwarded_for =
#   如果请求已有 X-Forwarded-For → 追加 $remote_addr
#   如果没有 → 等于 $remote_addr

# ============ realip 模块: 提取真实 IP ============
# 当 Nginx 前面有 CDN/LB 时, $remote_addr 是 CDN/LB 的 IP
# 需要 realip 模块提取客户端真实 IP

# 信任的代理 IP (CDN/LB 地址段)
set_real_ip_from 10.0.0.0/8;
set_real_ip_from 172.16.0.0/12;
set_real_ip_from 192.168.0.0/16;
set_real_ip_from 103.21.244.0/22;   # Cloudflare

# 从哪个头部获取真实 IP
real_ip_header X-Forwarded-For;

# 递归查找 (跳过信任 IP, 找到第一个非信任 IP)
real_ip_recursive on;

# 配置后 $remote_addr 会被替换为真实客户端 IP

# ============ 安全注意 ============
# X-Forwarded-For 可以被客户端伪造!
# 必须配合 set_real_ip_from 设置可信代理范围
# 不要信任未经验证的 X-Forwarded-For
```

## 4. proxy_buffering 缓冲机制？

**回答：**

```
缓冲 vs 非缓冲:

  proxy_buffering on (默认):
    Client ← Nginx [Buffer] ← Backend
    • Nginx 先将后端响应完整读入缓冲
    • 再发送给客户端
    • 后端连接快速释放
    • 适合: 普通 API, 静态文件
    
  proxy_buffering off:
    Client ← Nginx ← Backend (同步)
    • Nginx 直接将后端数据传给客户端
    • 后端连接占用时间长
    • 适合: SSE, 长轮询, 实时流
```

```nginx
# ============ 缓冲参数详解 ============

# 响应头缓冲 (第一个 buffer)
proxy_buffer_size 4k;
# 用于缓冲后端响应头
# 如果后端返回大 Header (如大 Cookie), 需要增大

# 响应体缓冲
proxy_buffers 8 4k;
# 8 个 4k = 32k 总缓冲
# 响应超过缓冲 → 临时写入磁盘

# 可以忙碌发送的缓冲大小
proxy_busy_buffers_size 8k;

# 临时文件 (缓冲不够时写磁盘)
proxy_temp_path /var/cache/nginx/proxy_temp;
proxy_max_temp_file_size 1024m;
proxy_temp_file_write_size 8k;

# ============ 特殊场景 ============

# SSE (Server-Sent Events) - 必须关闭缓冲
location /events {
    proxy_pass http://backend;
    proxy_buffering off;
    proxy_cache off;
    proxy_set_header Connection "";
    proxy_http_version 1.1;
    chunked_transfer_encoding off;
}

# 大文件下载 - 增大缓冲
location /download/ {
    proxy_pass http://storage;
    proxy_buffering on;
    proxy_buffers 16 64k;
    proxy_busy_buffers_size 128k;
}

# 按请求控制 (后端通过 Header)
# 后端返回 X-Accel-Buffering: no → 禁用缓冲
# 后端返回 X-Accel-Buffering: yes → 启用缓冲
```

## 5. WebSocket 代理配置？

**回答：**

```nginx
# ============ WebSocket 代理 ============

# WebSocket 握手是 HTTP Upgrade 请求
# 需要传递 Upgrade 和 Connection 头

map $http_upgrade $connection_upgrade {
    default upgrade;
    ''      close;
}

upstream ws_backend {
    server 10.0.1.101:8080;
    server 10.0.1.102:8080;
    # WebSocket 需要会话保持
    ip_hash;
}

server {
    listen 443 ssl http2;
    server_name ws.example.com;

    location /ws {
        proxy_pass http://ws_backend;
        proxy_http_version 1.1;

        # 关键: 传递 Upgrade 头
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection $connection_upgrade;

        # 标准代理头
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;

        # WebSocket 超时 (默认 60s 无数据就断开)
        proxy_read_timeout 3600s;    # 1 小时
        proxy_send_timeout 3600s;

        # 关闭缓冲
        proxy_buffering off;
    }
}

# ============ 为什么需要 map? ============
# 
# 普通 HTTP 请求: 不需要 Upgrade
#   → Connection: close (默认行为)
# 
# WebSocket 请求: 有 Upgrade: websocket 头
#   → Connection: upgrade (需要传递)
#
# map 根据 $http_upgrade 值动态决定 Connection 的值

# ============ Socket.IO 特殊处理 ============
location /socket.io/ {
    proxy_pass http://socketio_backend;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection $connection_upgrade;
    
    # Socket.IO 同时使用 HTTP 长轮询和 WebSocket
    # 需要会话保持 (sticky session)
}
```

## 6. gRPC 代理配置？

**回答：**

```nginx
# gRPC 使用 HTTP/2 + Protobuf
# Nginx 1.13.10+ 支持 gRPC 代理

# ============ gRPC 代理 (明文) ============
upstream grpc_backend {
    server 10.0.1.101:50051;
    server 10.0.1.102:50051;
}

server {
    listen 80 http2;   # 明文 HTTP/2

    location / {
        grpc_pass grpc://grpc_backend;
        # grpc_pass  → 明文 gRPC
        # grpcs_pass → TLS gRPC

        # 超时
        grpc_connect_timeout 5s;
        grpc_read_timeout 60s;
        grpc_send_timeout 60s;

        # 传递 metadata
        grpc_set_header X-Real-IP $remote_addr;
    }
}

# ============ gRPC + TLS 终止 ============
server {
    listen 443 ssl http2;
    server_name grpc.example.com;

    ssl_certificate     /etc/ssl/grpc.example.com/fullchain.pem;
    ssl_certificate_key /etc/ssl/grpc.example.com/privkey.pem;

    location / {
        # Nginx 终止 TLS, 后端明文 gRPC
        grpc_pass grpc://grpc_backend;
    }
}

# ============ gRPC 按服务路由 ============
location /mypackage.MyService/ {
    grpc_pass grpc://service_a;
}

location /mypackage.OtherService/ {
    grpc_pass grpc://service_b;
}

# ============ gRPC 健康检查 ============
location /grpc.health.v1.Health/ {
    grpc_pass grpc://grpc_backend;
    access_log off;
}
```

## 7. 反向代理的故障转移与重试？

**回答：**

```nginx
upstream backend {
    server 10.0.1.101:8080 max_fails=3 fail_timeout=30s;
    server 10.0.1.102:8080 max_fails=3 fail_timeout=30s;
    server 10.0.1.103:8080 backup;  # 备用: 主服务器全挂时启用
}

# max_fails=3:   30s 内失败 3 次 → 标记为不可用
# fail_timeout=30s: 不可用持续 30s 后重新尝试

server {
    location / {
        proxy_pass http://backend;

        # ============ 重试条件 ============
        # 哪些情况触发转发到下一台服务器
        proxy_next_upstream error timeout http_502 http_503 http_504;
        
        # error:      连接错误
        # timeout:    超时
        # http_502:   后端返回 502
        # http_503:   后端返回 503
        # http_504:   后端返回 504
        # invalid_header: 无效响应头
        # non_idempotent: 允许非幂等请求重试 (POST等, 慎用!)

        # 重试超时限制 (防止重试时间过长)
        proxy_next_upstream_timeout 10s;  # 总重试时间不超过 10s
        proxy_next_upstream_tries 3;      # 最多重试 3 次

        # ============ 超时设置 ============
        proxy_connect_timeout 3s;    # 连接超时 (快速失败)
        proxy_read_timeout 30s;      # 读取超时
        proxy_send_timeout 10s;      # 发送超时
    }
}

# ============ 注意事项 ============
# 1. POST 请求默认不重试 (非幂等)
#    如需重试: 加 non_idempotent
#    但会导致重复提交! 后端必须幂等

# 2. proxy_next_upstream 对大文件要谨小
#    已传输部分数据后重试 → 客户端可能收到错乱数据

# 3. 被动健康检查 vs 主动健康检查
#    OSS: 只有被动 (请求失败时才标记)
#    Plus: 支持主动 (定期发探针)
```

## 8. 代理请求体与大文件上传？

**回答：**

```nginx
# ============ 请求体大小限制 ============

# 限制客户端请求体大小 (上传大小)
client_max_body_size 100m;
# 超过限制 → 413 Request Entity Too Large

# 大文件上传场景
location /upload {
    proxy_pass http://upload_backend;
    
    client_max_body_size 5g;          # 允许 5GB 上传
    client_body_timeout 300s;          # 上传超时 5 分钟
    proxy_read_timeout 300s;           # 后端处理超时
    
    # 请求体缓冲
    proxy_request_buffering on;        # 缓冲请求体 (默认)
    client_body_buffer_size 128k;      # 内存缓冲大小
    client_body_temp_path /var/cache/nginx/upload_temp;
    
    # 或者: 不缓冲, 边收边转 (减少磁盘 I/O)
    # proxy_request_buffering off;
    # 适合超大文件 + 后端能逐块处理
}

# ============ Nginx 作为文件下载代理 ============
location /download/ {
    # 内部重定向 (X-Accel-Redirect)
    # 后端返回 X-Accel-Redirect: /internal/files/xxx
    # Nginx 替后端发送文件
    internal;
    alias /var/www/files/;
}

# 后端 (Python Flask 示例):
# @app.route('/api/download/<filename>')
# def download(filename):
#     # 鉴权逻辑...
#     response = make_response()
#     response.headers['X-Accel-Redirect'] = f'/download/{filename}'
#     response.headers['Content-Type'] = 'application/octet-stream'
#     return response

# 好处: 鉴权在后端, 文件传输在 Nginx (高性能)
```

## 9. 反向代理中的 Header 处理？

**回答：**

```nginx
# ============ 请求头 (Client → Backend) ============

# 设置/覆盖请求头
proxy_set_header Host $host;
proxy_set_header X-Custom "value";

# 传递所有原始请求头 (默认行为)
# Nginx 默认传递大多数请求头, 但会:
#   • 将 Host 改为 proxy_pass 中的值
#   • 删掉值为空的头
#   • Connection 设为 close

# 隐藏某些请求头不传给后端
proxy_set_header Authorization "";  # 清空 Auth 头

# ============ 响应头 (Backend → Client) ============

# 隐藏后端响应头
proxy_hide_header X-Powered-By;    # 隐藏技术栈信息
proxy_hide_header Server;           # 隐藏后端 Server
proxy_hide_header X-AspNet-Version;

# 传递后端默认隐藏的响应头
proxy_pass_header X-Custom-Header;

# 添加/修改响应头
add_header X-Frame-Options "SAMEORIGIN" always;
add_header X-Content-Type-Options "nosniff" always;
add_header X-XSS-Protection "1; mode=block" always;

# ============ add_header 的坑 ============
# 
# add_header 在内部 location 会覆盖外部的所有 add_header
# 不是追加, 是完全覆盖!
#
# 解决方案 1: 每个 location 重复声明所有 header
# 解决方案 2: 使用 headers-more 第三方模块
#
# more_set_headers "X-Custom: value";  # 总是添加
# more_clear_headers "Server";          # 清除

# ============ Cookie 处理 ============
# 当后端设置的 Cookie 域名/路径需要调整时

proxy_cookie_domain backend.internal example.com;
# Set-Cookie: ... domain=backend.internal → domain=example.com

proxy_cookie_path /internal/ /;
# Set-Cookie: ... path=/internal/ → path=/
```

## 10. 反向代理安全最佳实践？

**回答：**

```nginx
# ============ 安全头部 ============
server {
    # 防止点击劫持
    add_header X-Frame-Options "SAMEORIGIN" always;
    
    # 防止 MIME 类型嗅探
    add_header X-Content-Type-Options "nosniff" always;
    
    # XSS 防护
    add_header X-XSS-Protection "1; mode=block" always;
    
    # CSP 内容安全策略
    add_header Content-Security-Policy "default-src 'self'" always;
    
    # Referrer 策略
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    # 隐藏 Nginx 版本号
    server_tokens off;
    
    # 隐藏后端信息
    proxy_hide_header X-Powered-By;
}

# ============ 访问控制 ============
# IP 白名单
location /admin/ {
    allow 10.0.0.0/8;
    allow 192.168.0.0/16;
    deny all;
    proxy_pass http://admin_backend;
}

# ============ 限流防护 ============
limit_req_zone $binary_remote_addr zone=api_limit:10m rate=10r/s;
limit_conn_zone $binary_remote_addr zone=conn_limit:10m;

location /api/ {
    limit_req zone=api_limit burst=20 nodelay;
    limit_conn conn_limit 50;
    proxy_pass http://backend;
}

# ============ 防止请求走私 ============
# 确保前后端对请求的解析一致
proxy_http_version 1.1;
proxy_set_header Connection "";

# ============ 内部路径保护 ============
# 防止客户端直接访问内部路径
location /internal/ {
    internal;   # 只允许内部重定向
}

# ============ 后端真实 IP ============
# 确保后端拿到的 IP 是客户端而非 Nginx
proxy_set_header X-Real-IP $remote_addr;
proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
```
