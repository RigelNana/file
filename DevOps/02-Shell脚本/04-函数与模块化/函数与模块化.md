## 1. Shell 函数的定义方式有哪些？有什么区别？

**回答：**

```bash
# 方式1：function 关键字（Bash 扩展）
function greet {
    echo "Hello, $1"
}

# 方式2：POSIX 标准语法
greet() {
    echo "Hello, $1"
}

# 方式3：混合语法（Bash 支持，不推荐）
function greet() {
    echo "Hello, $1"
}
```

| 特性 | `function name { }` | `name() { }` |
|------|---------------------|---------------|
| 标准 | Bash 扩展 | POSIX 标准 |
| 可移植性 | 仅 Bash/Ksh/Zsh | 所有 POSIX Shell |
| 花括号 | 可选用 `()` 替代 | 必须用 `{ }` |
| 推荐度 | 一般 | 推荐 |

**最佳实践：** 使用 `name() { }` POSIX 语法以保证最大兼容性。

### 函数体可以是任何复合命令

```bash
# 用子 Shell 作为函数体（变量修改不影响调用者）
isolated_func() (
    cd /tmp
    var="modified"
    echo "在子 Shell 中"
)
# 调用后当前目录和 var 不变

# 普通花括号（共享当前 Shell 环境）
normal_func() {
    cd /tmp           # 会改变调用者的目录！
    var="modified"    # 会修改调用者的变量
}
```

## 2. 函数参数传递和访问机制是怎样的？

**回答：**

```bash
# 函数内使用 $1, $2, ... 访问参数（和脚本参数独立）
show_args() {
    echo "函数名: ${FUNCNAME[0]}"  # 函数名（非 $0）
    echo "参数个数: $#"
    echo "第1个参数: $1"
    echo "第2个参数: $2"
    echo "所有参数: $@"
    echo "所有参数(合并): $*"
}

show_args "hello" "world" "foo"
```

### `$@` vs `$*` 在函数中的区别

```bash
test_args() {
    echo "--- \$@ (分别引用) ---"
    for arg in "$@"; do
        echo "  [$arg]"
    done

    echo "--- \$* (合并为一个) ---"
    for arg in "$*"; do
        echo "  [$arg]"
    done
}

test_args "hello world" "foo bar"
# $@ 输出：[hello world] [foo bar]     — 保持原样
# $* 输出：[hello world foo bar]        — 合并成一个字符串
```

### 传递数组给函数

```bash
# 方法1：展开传递
process_array() {
    local arr=("$@")   # 用 $@ 重建数组
    for item in "${arr[@]}"; do
        echo "$item"
    done
}
my_arr=(a b c "d e")
process_array "${my_arr[@]}"

# 方法2：通过 nameref（Bash 4.3+）
process_by_ref() {
    local -n ref=$1     # nameref 引用调用者的变量
    for item in "${ref[@]}"; do
        echo "$item"
    done
}
my_arr=(x y z)
process_by_ref my_arr

# 方法3：传递数组名 + eval（不推荐，安全风险）
```

### shift 操作

```bash
# shift 在函数内独立于脚本
log_msg() {
    local level=$1
    shift               # 移除第一个参数
    echo "[$level] $*"  # 剩余参数作为消息
}
log_msg "ERROR" "Something" "went" "wrong"
# 输出：[ERROR] Something went wrong
```

## 3. 函数返回值有哪些方式？return 和 echo 的区别？

**回答：**

### 方式1：return（退出状态码）

```bash
# return 只能返回 0-255 的整数
is_even() {
    (( $1 % 2 == 0 ))
    return $?    # 可省略，默认返回最后一条命令的状态
}

if is_even 4; then
    echo "是偶数"
fi
```

### 方式2：echo/printf 输出（返回字符串或数据）

```bash
# 通过命令替换捕获输出
get_timestamp() {
    date "+%Y-%m-%d %H:%M:%S"
}
ts=$(get_timestamp)
echo "当前时间: $ts"

# 返回计算结果
add() {
    echo $(( $1 + $2 ))
}
result=$(add 3 5)
echo "结果: $result"
```

### 方式3：全局变量

```bash
# 直接修改全局变量（简单但不推荐，污染命名空间）
get_info() {
    RESULT_NAME="server01"
    RESULT_STATUS="running"
}
get_info
echo "$RESULT_NAME is $RESULT_STATUS"
```

### 方式4：nameref（Bash 4.3+，推荐）

```bash
# 通过引用变量"返回"结果
divide() {
    local -n _quotient=$1
    local -n _remainder=$2
    _quotient=$(( $3 / $4 ))
    _remainder=$(( $3 % $4 ))
}

divide q r 17 5
echo "商=$q 余=$r"   # 商=3 余=2
```

### 各方式对比

| 方式 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| `return` | 简洁，惯用 | 只能返回 0-255 | 成功/失败判断 |
| `echo` + `$()` | 灵活，可返回任意字符串 | 创建子 Shell，有性能开销 | 返回数据 |
| 全局变量 | 无子 Shell 开销 | 污染命名空间 | 简单脚本 |
| `nameref` | 无子 Shell，可返回多值 | 需要 Bash 4.3+ | 复杂返回值 |

### 注意事项

```bash
# 陷阱：函数中的 echo 会污染返回值
get_value() {
    echo "调试信息"       # 这也会被 $() 捕获！
    echo "actual_value"
}
result=$(get_value)   # result = "调试信息\nactual_value"

# 解决：调试信息输出到 stderr
get_value() {
    echo "调试信息" >&2   # 输出到 stderr，不被 $() 捕获
    echo "actual_value"   # 只有这个被捕获
}
```

## 4. local 变量和变量作用域在函数中如何工作？

**回答：**

```bash
# 默认：函数中的变量是全局的！
func1() {
    var="I'm global"     # 全局变量
    local lvar="I'm local"  # 局部变量
}
func1
echo "$var"    # I'm global
echo "$lvar"   # 空（local 变量已销毁）
```

### local 的特性

```bash
# local 只在当前函数及其调用的函数中可见（动态作用域）
outer() {
    local x=10
    inner
}
inner() {
    echo "$x"    # 输出 10！动态作用域，能看到调用者的 local
}
outer

# 这和大多数编程语言的词法作用域不同
# Bash 使用 动态作用域（dynamic scoping）
```

### local 声明类型

```bash
func() {
    local str="hello"          # 普通局部变量
    local -i num=42            # 整数类型
    local -r const="fixed"     # 只读
    local -a arr=(1 2 3)       # 索引数组
    local -A map               # 关联数组
    map[key]="value"
    local -n ref=outer_var     # nameref 引用
    local -l lower="HELLO"     # 自动转小写（Bash 4+）
    local -u upper="hello"     # 自动转大写（Bash 4+）
}
```

### local 与子 Shell

```bash
# local 在子 Shell 中的行为
func() {
    local x=10
    (
        echo "子Shell中 x=$x"  # 10（继承副本）
        x=20
        echo "修改后 x=$x"     # 20（子 Shell 内修改）
    )
    echo "父Shell中 x=$x"     # 10（不受子 Shell 影响）
}
```

### 变量遮蔽

```bash
x="global"
func() {
    local x="local"    # 遮蔽全局 x
    echo "函数内: $x"  # local
    subfunc
}
subfunc() {
    echo "子函数: $x"  # local（动态作用域，看到调用者的 local）
}
func
echo "函数外: $x"      # global
```

## 5. 递归函数在 Shell 中怎么写？有什么限制？

**回答：**

```bash
# 阶乘
factorial() {
    if (( $1 <= 1 )); then
        echo 1
    else
        local prev
        prev=$(factorial $(( $1 - 1 )))
        echo $(( $1 * prev ))
    fi
}
echo "5! = $(factorial 5)"   # 120

# 递归遍历目录
tree_walk() {
    local dir=$1
    local indent=${2:-0}
    local prefix=""
    for (( i = 0; i < indent; i++ )); do
        prefix+="  "
    done

    for item in "$dir"/*; do
        [ -e "$item" ] || continue
        echo "${prefix}$(basename "$item")"
        if [ -d "$item" ]; then
            tree_walk "$item" $(( indent + 1 ))
        fi
    done
}
tree_walk /etc/nginx
```

### 递归限制

```bash
# 查看递归深度限制
echo "$FUNCNEST"          # 默认未设置（无限制，受栈大小限制）
# ulimit -s               # 查看栈大小

# 设置递归深度保护
FUNCNEST=100              # 最多递归 100 层，超过报错

# 性能问题
# 每次递归调用 $() 会创建子 Shell，开销很大
# 深度递归建议改用循环
```

### 用循环替代递归（尾递归优化）

```bash
# 递归版（慢）
fib_recursive() {
    if (( $1 <= 1 )); then
        echo "$1"
    else
        local a=$(fib_recursive $(($1 - 1)))
        local b=$(fib_recursive $(($1 - 2)))
        echo $(( a + b ))
    fi
}

# 循环版（快得多）
fib_iterative() {
    local n=$1 a=0 b=1 tmp
    for (( i = 0; i < n; i++ )); do
        tmp=$b
        b=$(( a + b ))
        a=$tmp
    done
    echo "$a"
}
```

## 6. 如何创建和使用函数库（source/. 命令）？

**回答：**

### 创建函数库文件

```bash
# lib/logging.sh — 日志函数库
#!/bin/bash
# 不需要执行权限，只通过 source 加载

LOG_LEVEL=${LOG_LEVEL:-INFO}

_log_levels=(DEBUG INFO WARN ERROR FATAL)

_get_level_num() {
    local level=$1
    for i in "${!_log_levels[@]}"; do
        if [[ "${_log_levels[$i]}" == "$level" ]]; then
            echo "$i"
            return
        fi
    done
    echo 1  # 默认 INFO
}

log() {
    local level=$1
    shift
    local msg="$*"
    local current=$(_get_level_num "$LOG_LEVEL")
    local target=$(_get_level_num "$level")

    if (( target >= current )); then
        printf "[%s] [%s] [%s] %s\n" \
            "$(date '+%Y-%m-%d %H:%M:%S')" \
            "$level" \
            "${FUNCNAME[1]:-main}" \
            "$msg" >&2
    fi
}

log_debug() { log DEBUG "$@"; }
log_info()  { log INFO "$@"; }
log_warn()  { log WARN "$@"; }
log_error() { log ERROR "$@"; }
log_fatal() { log FATAL "$@"; exit 1; }
```

### 使用函数库

```bash
#!/bin/bash
# main.sh

# 获取脚本所在目录（处理符号链接）
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

# 加载函数库
source "$SCRIPT_DIR/lib/logging.sh"
source "$SCRIPT_DIR/lib/utils.sh"

# 或用 . 命令（POSIX 等价）
. "$SCRIPT_DIR/lib/config.sh"

# 使用库函数
log_info "脚本启动"
```

### 防重复加载（Include Guard）

```bash
# lib/utils.sh
[[ -n "${_UTILS_LOADED+x}" ]] && return 0
_UTILS_LOADED=1

# ... 函数定义 ...
trim() {
    local str="$1"
    str="${str#"${str%%[![:space:]]*}"}"
    str="${str%"${str##*[![:space:]]}"}"
    echo "$str"
}
```

### source vs 子 Shell 执行

```bash
# source（在当前 Shell 中执行）
source lib.sh       # 函数和变量在当前 Shell 中可用
. lib.sh            # 同上（POSIX 语法）

# 子 Shell 执行
bash lib.sh         # 新进程，函数和变量对当前 Shell 不可见
./lib.sh            # 同上
```

### 动态加载库

```bash
# 加载目录下所有 .sh 文件
for lib in "$SCRIPT_DIR"/lib/*.sh; do
    [ -f "$lib" ] || continue
    source "$lib"
done

# 按需加载
require() {
    local lib="$SCRIPT_DIR/lib/$1.sh"
    if [ -f "$lib" ]; then
        source "$lib"
    else
        echo "库文件不存在: $lib" >&2
        return 1
    fi
}
require logging
require utils
```

## 7. FUNCNAME、BASH_SOURCE、BASH_LINENO 数组有什么用？

**回答：**

这三个数组记录了函数**调用栈**信息：

```bash
# FUNCNAME[0]     = 当前函数名
# FUNCNAME[1]     = 调用者函数名
# BASH_SOURCE[0]  = 当前函数所在的源文件
# BASH_SOURCE[1]  = 调用者所在的源文件
# BASH_LINENO[0]  = 调用当前函数的行号
# BASH_LINENO[1]  = 调用上一级函数的行号
```

### 打印调用栈

```bash
stacktrace() {
    local frame=0
    echo "=== Stack Trace ==="
    while caller $frame; do
        (( frame++ ))
    done | while IFS=' ' read -r line func file; do
        printf "  at %s() in %s:%s\n" "$func" "$file" "$line"
    done

    # 或者手动遍历
    echo "=== 详细调用栈 ==="
    for (( i = 0; i < ${#FUNCNAME[@]}; i++ )); do
        printf "  [%d] %s() at %s:%d\n" \
            "$i" \
            "${FUNCNAME[$i]}" \
            "${BASH_SOURCE[$i]}" \
            "${BASH_LINENO[$i]}"
    done
}

# 错误处理中使用
die() {
    echo "FATAL: $1" >&2
    echo "  at ${FUNCNAME[1]}() in ${BASH_SOURCE[1]}:${BASH_LINENO[0]}" >&2
    exit 1
}
```

### 获取脚本自身路径

```bash
# 在被 source 的脚本中获取自身路径
# $0 不可靠（source 时 $0 是调用者）
# BASH_SOURCE[0] 始终是当前文件
get_script_dir() {
    local source="${BASH_SOURCE[0]}"
    while [ -L "$source" ]; do  # 解析符号链接
        local dir="$(cd "$(dirname "$source")" && pwd)"
        source="$(readlink "$source")"
        [[ "$source" != /* ]] && source="$dir/$source"
    done
    cd "$(dirname "$source")" && pwd
}
SCRIPT_DIR=$(get_script_dir)
```

## 8. 回调函数和高阶函数模式在 Shell 中怎么实现？

**回答：**

### 回调函数

```bash
# 将函数名作为参数传递
retry_with_callback() {
    local max=$1
    local on_fail=$2   # 回调函数名
    shift 2
    local cmd=("$@")

    for (( i = 1; i <= max; i++ )); do
        if "${cmd[@]}"; then
            return 0
        fi
        "$on_fail" "$i" "$max"   # 调用回调
    done
    return 1
}

# 回调函数
on_retry_fail() {
    echo "第 $1/$2 次重试失败，等待..." >&2
    sleep 2
}

retry_with_callback 3 on_retry_fail curl -sf http://example.com
```

### 高阶函数：map/filter/reduce

```bash
# map：对每个元素应用函数
map() {
    local func=$1
    shift
    for item in "$@"; do
        "$func" "$item"
    done
}

to_upper() { echo "${1^^}"; }
map to_upper hello world foo
# HELLO WORLD FOO

# filter：筛选元素
filter() {
    local func=$1
    shift
    for item in "$@"; do
        if "$func" "$item"; then
            echo "$item"
        fi
    done
}

is_file() { [ -f "$1" ]; }
filter is_file *   # 只输出文件（不含目录）

# reduce/fold
reduce() {
    local func=$1
    local acc=$2
    shift 2
    for item in "$@"; do
        acc=$("$func" "$acc" "$item")
    done
    echo "$acc"
}

sum() { echo $(( $1 + $2 )); }
reduce sum 0 1 2 3 4 5   # 15
```

### 钩子（Hook）模式

```bash
# 注册钩子
declare -a CLEANUP_HOOKS=()

register_cleanup() {
    CLEANUP_HOOKS+=("$1")
}

run_cleanup() {
    for hook in "${CLEANUP_HOOKS[@]}"; do
        "$hook" || true
    done
}
trap run_cleanup EXIT

# 各模块注册自己的清理逻辑
setup_temp() {
    TMPDIR=$(mktemp -d)
    register_cleanup "rm -rf $TMPDIR"
}

setup_lock() {
    touch /tmp/myapp.lock
    register_cleanup "rm -f /tmp/myapp.lock"
}

setup_temp
setup_lock
# 退出时自动按注册顺序清理
```

## 9. 如何实现函数的默认参数和参数校验？

**回答：**

### 默认参数

```bash
# 方式1：${var:-default}
greet() {
    local name=${1:-"World"}
    local greeting=${2:-"Hello"}
    echo "$greeting, $name!"
}
greet                  # Hello, World!
greet "Alice"          # Hello, Alice!
greet "Alice" "Hi"     # Hi, Alice!

# 方式2：条件赋值
deploy() {
    local env=${1:-production}
    local version=${2:-latest}
    local replicas=${3:-3}
    echo "部署 $version 到 $env，副本数 $replicas"
}
```

### 参数校验

```bash
# 简单校验
backup_db() {
    local db=${1:?  "错误：缺少数据库名"}   # 为空或未设置则报错退出
    local dest=${2:?"错误：缺少备份目标路径"}
    # ...
}

# 完整校验
create_user() {
    # 参数个数检查
    if (( $# < 2 )); then
        echo "用法: create_user <username> <email> [role]" >&2
        return 1
    fi

    local username=$1
    local email=$2
    local role=${3:-viewer}

    # 格式校验
    if [[ ! "$username" =~ ^[a-zA-Z][a-zA-Z0-9_]{2,31}$ ]]; then
        echo "错误：用户名格式无效" >&2
        return 1
    fi

    if [[ ! "$email" =~ ^[^@]+@[^@]+\.[^@]+$ ]]; then
        echo "错误：邮箱格式无效" >&2
        return 1
    fi

    # 值域校验
    case "$role" in
        admin|editor|viewer) ;;
        *) echo "错误：无效角色 '$role'" >&2; return 1 ;;
    esac

    echo "创建用户: $username ($email) 角色: $role"
}
```

### 命名参数模式

```bash
# 模拟命名参数
deploy_service() {
    local name="" image="" replicas=1 port=8080

    while [[ $# -gt 0 ]]; do
        case "$1" in
            --name)     name="$2"; shift 2 ;;
            --image)    image="$2"; shift 2 ;;
            --replicas) replicas="$2"; shift 2 ;;
            --port)     port="$2"; shift 2 ;;
            *)          echo "未知参数: $1" >&2; return 1 ;;
        esac
    done

    # 必选参数校验
    [[ -z "$name" ]] && { echo "错误：--name 必填" >&2; return 1; }
    [[ -z "$image" ]] && { echo "错误：--image 必填" >&2; return 1; }

    echo "部署 $name: image=$image replicas=$replicas port=$port"
}

deploy_service --name myapp --image nginx:latest --replicas 3
```

## 10. 如何组织大型 Shell 项目的代码结构？

### 推荐项目结构

```
my-project/
├── bin/
│   ├── deploy.sh           # 入口脚本
│   └── backup.sh
├── lib/

│   ├── config.sh           # 配置管理
│   ├── utils.sh            # 通用工具函数

│   ├── dev.env

│   └── prod.env

│   └── test_docker.sh
```

### 入口脚本模板

```bash
#!/bin/bash
# bin/deploy.sh
set -euo pipefail

# 获取项目根目录
PROJECT_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"

# 加载库
source "$PROJECT_ROOT/lib/logging.sh"
source "$PROJECT_ROOT/lib/config.sh"
source "$PROJECT_ROOT/lib/utils.sh"

# 加载环境配置
load_config "$PROJECT_ROOT/conf/${ENV:-dev}.env"

# 主函数模式
main() {
    log_info "开始部署..."
    parse_args "$@"
    validate_env
    do_deploy
    log_info "部署完成"
}

parse_args() {
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -e|--env)     ENV="$2"; shift 2 ;;
            -v|--version) VERSION="$2"; shift 2 ;;
            -h|--help)    usage; exit 0 ;;
            *)            usage; exit 1 ;;
        esac
    done
}

usage() {
    cat <<EOF
Usage: $(basename "$0") [OPTIONS]

Options:
    -e, --env ENV        目标环境 (dev|staging|prod)
    -v, --version VER    部署版本
    -h, --help           显示帮助
EOF
}

validate_env() {
    command -v docker >/dev/null || log_fatal "Docker 未安装"
    command -v kubectl >/dev/null || log_fatal "kubectl 未安装"
}

do_deploy() {
    # 部署逻辑...
    :
}

# 仅在直接执行时运行 main（被 source 时不执行）
if [[ "${BASH_SOURCE[0]}" == "${0}" ]]; then
    main "$@"
fi
```

### 关键原则

| 原则 | 说明 |
|------|------|
| 单一职责 | 每个文件/函数只做一件事 |
| main 函数模式 | 入口脚本用 `main()` 函数包装逻辑 |
| Source Guard | `[[ "${BASH_SOURCE[0]}" == "${0}" ]]` 区分执行/加载 |
| Include Guard | 防止库被重复 source |
| 配置分离 | 环境配置独立于代码 |
| 函数前缀 | 库函数加命名前缀避免冲突（如 `log_info`、`k8s_deploy`） |

## 11. 什么是 main 函数模式？为什么推荐使用？

**回答：**

```bash
#!/bin/bash
# main 函数模式的完整示例

set -euo pipefail

# 全局常量
readonly VERSION="1.0.0"
readonly PROG_NAME="$(basename "$0")"

# 函数定义
usage() {
    cat <<-EOF
	Usage: $PROG_NAME [options] <command>
	Version: $VERSION
	EOF
}

init() {
    # 初始化逻辑
    :
}

run() {
    # 核心业务逻辑
    :
}

cleanup() {
    # 清理资源
    :
}

# 主函数
main() {
    trap cleanup EXIT

    init
    parse_args "$@"
    run
}

# 关键：仅在直接执行时运行
if [[ "${BASH_SOURCE[0]}" == "${0}" ]]; then
    main "$@"
fi
```

**为什么推荐：**
1. **可测试性**：被 `source` 加载后可以单独调用和测试各个函数
2. **清晰的入口**：读者一眼就知道执行逻辑
3. **避免副作用**：函数定义阶段不执行任何操作
4. **可复用性**：其他脚本可以 `source` 此文件使用其中的函数
5. **变量隔离**：`main` 中的 `local` 变量不会污染全局

```bash
# 在测试中：
source ./deploy.sh    # 加载函数但不执行 main
# 现在可以测试单个函数
result=$(validate_input "test_data")
assert_equals "expected" "$result"
```

## 12. 如何用 Shell 实现简单的面向对象模式？

**回答：**

Shell 不支持面向对象，但可以用约定和技巧模拟：

### 命名空间模式

```bash
# 用前缀模拟命名空间
# --- stack.sh ---
declare -a _stack_data=()

stack_push() {
    _stack_data+=("$1")
}

stack_pop() {
    local last=${#_stack_data[@]}
    if (( last == 0 )); then
        return 1
    fi
    echo "${_stack_data[$((last-1))]}"
    unset '_stack_data[$((last-1))]'
}

stack_peek() {
    local last=${#_stack_data[@]}
    (( last > 0 )) && echo "${_stack_data[$((last-1))]}"
}

stack_size() {
    echo "${#_stack_data[@]}"
}
```

### 构造器模式（用关联数组模拟对象）

```bash
declare -A OBJECTS=()
_OBJ_ID=0

# 构造器
Server_new() {
    local name=$1
    local host=$2
    local port=${3:-22}

    (( _OBJ_ID++ ))
    local id="Server_${_OBJ_ID}"

    OBJECTS["${id}.name"]="$name"
    OBJECTS["${id}.host"]="$host"
    OBJECTS["${id}.port"]="$port"
    OBJECTS["${id}.status"]="unknown"

    echo "$id"  # 返回对象 ID
}

# 方法
Server_connect() {
    local self=$1
    local host="${OBJECTS["${self}.host"]}"
    local port="${OBJECTS["${self}.port"]}"
    echo "连接到 $host:$port..."
    OBJECTS["${self}.status"]="connected"
}

Server_get() {
    local self=$1
    local prop=$2
    echo "${OBJECTS["${self}.${prop}"]}"
}

# 使用
srv=$(Server_new "web01" "192.168.1.10" 22)
Server_connect "$srv"
echo "名称: $(Server_get "$srv" name)"
echo "状态: $(Server_get "$srv" status)"
```

> **注意：** 这种模式在 Shell 中并不常见，代码复杂度高。如果需要面向对象，建议使用 Python 等更合适的语言。Shell 的优势在于管道组合和系统调用，不在 OOP。
