## 1. set 命令有哪些常用的调试和安全选项？

**回答：**

### 核心选项

| 选项 | 含义 | 说明 |
|------|------|------|
| `set -e` (`errexit`) | 命令失败立即退出 | 管道、条件、逻辑运算中有例外 |
| `set -u` (`nounset`) | 引用未定义变量报错 | 防止拼写错误导致空变量 |
| `set -o pipefail` | 管道中任一命令失败则管道失败 | 默认只看最后一个命令 |
| `set -x` (`xtrace`) | 打印执行的每条命令 | 调试利器 |
| `set -v` (`verbose`) | 打印读取的每行输入 | 比 -x 更原始 |
| `set -f` (`noglob`) | 禁用文件名展开 | 防止通配符意外展开 |
| `set -n` (`noexec`) | 只检查语法不执行 | 快速语法检查 |

### 推荐的脚本头部

```bash
#!/bin/bash
set -euo pipefail    # 严格模式（Strict Mode）

# -e：任何命令失败立即退出
# -u：引用未设置变量报错退出
# -o pipefail：管道中任一命令失败则整个管道失败
```

### set -e 的例外情况

```bash
set -e

# 以下不会触发 -e 退出：
# 1. if/while/until 的条件部分
if false; then echo "no"; fi           # 不退出

# 2. ! 取反
! false                                 # 不退出

# 3. && / || 链（非最后一个）
false && echo "not reached"            # 不退出
false || echo "fallback"               # 不退出

# 4. 被明确检查返回值
cmd || true                            # 不退出
if cmd; then ...; fi                   # 不退出

# 注意：以下会退出
false                                   # 退出！
result=$(false)                         # 退出！
local result=$(false)  # 不退出！local 的返回值覆盖了 false
```

### set -x 调试输出

```bash
set -x
name="Alice"
echo "Hello $name"
# 输出：
# + name=Alice
# + echo 'Hello Alice'
# Hello Alice

# 自定义调试前缀
PS4='+(${BASH_SOURCE}:${LINENO}): ${FUNCNAME[0]:+${FUNCNAME[0]}(): }'
set -x
# 输出类似：+(script.sh:15): main(): echo 'Hello Alice'

# 局部调试
set -x
problematic_section
set +x   # 关闭调试
```

## 2. trap 命令怎么使用？有哪些常用信号？

**回答：**

`trap` 用于捕获信号或特殊事件并执行指定的命令。

### 基本语法

```bash
trap 'commands' SIGNAL [SIGNAL...]
trap -        SIGNAL     # 恢复默认行为
trap ''       SIGNAL     # 忽略信号
trap -p                  # 列出当前 trap 设置
trap -l                  # 列出所有信号
```

### 常用信号

| 信号 | 编号 | 触发时机 |
|------|------|----------|
| EXIT | 0 | 脚本退出（任何原因） |
| ERR | - | 命令返回非零退出码（需 set -e 或 set -o errtrace） |
| DEBUG | - | 每条命令执行前 |
| RETURN | - | 函数或 source 返回时 |
| INT | 2 | Ctrl+C |
| TERM | 15 | kill 默认信号 |
| HUP | 1 | 终端断开 |
| QUIT | 3 | Ctrl+\\ |
| USR1/USR2 | 10/12 | 用户自定义信号 |

### 常见模式

```bash
# 1. 清理临时文件
tmpfile=$(mktemp)
trap 'rm -f "$tmpfile"' EXIT

# 2. 多操作清理
cleanup() {
    echo "清理中..." >&2
    rm -f "$tmpfile"
    [ -n "${PID:-}" ] && kill "$PID" 2>/dev/null
    [ -f /tmp/myapp.lock ] && rm -f /tmp/myapp.lock
}
trap cleanup EXIT

# 3. 优雅退出（捕获 Ctrl+C）
graceful_exit() {
    echo ""
    echo "收到中断信号，正在优雅退出..."
    # 停止后台进程、保存状态等
    exit 130   # 128 + SIGINT(2) = 130，表示被信号终止
}
trap graceful_exit INT TERM

# 4. 错误追踪
trap 'echo "错误发生在第 $LINENO 行，命令: $BASH_COMMAND" >&2' ERR

# 5. 锁文件（防止并发执行）
LOCK_FILE="/tmp/myapp.lock"
trap 'rm -f "$LOCK_FILE"' EXIT
if ! (set -o noclobber; echo $$ > "$LOCK_FILE") 2>/dev/null; then
    echo "已有实例在运行 (PID: $(cat "$LOCK_FILE"))" >&2
    exit 1
fi
```

### trap 在函数中的行为

```bash
# trap ERR 默认不在函数中触发
set -e
trap 'echo "ERR caught"' ERR

fail() { false; }
fail    # 不触发 trap ERR！

# 开启 errtrace 使 ERR trap 在函数和子 Shell 中也生效
set -o errtrace   # 或 set -E
fail    # 现在触发 trap ERR

# 同理，trap DEBUG 需要 functrace
set -o functrace  # 或 set -T
```

## 3. 如何实现完整的错误处理框架？

**回答：**

```bash
#!/bin/bash
set -euo pipefail

# ============= 错误处理框架 =============

# 颜色
readonly RED='\033[0;31m'
readonly YELLOW='\033[0;33m'
readonly GREEN='\033[0;32m'
readonly NC='\033[0m'

# 日志函数
log_info()  { printf "${GREEN}[INFO]${NC}  %s\n" "$*" >&2; }
log_warn()  { printf "${YELLOW}[WARN]${NC}  %s\n" "$*" >&2; }
log_error() { printf "${RED}[ERROR]${NC} %s\n" "$*" >&2; }

# 错误处理器
on_error() {
    local exit_code=$?
    local line=$1
    local cmd=$2
    log_error "命令 '$cmd' 在第 $line 行失败 (退出码: $exit_code)"

    # 打印调用栈
    local frame=0
    log_error "调用栈:"
    while caller $frame; do
        (( frame++ ))
    done 2>/dev/null | while IFS=' ' read -r ln func file; do
        log_error "  at $func() in $file:$ln"
    done
}

# 清理函数
cleanup() {
    local exit_code=$?
    # 清理临时资源
    rm -f "${TMPFILES[@]}" 2>/dev/null || true
    [ -n "${LOCK_FILE:-}" ] && rm -f "$LOCK_FILE" 2>/dev/null || true

    if (( exit_code == 0 )); then
        log_info "脚本执行成功"
    else
        log_error "脚本执行失败 (退出码: $exit_code)"
    fi
}

# 注册 trap
trap 'on_error $LINENO "$BASH_COMMAND"' ERR
trap cleanup EXIT
trap 'log_warn "收到中断信号"; exit 130' INT TERM

# 临时文件管理
declare -a TMPFILES=()
make_temp() {
    local tmp
    tmp=$(mktemp)
    TMPFILES+=("$tmp")
    echo "$tmp"
}

# 安全执行函数
run() {
    log_info "执行: $*"
    if "$@"; then
        log_info "成功: $*"
    else
        local rc=$?
        log_error "失败: $* (退出码: $rc)"
        return $rc
    fi
}

# ============= 业务逻辑 =============
main() {
    log_info "开始执行..."

    local tmp=$(make_temp)

    run curl -sSf http://example.com -o "$tmp"
    run process_data "$tmp"

    log_info "执行完成"
}

main "$@"
```

## 4. ShellCheck 是什么？能检查哪些问题？

**回答：**

ShellCheck 是 Shell 脚本的**静态分析工具**，能检测语法错误、语义问题和最佳实践违规。

### 安装和使用

```bash
# 安装
apt install shellcheck       # Debian/Ubuntu
yum install ShellCheck       # CentOS/RHEL
brew install shellcheck      # macOS

# 使用
shellcheck script.sh
shellcheck -s bash script.sh    # 指定 Shell 方言
shellcheck -x script.sh         # 跟踪 source 的文件
shellcheck -f diff script.sh    # diff 格式（可自动修复）
shellcheck -e SC2086 script.sh  # 排除特定规则
```

### 常见检测规则

```bash
# SC2086: 变量未加双引号
echo $var            # 警告
echo "$var"          # 正确

# SC2046: 命令替换未加双引号
files=$(find . -name "*.txt")
for f in $files; do  # 警告
for f in $(find . -name "*.txt"); do  # 也警告
# 推荐：while + read -r

# SC2034: 变量已赋值但未使用
unused_var="hello"   # 警告

# SC2155: declare 和赋值分开
local var=$(cmd)     # 警告（local 的返回值覆盖 cmd 的）
local var             # 正确
var=$(cmd)

# SC2181: 直接用命令做条件，不用 $?
cmd
if [ $? -eq 0 ]; then  # 不推荐
if cmd; then            # 推荐

# SC2164: cd 没有检查失败
cd /some/dir            # 警告
cd /some/dir || exit 1  # 正确

# SC2006: 用 $() 替代反引号
var=`cmd`              # 不推荐
var=$(cmd)             # 推荐
```

### CI 集成

```yaml
# GitHub Actions
- name: ShellCheck
  uses: ludeeus/action-shellcheck@master
  with:
    scandir: './scripts'
    severity: warning

# 本地 pre-commit
# .pre-commit-config.yaml
repos:
  - repo: https://github.com/koalaman/shellcheck-precommit
    rev: v0.9.0
    hooks:
      - id: shellcheck
```

### 行内禁用

```bash
# 禁用单行
# shellcheck disable=SC2086
echo $var

# 禁用整个文件
# shellcheck disable=SC2086,SC2046

# 禁用代码块
# shellcheck disable=SC2086
{
    echo $var1
    echo $var2
}
```

## 5. 如何调试复杂脚本？有哪些调试技巧？

**回答：**

### 1. bash -x 执行追踪

```bash
# 整个脚本
bash -x script.sh

# 脚本内局部追踪
set -x
complex_logic_here
set +x

# 自定义 PS4 显示更多信息
export PS4='+(${BASH_SOURCE[0]}:${LINENO}): ${FUNCNAME[0]:+${FUNCNAME[0]}(): }'
bash -x script.sh
# 输出：+(script.sh:42): deploy(): curl -sSf http://...
```

### 2. 输出到调试日志

```bash
# 将 set -x 输出重定向到文件（不影响 stdout/stderr）
exec 5> /tmp/debug.log
BASH_XTRACEFD=5
set -x

echo "正常输出"     # stdout
echo "错误信息" >&2  # stderr
# set -x 的调试信息写入 /tmp/debug.log 而不是 stderr
```

### 3. 条件调试

```bash
# DEBUG 环境变量控制
debug() {
    if [[ "${DEBUG:-0}" == "1" ]]; then
        printf "[DEBUG] %s\n" "$*" >&2
    fi
}

debug "变量值: var=$var"

# 使用
DEBUG=1 ./script.sh
```

### 4. trap DEBUG 逐步追踪

```bash
# 每条命令执行前打印信息
trap 'echo "即将执行: $BASH_COMMAND (行 $LINENO)"' DEBUG

# 交互式断点
breakpoint() {
    echo "=== 断点 (${FUNCNAME[1]}:${BASH_LINENO[0]}) ===" >&2
    echo "按 Enter 继续，输入 'q' 退出" >&2
    local cmd
    while read -rp "debug> " cmd; do
        case "$cmd" in
            q|quit) exit 1 ;;
            "") break ;;       # Enter 继续
            *)  eval "$cmd" ;;  # 执行调试命令（注意安全性）
        esac
    done
}

# 在代码中插入断点
process_data
breakpoint    # 程序暂停，可检查变量
deploy
```

### 5. 变量状态检查

```bash
# 打印所有变量
declare -p             # 所有变量
declare -p var_name    # 特定变量
declare -p | grep MY_  # 过滤前缀

# 打印函数定义
declare -f func_name
type func_name

# 检查变量是否设置
[[ -v var_name ]] && echo "已设置" || echo "未设置"
```

### 6. 子 Shell 调试

```bash
# 子 Shell 中的错误不好追踪
# 使用 set -E 传播 ERR trap
set -eE
trap 'echo "ERR at line $LINENO"' ERR

# 管道调试
cmd1 | cmd2 | cmd3
echo "PIPESTATUS: ${PIPESTATUS[*]}"
```

## 6. set -e 有哪些"坑"？如何正确使用？

**回答：**

### 坑1：local 掩盖错误

```bash
set -e

# local 的返回值覆盖了 $(cmd) 的返回值
func() {
    local result=$(false)  # 不退出！local 返回 0
    echo "这行会执行"
}

# 修复：分开声明和赋值
func() {
    local result
    result=$(false)         # 退出！
}
```

### 坑2：命令替换中的子 Shell

```bash
set -e

# 在 $() 中的错误不会传播到外部（某些 Bash 版本）
output=$(
    false
    echo "after false"    # 可能执行也可能不执行
)
```

### 坑3：条件链中的行为

```bash
set -e

# 以下不触发退出
false && true   # 不退出
false || true   # 不退出
if false; then true; fi  # 不退出

# 但是！
func() { false; }

# 直接调用
func          # 退出

# 在条件中
if func; then echo "ok"; fi   # 不退出
func && echo "ok"             # 不退出

# 注意：在 && 左边不退出，但 && 右边的错误会退出
true && false   # 退出！
```

### 坑4：数组遍历

```bash
set -e
arr=()
echo "${arr[@]}"   # 在某些 Bash 版本中，空数组会触发 -u 错误
echo "${arr[@]+}"  # 安全写法：${var+word}
```

### 最佳实践

```bash
# 1. 明确允许失败的命令
set -e
cmd_may_fail || true   # 明确忽略失败
if ! cmd_may_fail; then
    echo "失败了但可以继续"
fi

# 2. 分开 local 和赋值
local var
var=$(command)

# 3. 配合 pipefail
set -eo pipefail

# 4. 关键代码段临时关闭 -e
set +e
risky_command
rc=$?
set -e
if (( rc != 0 )); then
    handle_error
fi
```

## 7. 如何实现脚本的日志记录系统？

**回答：**

### 简单日志函数

```bash
#!/bin/bash

LOG_FILE="${LOG_FILE:-/var/log/myapp.log}"
LOG_LEVEL="${LOG_LEVEL:-INFO}"

# 日志级别数值
declare -A LOG_LEVELS=(
    [DEBUG]=0 [INFO]=1 [WARN]=2 [ERROR]=3 [FATAL]=4
)

log() {
    local level=$1
    shift
    local msg="$*"

    # 级别过滤
    local level_num=${LOG_LEVELS[$level]:-1}
    local current_num=${LOG_LEVELS[$LOG_LEVEL]:-1}
    (( level_num < current_num )) && return 0

    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    local caller_info="${BASH_SOURCE[1]##*/}:${BASH_LINENO[0]}"
    local func="${FUNCNAME[1]:-main}"

    local line="[$timestamp] [$level] [$func@$caller_info] $msg"

    # 输出到终端（带颜色）
    case "$level" in
        DEBUG) printf "\033[0;37m%s\033[0m\n" "$line" >&2 ;;
        INFO)  printf "\033[0;32m%s\033[0m\n" "$line" >&2 ;;
        WARN)  printf "\033[0;33m%s\033[0m\n" "$line" >&2 ;;
        ERROR) printf "\033[0;31m%s\033[0m\n" "$line" >&2 ;;
        FATAL) printf "\033[1;31m%s\033[0m\n" "$line" >&2 ;;
    esac

    # 写入日志文件（无颜色）
    echo "$line" >> "$LOG_FILE"
}

log_debug() { log DEBUG "$@"; }
log_info()  { log INFO "$@"; }
log_warn()  { log WARN "$@"; }
log_error() { log ERROR "$@"; }
log_fatal() { log FATAL "$@"; exit 1; }

# 使用
log_info "脚本启动"
log_debug "调试信息: var=$var"
log_error "连接失败: $host"
log_fatal "无法继续"   # 会 exit 1
```

### 日志轮转

```bash
rotate_log() {
    local log_file=$1
    local max_size=${2:-10485760}   # 默认 10MB
    local max_files=${3:-5}

    local size
    size=$(stat -f%z "$log_file" 2>/dev/null || stat --format=%s "$log_file" 2>/dev/null || echo 0)

    if (( size > max_size )); then
        for (( i = max_files - 1; i > 0; i-- )); do
            local prev=$((i - 1))
            [ -f "${log_file}.$prev" ] && mv "${log_file}.$prev" "${log_file}.$i"
        done
        mv "$log_file" "${log_file}.0"
        : > "$log_file"   # 创建新的空日志
    fi
}
```

## 8. 如何实现脚本的超时控制和进程管理？

**回答：**

### timeout 命令

```bash
# 基本用法
timeout 30 long_running_command

# 指定信号
timeout --signal=KILL 10 command   # 10秒后发 SIGKILL

# 两阶段超时：先 TERM，等5秒后 KILL
timeout --kill-after=5 30 command

# 检查超时
timeout 10 command
if (( $? == 124 )); then
    echo "命令超时"
fi
```

### 手动实现超时

```bash
# 后台执行 + 超时检查
run_with_timeout() {
    local timeout=$1
    shift

    "$@" &
    local pid=$!

    (
        sleep "$timeout"
        if kill -0 "$pid" 2>/dev/null; then
            echo "超时，终止进程 $pid" >&2
            kill -TERM "$pid" 2>/dev/null
            sleep 2
            kill -KILL "$pid" 2>/dev/null
        fi
    ) &
    local watchdog=$!

    wait "$pid"
    local rc=$?

    kill "$watchdog" 2>/dev/null
    wait "$watchdog" 2>/dev/null

    return $rc
}

run_with_timeout 30 curl -sSf http://slow-server.com/api
```

### 等待条件满足

```bash
wait_for() {
    local description=$1
    local timeout=${2:-60}
    local interval=${3:-2}
    shift 3
    local cmd=("$@")

    local elapsed=0
    log_info "等待 $description (超时: ${timeout}s)..."

    while ! "${cmd[@]}" > /dev/null 2>&1; do
        if (( elapsed >= timeout )); then
            log_error "$description 超时"
            return 1
        fi
        sleep "$interval"
        (( elapsed += interval ))
    done

    log_info "$description 就绪 (耗时: ${elapsed}s)"
}

# 使用
wait_for "MySQL启动" 60 2 mysqladmin ping -h localhost
wait_for "端口8080可用" 30 1 bash -c "echo > /dev/tcp/localhost/8080"
```

## 9. exit、return、exit code 有什么区别和约定？

**回答：**

### exit vs return

| 命令 | 作用范围 | 用途 |
|------|----------|------|
| `exit N` | 终止整个脚本/子 Shell | 脚本结束，返回状态码 |
| `return N` | 从函数返回 | 函数结束，返回状态码 |

```bash
# exit 在函数中会终止整个脚本
func() {
    exit 1     # 脚本结束！
}

func() {
    return 1   # 只从函数返回
}

# 在 source 的文件中
# exit 会终止调用者的 Shell！
# return 才是正确的（从 source 的上下文返回）
```

### 退出码约定

| 退出码 | 含义 |
|--------|------|
| 0 | 成功 |
| 1 | 一般性错误 |
| 2 | 命令使用错误（参数错误等） |
| 126 | 命令不可执行（权限问题） |
| 127 | 命令未找到 |
| 128 | 无效退出码 |
| 128+N | 被信号 N 终止 |
| 130 | 被 SIGINT (Ctrl+C) 终止（128+2） |
| 137 | 被 SIGKILL 终止（128+9） |
| 143 | 被 SIGTERM 终止（128+15） |
| 255 | 退出码超出范围 |

### 自定义退出码规范

```bash
# 定义退出码常量
readonly EXIT_SUCCESS=0
readonly EXIT_GENERAL_ERROR=1
readonly EXIT_USAGE_ERROR=2
readonly EXIT_CONFIG_ERROR=3
readonly EXIT_NETWORK_ERROR=4
readonly EXIT_PERMISSION_ERROR=5
readonly EXIT_DEPENDENCY_ERROR=6

# 使用
main() {
    check_deps || exit $EXIT_DEPENDENCY_ERROR
    load_config || exit $EXIT_CONFIG_ERROR
    deploy || exit $EXIT_GENERAL_ERROR
    exit $EXIT_SUCCESS
}
```

### $? 的使用

```bash
command
echo $?   # 上一条命令的退出码

# 注意：$? 会被任何命令覆盖
false
echo "status: $?"    # $? 已经被 echo 改变了
# 正确做法
false
rc=$?
echo "status: $rc"

# 在 set -e 模式下安全检查
if ! command; then
    echo "失败" >&2
fi
# 或
command || {
    echo "失败" >&2
    exit 1
}
```

## 10. 如何实现脚本的单例运行（防止并发执行）？

**回答：**

### 方法1：锁文件 + flock

```bash
#!/bin/bash
LOCK_FILE="/var/lock/myapp.lock"
LOCK_FD=200

# 打开锁文件描述符
eval "exec $LOCK_FD>$LOCK_FILE"

# 尝试获取排他锁（非阻塞）
if ! flock -n $LOCK_FD; then
    echo "另一个实例正在运行" >&2
    exit 1
fi

# 写入 PID
echo $$ >&$LOCK_FD

# 脚本正常执行...
# 退出时锁自动释放（fd 关闭）
```

### 方法2：PID 文件

```bash
PID_FILE="/var/run/myapp.pid"

check_running() {
    if [ -f "$PID_FILE" ]; then
        local old_pid
        old_pid=$(cat "$PID_FILE")
        if kill -0 "$old_pid" 2>/dev/null; then
            echo "实例已运行 (PID: $old_pid)" >&2
            return 1
        else
            echo "清理过期 PID 文件" >&2
            rm -f "$PID_FILE"
        fi
    fi
    echo $$ > "$PID_FILE"
    trap 'rm -f "$PID_FILE"' EXIT
    return 0
}

check_running || exit 1
```

## 11. 如何优雅地处理用户中断（Ctrl+C）和信号？

**回答：**

```bash
#!/bin/bash

# 标记是否正在清理
CLEANING_UP=false

cleanup() {
    # 防止重复清理
    $CLEANING_UP && return
    CLEANING_UP=true

    echo ""
    echo "正在清理..."

    # 停止后台进程
    if [ -n "${WORKER_PID:-}" ]; then
        kill "$WORKER_PID" 2>/dev/null
        wait "$WORKER_PID" 2>/dev/null
    fi

    # 删除临时文件
    rm -f "$TMP_FILE" 2>/dev/null

    # 恢复终端设置（如果修改过）
    stty sane 2>/dev/null

    echo "清理完成"
}

# EXIT 捕获所有退出情况
trap cleanup EXIT
# INT/TERM 设置退出码
trap 'exit 130' INT     # Ctrl+C
trap 'exit 143' TERM    # kill

# 主逻辑
TMP_FILE=$(mktemp)
long_running_task &
WORKER_PID=$!

# 等待完成
wait "$WORKER_PID"
```

### 批处理中的中断处理

```bash
# 允许当前任务完成后退出
INTERRUPTED=false
trap 'INTERRUPTED=true; echo "将在当前任务完成后退出..."' INT

for task in task1 task2 task3 task4 task5; do
    if $INTERRUPTED; then
        echo "用户中断，跳过剩余任务"
        break
    fi

    echo "执行: $task"
    process_task "$task"
done

$INTERRUPTED && exit 130
```
