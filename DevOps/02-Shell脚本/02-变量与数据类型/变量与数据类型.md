## 一、变量基础

### 1. Shell 变量的定义和使用规则？

**答：**

```bash
# ===== 变量定义 =====
# ⚠️ 等号两边不能有空格！
name="DevOps"                  # ✓ 正确
name = "DevOps"                # ✗ 错误！被解释为命令 name 参数 = DevOps

# 变量命名规则:
# - 字母、数字、下划线
# - 不能以数字开头
# - 区分大小写
# - 约定: 环境变量大写，局部变量小写/驼峰
MY_GLOBAL="env var"
local_var="local var"

# ===== 变量引用 =====
echo $name                     # 简单引用
echo ${name}                   # 推荐（避免歧义）
echo "${name}_suffix"          # 加花括号区分边界
echo "$name_suffix"            # 错！找的是 $name_suffix 变量

# ===== 变量赋值的多种方式 =====
# 直接赋值
var="hello"

# 命令替换
today=$(date +%F)
files=$(ls *.txt 2>/dev/null)

# 读取输入
read -p "Enter name: " user_name
read -s -p "Password: " password   # -s 静默（不显示输入）
read -t 5 -p "Quick: " answer      # -t 超时

# 默认值
echo ${var:-default}           # var 为空或未定义时返回 default
echo ${var:=default}           # var 为空或未定义时赋值并返回
echo ${var:+alternate}         # var 非空时返回 alternate
echo ${var:?error message}     # var 为空时报错退出

# 与不带冒号的区别
echo ${var-default}            # 仅未定义时（空字符串不触发）
echo ${var=default}            # 仅未定义时赋值
```

### 2. 特殊变量详解？

**答：**

```bash
# ===== 位置参数 =====
$0              # 脚本名称（含路径）
$1 ~ $9         # 第 1~9 个参数
${10}           # 第 10 个起必须用花括号
$#              # 参数个数
$@              # 所有参数（"$@" 保持独立）
$*              # 所有参数（"$*" 合并为一个）

# ===== 特殊状态变量 =====
$?              # 上一条命令的退出码 (0=成功，1-255=失败)
$$              # 当前 Shell 的 PID
$!              # 最近一个后台进程的 PID
$-              # 当前 Shell 的选项标志 (如 himBHs)
$_              # 上一条命令的最后一个参数

# ===== 实用场景 =====
#!/bin/bash
echo "脚本名: $0"
echo "参数数量: $#"
echo "所有参数: $@"
echo "第一个参数: $1"
echo "当前PID: $$"

# shift 移动位置参数
while [[ $# -gt 0 ]]; do
    echo "当前参数: $1"
    shift                   # $1=$2, $2=$3, ... $#=(n-1)
done

# $? 用法
grep "pattern" file
if [[ $? -eq 0 ]]; then
    echo "找到了"
fi
# 简写:
if grep -q "pattern" file; then
    echo "找到了"
fi

# $! 用法（等待后台进程）
long_task &
bg_pid=$!
echo "后台任务 PID: $bg_pid"
wait $bg_pid
echo "后台任务完成，退出码: $?"
```

### 3. declare / typeset 变量声明？

**答：**

```bash
# declare 可以声明变量类型和属性

# 整数变量
declare -i num=42
num=num+8                  # 自动算术运算
echo $num                  # 50
num="hello"                # 赋值为 0（不是字符串）

# 只读变量
declare -r CONST="固定值"
CONST="新值"               # 报错: readonly variable
# 等价于 readonly CONST="固定值"

# 大写转换
declare -u upper_var
upper_var="hello"
echo $upper_var            # HELLO

# 小写转换
declare -l lower_var
lower_var="HELLO"
echo $lower_var            # hello

# 数组
declare -a arr=(1 2 3)     # 索引数组

# 关联数组
declare -A map             # 关联数组（必须显式声明）
map[key]="value"

# 导出为环境变量
declare -x MY_VAR="exported"  # 等价于 export

# 查看变量属性
declare -p my_var           # 显示变量类型和值

# 列出特定类型的变量
declare -i                  # 所有整数变量
declare -r                  # 所有只读变量
declare -x                  # 所有导出的环境变量
declare -f                  # 所有函数定义
declare -F                  # 所有函数名
```

## 二、字符串操作

### 4. 字符串操作完整指南？

**答：**

```bash
str="Hello, DevOps World!"

# ===== 长度 =====
echo ${#str}                    # 20
echo ${#str[@]}                 # 1（字符串不是数组时）
expr length "$str"              # 20（外部命令方式）

# ===== 截取子串 =====
echo ${str:0:5}                 # Hello（从0开始，取5个）
echo ${str:7:6}                 # DevOps
echo ${str:7}                   # DevOps World!（从7开始到结尾）
echo ${str: -6}                 # orld!（从末尾倒数，冒号后有空格）
echo ${str: -6:3}               # orl

# ===== 删除/截取模式 =====
file="/usr/local/bin/script.sh"

# 从左删除（#）
echo ${file#*/}                 # usr/local/bin/script.sh（最短匹配）
echo ${file##*/}                # script.sh（最长匹配 = basename）

# 从右删除（%）
echo ${file%/*}                 # /usr/local/bin（最短匹配 = dirname）
echo ${file%%/*}                #（空，最长匹配）

# 实用：提取文件名和扩展名
filename=${file##*/}            # script.sh
basename=${filename%.*}         # script
extension=${filename##*.}       # sh
dirpath=${file%/*}              # /usr/local/bin

# ===== 替换 =====
str="foo bar foo baz foo"
echo ${str/foo/FOO}             # FOO bar foo baz foo（替换第一个）
echo ${str//foo/FOO}            # FOO bar FOO baz FOO（替换所有）
echo ${str/#foo/FOO}            # FOO bar foo baz foo（替换开头）
echo ${str/%foo/FOO}            # foo bar foo baz FOO（替换结尾）

# ===== 大小写转换 (Bash 4+) =====
str="Hello World"
echo ${str^^}                   # HELLO WORLD（全大写）
echo ${str,,}                   # hello world（全小写）
echo ${str^}                    # Hello World（首字母大写）
echo ${str~}                    # hELLO wORLD（首字母翻转）
echo ${str~~}                   # hELLO wORLD（全部翻转）

# ===== 间接引用 =====
var_name="greeting"
greeting="Hello!"
echo ${!var_name}               # Hello!（间接引用）

# ===== 变量名前缀匹配 =====
APP_NAME="myapp"
APP_PORT=8080
APP_HOST="localhost"
echo ${!APP_*}                  # APP_HOST APP_NAME APP_PORT
```

### 5. 字符串判断和比较？

**答：**

```bash
# ===== [ ] 中的字符串比较 =====
[ "$a" = "$b" ]              # 等于
[ "$a" != "$b" ]             # 不等于
[ -z "$a" ]                  # 长度为零（空或未定义）
[ -n "$a" ]                  # 长度非零（非空）

# ===== [[ ]] 中的字符串比较（推荐） =====
[[ "$a" == "$b" ]]           # 等于
[[ "$a" != "$b" ]]           # 不等于
[[ "$a" < "$b" ]]            # 字典序小于
[[ "$a" > "$b" ]]            # 字典序大于
[[ "$a" == pattern* ]]       # 通配符匹配
[[ "$a" =~ ^[0-9]+$ ]]      # 正则匹配

# ===== 实用判断 =====
# 判断字符串是否为数字
if [[ "$str" =~ ^[0-9]+$ ]]; then
    echo "纯数字"
fi

# 判断是否包含子串
if [[ "$str" == *"substring"* ]]; then
    echo "包含"
fi

# 判断是否以某个前缀开头
if [[ "$str" == prefix* ]]; then
    echo "以 prefix 开头"
fi

# 判断变量是否已定义
if [[ -v var_name ]]; then       # Bash 4.2+
    echo "已定义"
fi

# 判断变量是否为空
if [[ -z "${var:-}" ]]; then     # 安全方式（set -u 下不报错）
    echo "为空或未定义"
fi
```

## 三、数值运算

### 6. Shell 中的数值运算方式？

**答：**

```bash
# ===== 方式一: $(( )) 算术展开（推荐） =====
echo $((3 + 5))                # 8
echo $((10 - 3))               # 7
echo $((4 * 5))                # 20
echo $((10 / 3))               # 3（整数除法）
echo $((10 % 3))               # 1（取余）
echo $((2 ** 10))              # 1024（幂）

a=10; b=3
echo $((a + b))                # 13（不需要 $）
echo $((a > b ? a : b))        # 10（三元运算符）

# 位运算
echo $((8 >> 1))               # 4（右移）
echo $((1 << 4))               # 16（左移）
echo $((0xFF & 0x0F))          # 15（按位与）
echo $((0xFF | 0x100))         # 511（按位或）
echo $((~0))                   # -1（按位取反）

# 自增自减
((a++))                        # a = a + 1
((a--))                        # a = a - 1
((a += 5))                     # a = a + 5
((a *= 2))                     # a = a * 2

# ===== 方式二: let =====
let "result = 3 + 5"
let "a++"
let "a = a * 2"

# ===== 方式三: expr（外部命令，慢） =====
expr 3 + 5                     # 运算符前后必须有空格
expr 4 \* 5                    # 乘号需要转义
result=$(expr $a + $b)

# ===== 方式四: bc（支持浮点数） =====
echo "3.14 * 2" | bc           # 6.28
echo "scale=3; 10/3" | bc      # 3.333
echo "sqrt(144)" | bc          # 12

# 在脚本中使用 bc
pi=$(echo "scale=10; 4*a(1)" | bc -l)  # 计算 π
echo $pi                       # 3.1415926535

# ===== 方式五: awk（浮点数另一种方式） =====
awk 'BEGIN {printf "%.2f\n", 10/3}'     # 3.33

# ===== 进制转换 =====
echo $((16#FF))                # 十六进制 → 十进制: 255
echo $((2#1010))               # 二进制 → 十进制: 10
echo $((8#77))                 # 八进制 → 十进制: 63
printf '%x\n' 255              # 十进制 → 十六进制: ff
printf '%o\n' 255              # 十进制 → 八进制: 377
printf '%d\n' 0xFF             # 十六进制 → 十进制: 255
```

## 四、数组

### 7. 索引数组的完整操作？

**答：**

```bash
# ===== 定义 =====
arr=()                          # 空数组
arr=(apple banana cherry)       # 初始化
arr=([0]=a [2]=c [5]=f)         # 稀疏数组
declare -a arr                  # 显式声明

# ===== 访问 =====
echo ${arr[0]}                  # 第一个元素
echo ${arr[-1]}                 # 最后一个（Bash 4.2+）
echo ${arr[@]}                  # 所有元素（空格分隔）
echo ${arr[*]}                  # 类似，但引号行为不同
echo ${#arr[@]}                 # 数组长度
echo ${!arr[@]}                 # 所有有效索引

# ===== 修改 =====
arr[3]="date"                   # 赋值
arr+=(fig grape)                # 追加元素
unset arr[1]                    # 删除元素（变稀疏，不会重排索引）
unset arr                       # 删除整个数组

# ===== 切片 =====
echo ${arr[@]:1:3}              # 从索引1开始取3个
echo ${arr[@]:2}                # 从索引2到末尾

# ===== 遍历 =====
# 按值遍历（推荐加引号）
for item in "${arr[@]}"; do
    echo "$item"
done

# 按索引遍历
for i in "${!arr[@]}"; do
    echo "arr[$i] = ${arr[$i]}"
done

# ===== 数组操作技巧 =====
# 复制数组
arr2=("${arr[@]}")

# 合并数组
merged=("${arr1[@]}" "${arr2[@]}")

# 反转数组
reversed=()
for ((i=${#arr[@]}-1; i>=0; i--)); do
    reversed+=("${arr[$i]}")
done

# 查找元素
contains() {
    local target="$1"; shift
    for item in "$@"; do
        [[ "$item" == "$target" ]] && return 0
    done
    return 1
}
if contains "banana" "${arr[@]}"; then
    echo "找到了"
fi

# 从命令输出创建数组
files=($(ls *.txt 2>/dev/null))    # ⚠️ 有空格文件名问题
# 安全方式:
mapfile -t files < <(find . -name "*.txt")  # Bash 4+
readarray -t lines < file.txt               # readarray = mapfile
```

### 8. 关联数组（字典/Map）的使用？

**答：**

```bash
# 关联数组需要显式声明（Bash 4+）
declare -A config

# ===== 赋值 =====
config[host]="localhost"
config[port]=8080
config[env]="production"

# 批量初始化
declare -A colors=(
    [red]="#FF0000"
    [green]="#00FF00"
    [blue]="#0000FF"
)

# ===== 访问 =====
echo ${config[host]}            # localhost
echo ${config[@]}               # 所有值
echo ${!config[@]}              # 所有键
echo ${#config[@]}              # 元素数量

# ===== 遍历 =====
for key in "${!config[@]}"; do
    echo "$key = ${config[$key]}"
done

# ===== 判断 key 是否存在 =====
if [[ -v config[host] ]]; then    # Bash 4.2+
    echo "存在"
fi

# 或
if [[ ${config[host]+_} ]]; then
    echo "存在"
fi

# ===== 删除 =====
unset config[port]              # 删除元素
unset config                    # 删除整个数组

# ===== 实用场景 =====
# 统计词频
declare -A word_count
while read -r word; do
    ((word_count[$word]++))
done < <(tr ' ' '\n' < text.txt)

for word in "${!word_count[@]}"; do
    echo "${word_count[$word]} $word"
done | sort -rn | head -10

# 配置文件解析
declare -A config
while IFS='=' read -r key value; do
    [[ "$key" =~ ^[[:space:]]*# ]] && continue   # 跳过注释
    [[ -z "$key" ]] && continue                    # 跳过空行
    config[${key// /}]="${value// /}"               # 去空格
done < config.ini

# 简单的 JSON 解析到关联数组
declare -A json_data
while IFS=': ' read -r key value; do
    key=${key//\"/}
    value=${value//\"/}
    value=${value%,}
    json_data[$key]=$value
done < <(grep -oP '"[^"]+"\s*:\s*"[^"]*"' data.json)
```

## 五、变量作用域

### 9. 变量作用域的完整规则？

**答：**

```bash
# ===== 全局变量 =====
global_var="I'm global"

func() {
    echo $global_var         # 可以访问
    global_var="modified"    # 可以修改
    new_global="created"     # 在函数中创建的也是全局
}
func
echo $global_var             # modified
echo $new_global             # created

# ===== 局部变量 (local) =====
func() {
    local local_var="I'm local"
    echo $local_var          # 可以访问
}
func
echo $local_var              # 空（函数外不可见）

# ⚠️ local 只能在函数内使用

# ===== 环境变量 (export) =====
export env_var="I'm exported"
# 子进程（包括脚本中调用的外部命令）可以继承
bash -c 'echo $env_var'     # I'm exported

# 非 export 的变量不会传给子进程
shell_var="not exported"
bash -c 'echo $shell_var'   # 空

# ===== 子Shell变量隔离 =====
parent_var="parent"
(
    sub_var="sub"
    parent_var="modified in sub"
    echo "In subshell: $parent_var"   # modified in sub
)
echo "In parent: $parent_var"          # parent（没被修改！）
echo "sub_var: $sub_var"               # 空

# ===== 变量作用域总结 =====
# ┌──────────────┬─────────┬─────────┬─────────┐
# │ 类型          │ 函数内   │ 子Shell │ 子进程   │
# ├──────────────┼─────────┼─────────┼─────────┤
# │ 普通变量      │ ✓ 可见   │ ✓ 可见  │ ✗ 不可见 │
# │ local 变量   │ ✓ 当前函数│ ✗      │ ✗       │
# │ export 变量  │ ✓ 可见   │ ✓ 可见  │ ✓ 可见   │
# │ 子Shell中的   │ -       │ ✓ 当前  │ ✗       │
# └──────────────┴─────────┴─────────┴─────────┘
```

### 10. 变量的间接引用和 nameref？

**答：**

```bash
# ===== 间接引用 ${!var} =====
key="name"
name="DevOps"
echo ${!key}                 # DevOps

# 动态变量名
for i in 1 2 3; do
    declare "server_${i}=10.0.0.${i}"
done
for i in 1 2 3; do
    varname="server_${i}"
    echo "${!varname}"       # 10.0.0.1, 10.0.0.2, 10.0.0.3
done

# ===== nameref (Bash 4.3+) =====
# declare -n 创建引用（类似指针）
declare -n ref=name
name="Original"
echo $ref                    # Original
ref="Modified"
echo $name                   # Modified

# 函数中使用 nameref（传引用）
modify_var() {
    local -n var_ref=$1       # 引用传入的变量名
    var_ref="new value"
}

my_var="old value"
modify_var my_var
echo $my_var                 # new value

# 遍历关联数组的 nameref
print_map() {
    local -n map=$1
    for key in "${!map[@]}"; do
        echo "$key: ${map[$key]}"
    done
}
declare -A config=([host]=localhost [port]=8080)
print_map config
```
