## 1. Shell 脚本的编码规范和风格指南？

**回答：**

### Google Shell Style Guide 核心要点

```bash
#!/bin/bash
# 文件描述：简要说明脚本用途

set -euo pipefail

# ======== 命名规范 ========

# 变量：小写 + 下划线
my_variable="value"
readonly CONSTANT_VALUE="fixed"   # 常量：大写 + 下划线

# 函数：小写 + 下划线
my_function() {
    local local_var="value"    # 局部变量用 local
    # ...
}

# 环境变量/导出变量：大写
export DATABASE_URL="..."

# 文件名：小写 + 下划线或连字符
# my_script.sh 或 my-script.sh

# ======== 格式规范 ========

# 缩进：2 个空格（非 Tab）
if [[ "$condition" == "true" ]]; then
  do_something
fi

# 行长度：不超过 80 字符
# 长命令换行用 \
docker run \
  --name myapp \
  --restart always \
  -p 8080:80 \
  nginx:latest

# 管道换行
command1 \
  | command2 \
  | command3

# ======== 注释规范 ========

# 函数注释
#######################################
# 描述函数功能
# Globals:
#   DATABASE_URL
# Arguments:
#   $1 - 用户名
#   $2 - 邮箱
# Outputs:
#   写入 stdout
# Returns:
#   0 成功，非 0 失败
#######################################
create_user() {
    local username=$1
    local email=$2
    # ...
}

# 行内注释：至少空两格
echo "hello"  # 这是注释

# TODO 注释
# TODO(username): 需要优化这个循环
```

### 推荐做法

| 做 | 不做 |
|-----|------|
| `[[ ]]` 测试 | `[ ]` 测试（除 POSIX sh）|
| `$(command)` 命令替换 | `` `command` `` 反引号 |
| `${var}` 显式引用 | `$var`（不明确） |
| `"$var"` 加引号 | `$var` 不加引号 |
| `printf` 格式化 | `echo -e` |
| `local` 声明局部变量 | 函数中用全局变量 |
| `main()` 函数模式 | 顶层写业务逻辑 |
| `readonly` / `declare -r` | 可修改的常量 |

### POSIX 兼容写法

```bash
#!/bin/sh
# 用 /bin/sh 时坚持 POSIX 语法

# 测试用 [ ]
[ -f "$file" ] && echo "存在"

# 函数用 func() { }
my_func() {
    # ...
}

# 不用 [[ ]]，不用数组，不用 <<<

# 条件用 test 或 [ ]
if [ "$a" = "$b" ]; then   # 用 = 不用 ==
    echo "相等"
fi

# 序列用 seq 不用 {1..10}
for i in $(seq 1 10); do
    echo "$i"
done

# 字符串操作用外部命令
upper=$(echo "$str" | tr 'a-z' 'A-Z')   # 不用 ${str^^}
```

### 平台差异处理

```bash
# 检测 OS
detect_os() {
    case "$(uname -s)" in
        Linux*)   OS="linux" ;;
        Darwin*)  OS="macos" ;;
        CYGWIN*|MINGW*) OS="windows" ;;
        *)        OS="unknown" ;;
    esac
    echo "$OS"
}

# 包管理器抽象
install_package() {
    local pkg=$1
    if command -v apt-get > /dev/null; then
        sudo apt-get install -y "$pkg"
    elif command -v yum > /dev/null; then
        sudo yum install -y "$pkg"
    elif command -v brew > /dev/null; then
        brew install "$pkg"
    else
        echo "不支持的包管理器" >&2
        return 1
    fi
}

# sed 的 -i 参数差异
# Linux: sed -i 's/old/new/' file
# macOS: sed -i '' 's/old/new/' file
sed_inplace() {
    if [[ "$(uname)" == "Darwin" ]]; then
        sed -i '' "$@"
    else
        sed -i "$@"
    fi
}

# date 命令差异
# Linux: date -d "2024-01-01" +%s
# macOS: date -j -f "%Y-%m-%d" "2024-01-01" +%s
```

## 3. Shell 脚本的安全编程有哪些要点？

**回答：**

### 输入验证

```bash
# 永远验证用户输入
validate_input() {
    local input=$1

    # 检查空值
    [ -z "$input" ] && { echo "输入不能为空" >&2; return 1; }

    # 白名单验证（推荐）
    [[ "$input" =~ ^[a-zA-Z0-9_.-]+$ ]] || {
        echo "输入包含非法字符" >&2
        return 1
    }

    # 长度限制
    (( ${#input} > 255 )) && { echo "输入过长" >&2; return 1; }
}

# 安全的文件路径处理
safe_path() {
    local path=$1
    # 防止路径遍历
    local resolved
    resolved=$(realpath -m "$path" 2>/dev/null) || return 1

    # 确保在允许的目录下
    local allowed_dir="/opt/app/data"
    [[ "$resolved" == "$allowed_dir"/* ]] || {
        echo "路径不在允许范围内" >&2
        return 1
    }
    echo "$resolved"
}
```

### 命令注入防护

```bash
# 危险！命令注入
user_file="file; rm -rf /"
cat $user_file           # 可能执行恶意命令

# 安全：引号保护
cat "$user_file"         # 作为一个单词处理

# 危险！eval 注入
eval "echo $user_input"

# 安全：避免 eval，或严格验证
echo "$user_input"

# 危险！动态命令构建
cmd="ls $user_dir"
$cmd

# 安全：使用数组
cmd=(ls "$user_dir")
"${cmd[@]}"
```

### 临时文件安全

```bash
# 危险！可预测的临时文件名
tmpfile="/tmp/myapp.$$"        # PID 可预测，存在竞态条件

# 安全：使用 mktemp
tmpfile=$(mktemp)              # /tmp/tmp.XXXXXXXXXX
tmpdir=$(mktemp -d)            # 临时目录

# 确保清理
trap 'rm -f "$tmpfile"' EXIT

# 设置安全权限
tmpfile=$(mktemp)
chmod 600 "$tmpfile"           # 只有拥有者可读写

# 使用 TMPDIR 环境变量
tmpfile=$(mktemp -p "${TMPDIR:-/tmp}" myapp.XXXXXX)
```

### 凭据安全

```bash
# 危险！命令行中的密码
mysql -u root -p"$MYSQL_PWD" ...
# 密码会出现在 ps aux 中！

# 安全：使用环境变量或配置文件
export MYSQL_PWD="$password"
mysql -u root ...

# 安全：使用配置文件
cat > ~/.my.cnf << EOF
[client]
password=$password
EOF
chmod 600 ~/.my.cnf
mysql -u root ...

# 不要在日志中输出敏感信息
log_info "连接数据库 host=$DB_HOST user=$DB_USER"   # 好
log_info "连接数据库 password=$DB_PASS"              # 不好！

# 不要在脚本中硬编码密码
# 使用外部密钥管理
password=$(vault kv get -field=password secret/db)
password=$(aws secretsmanager get-secret-value --secret-id db-password --query SecretString --output text)
```

### 权限最小化

```bash
# 不要用 chmod 777
chmod 777 /path/to/dir    # 不要这样！

# 合理的权限
chmod 755 /opt/app/bin/   # 目录：拥有者读写执行，其他人读执行
chmod 644 /etc/app.conf   # 配置：拥有者读写，其他人只读
chmod 600 /etc/app.key    # 密钥：只有拥有者可读写
chmod 700 ~/.ssh/          # SSH 目录

# 不要用 root 运行服务脚本（除非必需）
if [ "$(id -u)" -eq 0 ]; then
    exec su -s /bin/bash nobody -c "$0 $*"
fi
```

## 4. Shell 脚本中有哪些常见的陷阱（Gotchas）？

**回答：**

### 变量相关

```bash
# 陷阱1：未引用变量
file="my document.txt"
rm $file       # 删除 "my" 和 "document.txt" 两个文件！
rm "$file"     # 正确

# 陷阱2：空变量
var=""
[ $var = "hello" ]      # 语法错误！展开为 [ = "hello" ]
[ "$var" = "hello" ]    # 正确

# 陷阱3：变量中的换行
multi="line1
line2"
echo $multi      # line1 line2（换行变空格）
echo "$multi"    # 保持换行

# 陷阱4：local 掩盖退出码
set -e
local result=$(false)   # 不退出！local 返回 0
local result
result=$(false)          # 退出
```

### 命令相关

```bash
# 陷阱5：cd 可能失败
cd /some/dir
rm -rf *               # 如果 cd 失败，会在当前目录删除！
cd /some/dir || exit 1  # 正确
cd /some/dir && rm -rf *  # 也可以

# 陷阱6：for + 命令替换
for f in $(find . -name "*.txt"); do   # 文件名含空格会出错
    echo "$f"
done
# 正确：
while IFS= read -r -d '' f; do
    echo "$f"
done < <(find . -name "*.txt" -print0)

# 陷阱7：管道中的子 Shell
count=0
echo -e "a\nb\nc" | while read -r line; do
    (( count++ ))
done
echo "$count"   # 0！not 3

# 陷阱8：重定向覆盖源文件
sort file > file     # file 被清空！
sort file -o file    # 正确（sort 的 -o 选项）
```

### 逻辑相关

```bash
# 陷阱9：&& || 不等于 if-else
[[ cond ]] && echo "true" || echo "false"
# 如果 echo "true" 失败（几乎不可能），也会执行 echo "false"

# 陷阱10：数值与字符串比较混淆
[[ "10" > "9" ]]    # false！字典序比较 "1" < "9"
[[ 10 -gt 9 ]]      # true
(( 10 > 9 ))         # true

# 陷阱11：glob 不匹配时的行为
for f in /nonexistent/*.txt; do
    echo "$f"    # 输出字面量 "/nonexistent/*.txt"
done
# 修复：
shopt -s nullglob   # 不匹配时展开为空
for f in /nonexistent/*.txt; do
    echo "$f"    # 循环不执行
done

# 或防御性检查
for f in /path/*.txt; do
    [ -e "$f" ] || continue
    echo "$f"
done
```

### 特殊字符

```bash
# 陷阱12：IFS 影响分词
IFS=:
arr=($PATH)          # 按 : 分割
echo "${arr[0]}"     # 第一个 PATH 目录
unset IFS            # 记得恢复！

# 陷阱13：文件名以 - 开头
touch -- "-f"        # 创建名为 -f 的文件
rm "-f"              # 失败！被当作选项
rm -- "-f"           # 正确
rm ./-f              # 也正确
```

### 手写测试框架

```bash
#!/bin/bash
# simple_test.sh — 最小测试框架

TESTS_RUN=0
TESTS_PASSED=0
TESTS_FAILED=0

assert_equals() {
    local expected=$1
    local actual=$2
    local msg=${3:-""}
    (( TESTS_RUN++ ))

    if [ "$expected" = "$actual" ]; then
        (( TESTS_PASSED++ ))
        echo "  ✓ $msg"
    else
        (( TESTS_FAILED++ ))
        echo "  ✗ $msg"
        echo "    期望: '$expected'"
        echo "    实际: '$actual'"
    fi
}

assert_success() {
    local cmd="$*"
    (( TESTS_RUN++ ))
    if eval "$cmd" > /dev/null 2>&1; then
        (( TESTS_PASSED++ ))
        echo "  ✓ $cmd"
    else
        (( TESTS_FAILED++ ))
        echo "  ✗ $cmd (预期成功但失败)"
    fi
}

assert_failure() {
    local cmd="$*"
    (( TESTS_RUN++ ))
    if ! eval "$cmd" > /dev/null 2>&1; then
        (( TESTS_PASSED++ ))
        echo "  ✓ $cmd (预期失败)"
    else
        (( TESTS_FAILED++ ))
        echo "  ✗ $cmd (预期失败但成功)"
    fi
}

test_summary() {
    echo ""
    echo "================================"
    echo "运行: $TESTS_RUN  通过: $TESTS_PASSED  失败: $TESTS_FAILED"
    echo "================================"
    (( TESTS_FAILED == 0 )) && exit 0 || exit 1
}

# 使用
source ./lib/utils.sh

echo "测试 utils.sh"
assert_equals "hello" "$(trim '  hello  ')" "trim 基本功能"
assert_success "is_number 42"
assert_failure "is_number abc"

test_summary
```

## 6. 如何安全地处理密码和敏感数据？

**回答：**

### 密码输入

```bash
# 从终端安全读取
read -rsp "请输入密码: " password
echo

# 从文件读取（权限 600）
password=$(cat /etc/myapp/db_password)

# 从环境变量（CI/CD 系统常用）
password="${DB_PASSWORD:?需要设置 DB_PASSWORD}"
```

```bash

mysql -p"$password" ...    # 不好！ps 可见
MYSQL_PWD="$password" mysql ...  # 稍好

# 2. 不要 echo 密码
echo "$password"           # 不好！可能被日志记录
echo "密码长度: ${#password}"  # 可以

unset password

# 4. 防止 set -x 泄露
set +x   # 关闭 trace
SENSITIVE_CMD

# 或使用子 Shell 隔离
(
    set +x

# 5. 日志脱敏
log_sanitized() {
    local msg="$*"
    # 替换已知的敏感模式
    msg=$(echo "$msg" | sed -E 's/(password|token|secret|key)=[^ ]*/\1=***REDACTED***/gi')
    echo "$msg"
}
```

## 7. 脚本的错误恢复和幂等性如何保证？

**回答：**

### 幂等性原则

```bash
# 幂等 = 多次执行结果相同

# 创建目录（幂等）
mkdir -p /opt/app/data        # -p 已存在不报错

# 创建用户（幂等）
id deploy > /dev/null 2>&1 || useradd deploy

# 安装包（幂等）
dpkg -l | grep -q '^ii.*nginx' || apt-get install -y nginx

# 修改配置（幂等）
grep -q "^max_connections" /etc/my.cnf || \
    echo "max_connections=500" >> /etc/my.cnf

# 更好的幂等配置修改
ensure_config() {
    local file=$1
    local key=$2
    local value=$3

    if grep -q "^${key}" "$file"; then
        # 已存在则更新
        sed -i "s|^${key}.*|${key}=${value}|" "$file"
    else
        # 不存在则添加
        echo "${key}=${value}" >> "$file"
    fi
}
ensure_config /etc/my.cnf max_connections 500
```

### 检查点与恢复

```bash
STATE_FILE="/var/lib/myapp/deploy_state"

save_state() { echo "$1" > "$STATE_FILE"; }
get_state()  { cat "$STATE_FILE" 2>/dev/null || echo "init"; }

resume_deploy() {

            ;;
    esac
}
```

## 8. Shell 脚本的性能优化有哪些技巧？

**回答：**

### 减少外部命令调用

```bash
# 慢：每次循环调用外部命令
for f in *.txt; do
    basename "$f" .txt      # 外部命令
done

# 快：使用参数展开
for f in *.txt; do
    echo "${f%.txt}"        # 内置操作
done

# 慢
upper=$(echo "$str" | tr 'a-z' 'A-Z')
# 快
upper="${str^^}"

# 慢
length=$(echo "$str" | wc -c)
# 快
length=${#str}
```

### 减少子 Shell

```bash
# 慢：$() 创建子 Shell
result=$(echo "hello")
# 快：直接赋值
result="hello"

# 慢：管道创建子 Shell
echo "$var" | grep -q "pattern"
# 快：内置匹配
[[ "$var" == *pattern* ]]
```

### 批量 I/O

```bash
# 慢：循环中多次重定向
for i in {1..10000}; do
    echo "$i" >> output.txt
done

# 快：一次重定向
for i in {1..10000}; do
    echo "$i"
done > output.txt

# 快：printf 比 echo 更高效
printf '%s\n' {1..10000} > output.txt
```

### 使用更高效的工具

```bash
# grep 比 awk/sed 更快（简单搜索时）
# awk 比 shell 循环更快（处理文本时）

# 慢
while IFS= read -r line; do
    echo "$line" | cut -d, -f1
done < data.csv

# 快
awk -F, '{print $1}' data.csv

# 非常大的文件用 LC_ALL=C 加速
LC_ALL=C grep "pattern" huge_file    # 比默认 locale 快 5-10 倍
LC_ALL=C sort huge_file
```

### 并行处理

```bash
# 利用多核
find . -name "*.log" -print0 | xargs -0 -P "$(nproc)" gzip

# GNU Parallel
parallel -j "$(nproc)" process_file {} ::: *.dat
```

## 9. 如何编写健壮的启动/停止服务脚本？

**回答：**

```bash
#!/bin/bash
set -euo pipefail

APP_NAME="myapp"
APP_DIR="/opt/$APP_NAME"
PID_FILE="/var/run/${APP_NAME}.pid"
LOG_FILE="/var/log/${APP_NAME}.log"
APP_USER="appuser"
STOP_TIMEOUT=30

get_pid() {
    if [ -f "$PID_FILE" ]; then
        local pid
        pid=$(cat "$PID_FILE")
        if kill -0 "$pid" 2>/dev/null; then
            echo "$pid"
            return 0
        fi
        rm -f "$PID_FILE"
    fi
    return 1
}

is_running() {
    get_pid > /dev/null 2>&1
}

do_start() {
    if is_running; then
        echo "$APP_NAME 已在运行 (PID: $(get_pid))"
        return 0
    fi

    echo -n "启动 $APP_NAME..."

    cd "$APP_DIR"
    su -s /bin/bash "$APP_USER" -c \
        "nohup $APP_DIR/bin/server >> $LOG_FILE 2>&1 & echo \$!" > "$PID_FILE"

    # 等待启动
    local retries=10
    while (( retries > 0 )); do
        if is_running; then
            echo " 成功 (PID: $(get_pid))"
            return 0
        fi
        sleep 1
        (( retries-- ))
    done

    echo " 失败"
    rm -f "$PID_FILE"
    return 1
}

do_stop() {
    if ! is_running; then
        echo "$APP_NAME 未运行"
        return 0
    fi

    local pid
    pid=$(get_pid)
    echo -n "停止 $APP_NAME (PID: $pid)..."

    # 优雅停止
    kill "$pid"

    local elapsed=0
    while kill -0 "$pid" 2>/dev/null; do
        if (( elapsed >= STOP_TIMEOUT )); then
            echo " 超时，强制终止"
            kill -9 "$pid" 2>/dev/null
            sleep 1
            break
        fi
        sleep 1
        (( elapsed++ ))
        echo -n "."
    done

    rm -f "$PID_FILE"
    echo " 已停止"
}

do_restart() {
    do_stop
    sleep 2
    do_start
}

do_status() {
    if is_running; then
        echo "$APP_NAME 运行中 (PID: $(get_pid))"
        return 0
    else
        echo "$APP_NAME 未运行"
        return 3    # LSB 标准：程序未运行
    fi
}

case "${1:-}" in
    start)   do_start ;;
    stop)    do_stop ;;
    restart) do_restart ;;
    status)  do_status ;;
    *)
        echo "用法: $0 {start|stop|restart|status}"
        exit 2
        ;;
esac
```

## 10. 何时应该用 Shell，何时应该用 Python 或其他语言？

**回答：**

### 适合 Shell 的场景

| 场景 | 说明 |
|------|------|
| 系统管理 | 用户创建、权限管理、服务管理 |
| 文件操作 | 批量重命名、移动、清理 |
| 命令编排 | 串联多个 CLI 工具 |
| 管道处理 | grep/sed/awk 文本流处理 |
| 快速原型 | <100 行的自动化脚本 |
| 环境配置 | .bashrc、系统初始化 |
| CI/CD 步骤 | 构建、测试、部署的粘合脚本 |
| Cron 任务 | 简单的定时清理、备份 |

### 不适合 Shell 的场景

| 场景 | 更好的选择 |
|------|-----------|
| 复杂逻辑（>200行） | Python, Go |
| JSON/YAML 处理 | Python, jq+Shell |
| HTTP API 交互 | Python (requests) |
| 数据库操作 | Python, 专用工具 |
| 并发/异步 | Go, Python |
| 错误处理复杂 | Python |
| 需要数据结构 | Python |
| 跨平台可移植 | Python, Go |
| 单元测试重 | Python, Go |
| Web 应用 | 任何都不是 Shell |

### 经验法则

```
如果脚本需要：
  - 超过 200 行 → 考虑 Python
  - 复杂的字符串解析 → Python
  - 类、对象、复杂数据结构 → Python
  - 详细的错误处理 → Python
  - 跨平台运行 → Python/Go
  - 只是调用几个命令 → Shell
  - 文本管道处理 → Shell
  - 系统管理操作 → Shell
```

### 混合使用

```bash
#!/bin/bash
# Shell 负责编排，Python 负责复杂逻辑

# 用 Shell 做文件检查和命令调用
[ -f config.yaml ] || { echo "配置不存在"; exit 1; }

# 用 Python 处理复杂数据
result=$(python3 << 'PYEOF'
import yaml, json, sys

with open("config.yaml") as f:
    config = yaml.safe_load(f)

# 复杂的数据转换逻辑
output = {
    "hosts": [h["address"] for h in config["servers"]],
    "total": len(config["servers"])
}
print(json.dumps(output))
PYEOF
)

# 继续用 Shell 处理
hosts=$(echo "$result" | jq -r '.hosts[]')
for host in $hosts; do
    ssh "$host" "systemctl restart nginx"
done
```
