## 1. 子 Shell 和命令分组（`()`  vs `{}`）有什么区别？

**回答：**

### 子 Shell `( )`

```bash
# 在子进程中执行，不影响父 Shell
(
    cd /tmp
    var="modified"
    export NEW_VAR="hello"
)
# 当前目录没变，var 没变，NEW_VAR 不存在

# 用途：
# 1. 隔离副作用
( cd /tmp && tar czf backup.tar.gz data/ )
# 不改变当前目录

# 2. 并行执行
( task1 ) &
( task2 ) &
wait

# 3. 临时修改环境
( export LC_ALL=C; sort file )
```

### 命令分组 `{ }`

```bash
# 在当前 Shell 中执行，共享环境
{
    cd /tmp
    var="modified"
}
# 当前目录变了，var 也变了

# 注意语法：
# { 后必须有空格
# } 前必须有 ; 或换行
{ echo "hello"; echo "world"; }   # 正确
{echo "hello"}                      # 语法错误！

# 用途：
# 1. 重定向整组命令
{
    echo "header"
    cat data.txt
    echo "footer"
} > output.txt

# 2. 错误处理
cmd || { echo "失败" >&2; exit 1; }
```

### 对比

| 特性 | `( )` 子 Shell | `{ }` 命令组 |
|------|---------------|-------------|
| 执行环境 | 新子进程 | 当前 Shell |
| 变量改变 | 不影响父 Shell | 影响当前 Shell |
| cd 影响 | 不影响 | 影响 |
| 性能 | 有 fork 开销 | 无额外开销 |
| 语法 | 无特殊要求 | `{` 后要空格，`}` 前要 `;` |

## 2. 协程（Coproc）是什么？怎么使用？

**回答：**

`coproc` 在后台启动命令并建立双向管道通信。

```bash
# 基本语法
coproc command
# 创建 COPROC[0]（读取端）和 COPROC[1]（写入端）

# 命名协程
coproc MYPROC { command; }

# 示例：与 bc 计算器交互
coproc BC { bc -l; }

echo "3.14 * 2" >&"${BC[1]}"    # 向 bc 发送
read -r result <&"${BC[0]}"     # 读取结果
echo "结果: $result"             # 6.28

echo "sqrt(2)" >&"${BC[1]}"
read -r result <&"${BC[0]}"
echo "sqrt(2) = $result"

# 关闭写入端（让 bc 退出）
exec {BC[1]}>&-
wait "$BC_PID"
```

### 实际应用

```bash
# 与 Python 交互
coproc PY { python3 -u -c "
import sys
for line in sys.stdin:
    line = line.strip()
    print(eval(line), flush=True)
"; }

echo "2**10" >&"${PY[1]}"
read -r result <&"${PY[0]}"
echo "2^10 = $result"   # 1024

exec {PY[1]}>&-
wait "$PY_PID"
```

### 注意事项

- 协程是 Bash 4.0+ 的特性
- 同一时间只能有一个未命名协程
- 命名协程可以有多个
- 需要注意管道缓冲问题（使用 `-u` 或 `stdbuf`）

## 3. 信号处理和进程管理有哪些高级技巧？

**回答：**

### 信号转发

```bash
#!/bin/bash
# Wrapper 脚本需要转发信号给子进程

child_pid=""

forward_signal() {
    if [ -n "$child_pid" ]; then
        kill -"$1" "$child_pid" 2>/dev/null
    fi
}

trap 'forward_signal TERM' TERM
trap 'forward_signal INT' INT
trap 'forward_signal HUP' HUP

# 启动子进程
/usr/bin/actual-program "$@" &
child_pid=$!

# 等待子进程（wait 可以被信号中断）
wait "$child_pid"
exit $?
```

### 进程组管理

```bash
# 创建新的进程组
set -m   # 启用 job control

# 启动进程组
(
    trap '' INT    # 子进程忽略 SIGINT
    cmd1 &
    cmd2 &
    cmd3 &
    wait
) &
GROUP_PID=$!

# 终止整个进程组
kill -- -"$GROUP_PID"   # 注意负号，表示进程组
```

### 后台进程管理

```bash
# 管理多个后台进程
declare -A WORKERS

start_worker() {
    local name=$1
    shift
    "$@" &
    WORKERS[$name]=$!
    echo "启动 $name (PID: ${WORKERS[$name]})"
}

stop_worker() {
    local name=$1
    local pid=${WORKERS[$name]:-}
    if [ -n "$pid" ] && kill -0 "$pid" 2>/dev/null; then
        kill "$pid"
        wait "$pid" 2>/dev/null || true
        echo "停止 $name (PID: $pid)"
    fi
    unset 'WORKERS[$name]'
}

stop_all() {
    for name in "${!WORKERS[@]}"; do
        stop_worker "$name"
    done
}

trap stop_all EXIT

start_worker "web" python3 -m http.server 8080
start_worker "worker" ./process_queue.sh

wait   # 等待所有后台进程
```

## 4. 文件锁和并发控制怎么实现？

**回答：**

### flock 文件锁

```bash
# 方式1：flock 命令包裹
flock /tmp/myapp.lock ./critical_section.sh

# 方式2：flock 在脚本内
(
    flock -n 9 || { echo "锁被占用"; exit 1; }
    # 临界区代码
    echo "正在执行..."
    sleep 10
) 9>/tmp/myapp.lock

# 方式3：共享锁和排他锁
flock -s /tmp/data.lock cat data.txt      # 共享锁（读）
flock -x /tmp/data.lock ./update_data.sh  # 排他锁（写）

# 方式4：带超时
flock -w 30 /tmp/myapp.lock ./script.sh   # 等待最多 30 秒

# 方式5：脚本自锁（常用模式）
LOCK_FD=200
eval "exec $LOCK_FD>/tmp/${0##*/}.lock"
flock -n $LOCK_FD || { echo "已有实例运行"; exit 1; }
```

### 信号量（Semaphore）

```bash
# 用 FIFO 实现简单的并发控制
FIFO="/tmp/semaphore.$$"
MAX_CONCURRENT=4

mkfifo "$FIFO"
exec 3<>"$FIFO"
rm "$FIFO"

# 初始化令牌
for (( i = 0; i < MAX_CONCURRENT; i++ )); do
    echo >&3
done

# 并发执行任务
for task in task{1..20}; do
    read -u 3   # 获取令牌（阻塞直到有空位）
    {
        echo "执行 $task"
        sleep $((RANDOM % 5))
        echo "完成 $task"
        echo >&3   # 归还令牌
    } &
done

wait
exec 3>&-
```

### GNU Parallel

```bash
# 用 parallel 更简单地控制并发
cat tasks.txt | parallel -j 4 process_task {}

# 带进度条
parallel --bar -j 4 process {} ::: item{1..100}

# 远程并行
parallel -j 2 --sshlogin server1,server2 "uptime" ::: ""
```

## 5. eval 和间接引用的高级用法？有什么安全风险？

**回答：**

### eval 用法

```bash
# eval 将参数组合后作为命令执行
cmd="echo hello"
eval "$cmd"    # 执行 echo hello

# 动态变量名
for i in 1 2 3; do
    eval "var_$i=value_$i"
done
echo "$var_1"   # value_1

# 动态构建命令
sort_field=3
sort_order="-r"
eval "sort -t, -k${sort_field},${sort_field} ${sort_order} data.csv"
```

### 安全风险

```bash
# eval 是最危险的命令之一！
user_input="hello; rm -rf /"
eval "echo $user_input"   # 会执行 rm -rf / ！

# 永远不要 eval 用户输入！
```

### 更安全的替代方案

```bash
# 替代1：间接引用 ${!var}
name="PATH"
echo "${!name}"    # 输出 $PATH 的值

# 替代2：nameref（Bash 4.3+）
declare -n ref=PATH
echo "$ref"        # 输出 $PATH 的值

# 替代3：数组存储命令
cmd=(sort -t, -k3,3 -r data.csv)
"${cmd[@]}"        # 安全执行

# 替代4：关联数组代替动态变量名
declare -A config
for key in host port user; do
    config[$key]="value_$key"
done
echo "${config[host]}"
```

### 合理使用 eval 的场景

```bash
# 1. 解析 getopt 的输出（必须用 eval）
OPTS=$(getopt -o vh -l verbose,help -- "$@")
eval set -- "$OPTS"

# 2. SSH agent 设置
eval "$(ssh-agent -s)"

# 3. 动态 alias/function（已知可控输入）
```

## 6. 数组的高级操作有哪些？

**回答：**

### 数组切片和操作

```bash
arr=(a b c d e f g)

# 切片
echo "${arr[@]:2:3}"    # c d e（从索引2开始取3个）
echo "${arr[@]:3}"      # d e f g（从索引3到末尾）

# 长度
echo "${#arr[@]}"       # 7（元素个数）
echo "${#arr[0]}"       # 1（第一个元素的字符长度）

# 添加元素
arr+=(h i)              # 追加
arr=("${arr[@]}" j)     # 同上

# 删除元素
unset 'arr[2]'          # 删除索引2（不重排索引！）
arr=("${arr[@]}")       # 重排索引

# 查找元素
contains() {
    local target=$1
    shift
    for item in "$@"; do
        [[ "$item" == "$target" ]] && return 0
    done
    return 1
}
contains "c" "${arr[@]}" && echo "找到"

# 数组去重
readarray -t unique < <(printf '%s\n' "${arr[@]}" | sort -u)

# 数组反转
reverse=()
for (( i = ${#arr[@]} - 1; i >= 0; i-- )); do
    reverse+=("${arr[i]}")
done

# 数组连接为字符串
IFS=,
joined="${arr[*]}"    # a,b,c,d,...
unset IFS
```

### 关联数组高级用法

```bash
declare -A map

# 批量赋值
map=(
    [name]="myapp"
    [version]="1.0"
    [env]="production"
)

# 检查 key 是否存在
[[ -v map[name] ]] && echo "存在"
[[ -n "${map[name]+set}" ]] && echo "存在"  # 兼容空值

# 遍历
for key in "${!map[@]}"; do
    echo "$key = ${map[$key]}"
done

# 合并两个关联数组
declare -A base=(
    [host]="localhost"
    [port]="3306"
    [db]="mydb"
)
declare -A override=(
    [host]="prod-db.example.com"
    [port]="3307"
)

declare -A merged
for key in "${!base[@]}"; do
    merged[$key]="${base[$key]}"
done
for key in "${!override[@]}"; do
    merged[$key]="${override[$key]}"
done
```

### mapfile / readarray

```bash
# 将文件读入数组
mapfile -t lines < file.txt
echo "行数: ${#lines[@]}"
echo "第3行: ${lines[2]}"

# 从命令输出读取
mapfile -t files < <(find . -name "*.sh")

# 跳过前 N 行
mapfile -t -s 1 data < file.csv      # 跳过表头

# 回调处理
mapfile -t -C 'echo "读取: "' -c 1 data < file.txt
```

## 7. 字符串操作有哪些高级技巧？

**回答：**

### 参数展开进阶

```bash
str="hello.world.tar.gz"

# 模式删除
echo "${str#*.}"     # world.tar.gz（删除最短前缀匹配）
echo "${str##*.}"    # gz（删除最长前缀匹配）
echo "${str%.*}"     # hello.world.tar（删除最短后缀匹配）
echo "${str%%.*}"    # hello（删除最长后缀匹配）

# 记忆口诀：
# # 从左删（在键盘上 # 在 $ 左边）
# % 从右删（在键盘上 % 在 $ 右边）
# 单个 = 最短匹配，双个 = 最长匹配

# 文件名操作
filepath="/home/user/src/main.py"
echo "${filepath##*/}"     # main.py（等价 basename）
echo "${filepath%/*}"      # /home/user/src（等价 dirname）

# 替换
echo "${str/world/WORLD}"  # hello.WORLD.tar.gz（替换第一个）
echo "${str//./!}"         # hello!world!tar!gz（替换所有）
echo "${str/#hello/HI}"    # HI.world.tar.gz（开头替换）
echo "${str/%gz/bz2}"      # hello.world.tar.bz2（结尾替换）
```

### 大小写转换（Bash 4+）

```bash
str="Hello World"
echo "${str^^}"    # HELLO WORLD（全大写）
echo "${str,,}"    # hello world（全小写）
echo "${str^}"     # Hello World（首字母大写）
echo "${str~}"     # hELLO wORLD（反转首字母大小写）
echo "${str~~}"    # hELLO wORLD（反转所有大小写）
```

### 默认值和赋值

```bash
# ${var:-default}  — 如果 var 未设置或为空，返回 default
echo "${name:-anonymous}"

# ${var:=default}  — 如果 var 未设置或为空，赋值为 default 并返回
echo "${dir:=/tmp}"

# ${var:+alternate} — 如果 var 已设置且非空，返回 alternate
echo "${debug:+--verbose}"   # debug 非空则返回 --verbose

# ${var:?message} — 如果 var 未设置或为空，输出 message 并退出
: "${DB_HOST:?需要设置 DB_HOST}"

# 不含冒号版本（只检查是否设置，不检查是否为空）
# ${var-default}   ${var=default}   ${var+alternate}   ${var?message}
```

### 子串操作

```bash
str="Hello, World!"
echo "${str:7}"       # World!（从位置7到末尾）
echo "${str:7:5}"     # World（从位置7取5个字符）
echo "${str: -6}"     # orld!（注意空格！从右边取6个）
echo "${str: -6:4}"   # orld
echo "${#str}"        # 13（字符串长度）
```

## 8. Bash 的异步执行和并行处理有哪些方式？

**回答：**

### 后台执行 `&` + wait

```bash
# 基本并行
task1 &
task2 &
task3 &
wait   # 等待所有后台任务完成

# 获取各任务的退出码
task1 & pid1=$!
task2 & pid2=$!
task3 & pid3=$!

wait "$pid1"; rc1=$?
wait "$pid2"; rc2=$?
wait "$pid3"; rc3=$?
echo "结果: $rc1 $rc2 $rc3"
```

### wait -n（Bash 4.3+，等待任一完成）

```bash
# 工作池模式
MAX_JOBS=4
declare -A jobs

for task in task{1..20}; do
    while (( ${#jobs[@]} >= MAX_JOBS )); do
        wait -n -p finished_pid  # 等待任一完成（Bash 5.1+）
        unset 'jobs[$finished_pid]'
    done

    process_task "$task" &
    jobs[$!]=$task
done

wait
```

### xargs -P 并行

```bash
# 简单并行
seq 100 | xargs -P 4 -I {} bash -c 'echo "处理 {}"; sleep 1'

# 文件处理并行
find . -name "*.log" -print0 | xargs -0 -P 8 gzip
```

### GNU Parallel

```bash
# 替代 xargs 的更强大工具
parallel -j 4 process_item {} ::: item{1..100}

# 结合文件
parallel -j 4 -a input_list.txt process_item {}

# 远程执行
parallel -j 2 -S server1,server2 process {} ::: item{1..10}

# 带进度和日志
parallel --bar --joblog job.log -j 8 compress {} ::: *.raw
```

### 管道并行（GNU Parallel）

```bash
# 管道中并行处理

# 分块处理
cat big.csv | parallel --header : --pipe -N 10000 'python3 process.py'
```

## 9. Bash 中的网络操作（/dev/tcp 和 /dev/udp）？

**回答：**

Bash 内置了通过特殊路径进行 TCP/UDP 通信的能力（非真实设备文件）。

### TCP 连接

```bash
# 端口检测
check_port() {
    local host=$1 port=$2
    if timeout 3 bash -c "echo > /dev/tcp/$host/$port" 2>/dev/null; then
        echo "$host:$port 开放"
    else
        echo "$host:$port 关闭"
    fi
}
check_port localhost 80
check_port google.com 443

# 简单的 HTTP 请求
exec 3<>/dev/tcp/example.com/80
cat >&3 <<EOF
GET / HTTP/1.1
Host: example.com
Connection: close

EOF
cat <&3
exec 3>&-

# 简单端口扫描
for port in 22 80 443 3306 5432 6379 8080; do
    (echo > "/dev/tcp/localhost/$port") 2>/dev/null && echo "端口 $port 开放" &
done
wait
```

### 等待服务就绪

```bash
wait_for_port() {
    local host=$1
    local port=$2
    local timeout=${3:-30}
    local elapsed=0

    echo "等待 $host:$port..."
    while ! (echo > "/dev/tcp/$host/$port") 2>/dev/null; do
        if (( elapsed >= timeout )); then
            echo "超时！" >&2
            return 1
        fi
        sleep 1
        (( elapsed++ ))
    done
    echo "$host:$port 就绪 (${elapsed}s)"
}

wait_for_port localhost 5432 60
```

### 简易数据传输

```bash
# 发送方
echo "Hello from client" > /dev/tcp/server/9999

# 接收方（用 nc/netcat）
nc -l 9999
```

## 10. 如何在 Shell 中实现简单的数据结构？

**回答：**

### 栈（Stack）

```bash
declare -a STACK=()

stack_push() { STACK+=("$1"); }

stack_pop() {
    local last=$(( ${#STACK[@]} - 1 ))
    (( last >= 0 )) || return 1
    STACK_TOP="${STACK[$last]}"
    unset 'STACK[$last]'
}

stack_peek() {
    local last=$(( ${#STACK[@]} - 1 ))
    (( last >= 0 )) && echo "${STACK[$last]}"
}

stack_size() { echo "${#STACK[@]}"; }

# 使用
stack_push "first"
stack_push "second"
stack_push "third"
stack_pop; echo "$STACK_TOP"  # third
stack_peek                     # second
```

### 队列（Queue）

```bash
declare -a QUEUE=()
QUEUE_HEAD=0

queue_enqueue() { QUEUE+=("$1"); }

queue_dequeue() {
    (( QUEUE_HEAD < ${#QUEUE[@]} )) || return 1
    QUEUE_FRONT="${QUEUE[$QUEUE_HEAD]}"
    unset 'QUEUE[$QUEUE_HEAD]'
    (( QUEUE_HEAD++ ))
}

queue_size() { echo $(( ${#QUEUE[@]} - QUEUE_HEAD )); }

# 使用
queue_enqueue "task1"
queue_enqueue "task2"
queue_dequeue; echo "$QUEUE_FRONT"  # task1
```

### 集合（Set）— 去重

```bash
declare -A SET

set_add()      { SET[$1]=1; }
set_remove()   { unset 'SET[$1]'; }
set_contains() { [[ -v SET[$1] ]]; }
set_members()  { echo "${!SET[@]}"; }
set_size()     { echo "${#SET[@]}"; }

# 使用
set_add "apple"
set_add "banana"
set_add "apple"    # 已存在，不重复
set_contains "apple" && echo "有苹果"
echo "成员: $(set_members)"   # apple banana
```
