## 1. block / rescue / always (异常处理)？

**回答：**

类似编程语言中的 try/catch/finally。

```yaml
tasks:
  - block:
      # try — 正常任务
      - name: Download artifact
        get_url:
          url: "https://releases.example.com/app-{{ version }}.tar.gz"
          dest: /tmp/app.tar.gz

      - name: Extract and deploy
        unarchive:
          src: /tmp/app.tar.gz
          dest: /opt/app/releases/{{ version }}
          remote_src: yes

      - name: Switch symlink
        file:
          src: /opt/app/releases/{{ version }}
          dest: /opt/app/current
          state: link

      - name: Restart service
        service:
          name: myapp
          state: restarted

    rescue:
      # catch — 错误处理
      - name: Log failure
        debug:
          msg: "Deployment failed on {{ inventory_hostname }}"

      - name: Rollback symlink
        file:
          src: /opt/app/releases/{{ previous_version }}
          dest: /opt/app/current
          state: link

      - name: Restart old version
        service:
          name: myapp
          state: restarted

      - name: Send alert
        uri:
          url: "https://hooks.slack.com/services/xxx"
          method: POST
          body_format: json
          body:
            text: "Deploy {{ version }} failed on {{ inventory_hostname }}, rolled back to {{ previous_version }}"

    always:
      # finally — 始终执行
      - name: Clean up temp files
        file:
          path: /tmp/app.tar.gz
          state: absent

      - name: Verify service status
        uri:
          url: "http://localhost:8080/health"
          status_code: 200
        register: health
        ignore_errors: true

      - name: Report status
        debug:
          msg: "Service health: {{ 'OK' if health.status == 200 else 'FAILED' }}"
```

```
block 变量与条件:
  - block:
      - task1
      - task2
    when: env == "production"    # block 级别条件 (所有任务都应用)
    become: true                  # block 级别提权
    vars:                         # block 级别变量
      deploy_dir: /opt/app
```

## 2. delegate_to (委托执行)？

**回答：**

```yaml
# delegate_to — 在其他主机上执行任务

# 场景 1: 从负载均衡器移除主机
- name: Remove from load balancer
  uri:
    url: "http://lb.example.com/api/servers/{{ inventory_hostname }}"
    method: DELETE
  delegate_to: localhost          # 在控制节点执行

# 场景 2: 在数据库服务器上创建用户
- name: Create DB user for app
  mysql_user:
    name: "app_{{ inventory_hostname }}"
    password: "{{ db_password }}"
    priv: "app_db.*:ALL"
  delegate_to: "{{ groups['dbservers'][0] }}"

# 场景 3: 等待端口 (从控制节点检测)
- name: Wait for port
  wait_for:
    host: "{{ ansible_host }}"
    port: 8080
    timeout: 60
  delegate_to: localhost

# local_action — delegate_to: localhost 的简写
- name: Save facts locally
  local_action:
    module: copy
    content: "{{ ansible_facts | to_nice_json }}"
    dest: "/tmp/facts_{{ inventory_hostname }}.json"

# delegate_facts — 将 Facts 记录到被委托主机
- name: Gather facts from LB
  setup:
  delegate_to: lb.example.com
  delegate_facts: true            # Facts 记录到 lb, 非当前主机

# run_once + delegate_to — 全局一次性任务
- name: Create shared database
  mysql_db:
    name: shared_db
    state: present
  delegate_to: "{{ groups['dbservers'][0] }}"
  run_once: true                  # 只执行一次, 不管有多少目标主机
```

## 3. serial (滚动更新)？

**回答：**

```yaml
# serial — 控制每批执行的主机数量

# 固定数量
- hosts: webservers          # 假设 10 台
  serial: 2                  # 每次 2 台
  tasks:
    - name: Deploy
      command: deploy.sh

# 百分比
- hosts: webservers
  serial: "30%"              # 每次 30%

# 递增批次
- hosts: webservers
  serial:
    - 1                      # 第一批: 1 台 (金丝雀)
    - 5                      # 第二批: 5 台
    - "100%"                 # 第三批: 剩余全部

# 配合 max_fail_percentage
- hosts: webservers
  serial: 3
  max_fail_percentage: 30    # 每批超过 30% 失败则停止

# 完整滚动更新示例
- hosts: webservers
  serial: 2
  pre_tasks:
    - name: Disable in LB
      uri:
        url: "http://lb/api/disable/{{ inventory_hostname }}"
        method: POST
      delegate_to: localhost

  tasks:
    - name: Stop service
      service:
        name: myapp
        state: stopped

    - name: Deploy new version
      copy:
        src: "app-{{ version }}.tar.gz"
        dest: /opt/app/

    - name: Start service
      service:
        name: myapp
        state: started

    - name: Health check
      uri:
        url: "http://{{ ansible_host }}:8080/health"
        status_code: 200
      retries: 5
      delay: 10

  post_tasks:
    - name: Enable in LB
      uri:
        url: "http://lb/api/enable/{{ inventory_hostname }}"
        method: POST
      delegate_to: localhost
```

## 4. async / poll (异步任务)？

**回答：**

```yaml
# async — 异步执行长时间任务

# 异步执行, 定期轮询
- name: Long running task
  command: /opt/scripts/long_running.sh
  async: 3600                    # 最大运行时间 (秒)
  poll: 30                       # 每 30 秒检查一次

# Fire and forget (不等待完成)
- name: Start background job
  command: /opt/scripts/batch_job.sh
  async: 3600
  poll: 0                        # 不等待
  register: batch_job

# 后续检查状态
- name: Check batch job status
  async_status:
    jid: "{{ batch_job.ansible_job_id }}"
  register: job_result
  until: job_result.finished
  retries: 60
  delay: 30

# 并行执行多个异步任务
- name: Start multiple downloads
  get_url:
    url: "{{ item }}"
    dest: "/tmp/{{ item | basename }}"
  loop:
    - https://example.com/file1.tar.gz
    - https://example.com/file2.tar.gz
    - https://example.com/file3.tar.gz
  async: 300
  poll: 0
  register: download_jobs

- name: Wait for all downloads
  async_status:
    jid: "{{ item.ansible_job_id }}"
  loop: "{{ download_jobs.results }}"
  register: download_results
  until: download_results.finished
  retries: 30
  delay: 10
```

## 5. 策略 (Strategy)？

**回答：**

```yaml
# strategy — 控制任务执行方式

# linear (默认) — 等所有主机完成当前任务再执行下一个
- hosts: all
  strategy: linear
  tasks:
    - name: Task 1     # 所有主机完成 Task1 后, 才开始 Task2
      ...
    - name: Task 2
      ...

# free — 每台主机独立执行, 不等待其他主机
- hosts: all
  strategy: free       # 快的主机先完成, 慢的继续
  tasks:
    - name: Task 1
      ...
    - name: Task 2
      ...

# host_pinned — 一台主机完成所有任务后再处理下一台
- hosts: all
  strategy: host_pinned   # 类似 serial: 1 但更严格

# debug — 交互式调试
- hosts: webservers
  strategy: debug     # 遇到错误时进入调试器
  tasks:
    - name: Might fail
      command: /opt/suspect.sh
```

```
选择指南:
  linear (默认)  → 大部分场景, 有序执行
  free           → 独立任务, 追求速度
  host_pinned    → 需要主机级原子性
  debug          → 开发/调试阶段
```

## 6. 条件与循环高级用法？

**回答：**

```yaml
# 复杂 when 条件
- name: Complex condition
  debug:
    msg: "Match"
  when:
    - ansible_distribution == "Ubuntu"
    - ansible_distribution_major_version | int >= 20
    - "'webservers' in group_names"
    - my_var is defined and my_var | length > 0
    - not (skip_deploy | default(false) | bool)

# 循环 + 条件
- name: Install required packages only
  apt:
    name: "{{ item.name }}"
    state: present
  loop:
    - { name: nginx, install: true }
    - { name: apache2, install: false }
    - { name: python3, install: true }
  when: item.install | bool

# 嵌套循环 (subelements)
- name: Create user SSH keys
  authorized_key:
    user: "{{ item.0.name }}"
    key: "{{ item.1 }}"
  loop: "{{ users | subelements('ssh_keys') }}"
  # users = [{name: alice, ssh_keys: [key1, key2]}, ...]

# 字典循环  
- name: Create files from dict
  copy:
    content: "{{ item.value.content }}"
    dest: "{{ item.value.path }}"
  loop: "{{ config_files | dict2items }}"

# with_sequence (生成序列)
- name: Create directories
  file:
    path: "/opt/app/shard_{{ item }}"
    state: directory
  loop: "{{ range(1, 11) | list }}"    # 1-10

# register + loop
- name: Check services
  command: "systemctl is-active {{ item }}"
  loop:
    - nginx
    - postgresql
    - redis
  register: service_status
  ignore_errors: true

- name: Report failed services
  debug:
    msg: "{{ item.item }} is down!"
  loop: "{{ service_status.results }}"
  when: item.rc != 0
```

## 7. Callback 插件？

**回答：**

```ini
# ansible.cfg
[defaults]
# 主输出回调 (只能选一个)
stdout_callback = yaml              # yaml / json / debug / minimal / dense

# 额外回调 (可多个)
callback_whitelist = timer, profile_tasks, profile_roles, mail

# 常用回调插件:
#   yaml          → YAML 格式输出 (更可读)
#   json          → JSON 输出 (适合程序处理)
#   timer         → 显示总执行时间
#   profile_tasks → 显示每个任务耗时
#   profile_roles → 显示每个 Role 耗时
#   dense         → 紧凑输出
#   mail          → 执行完后发邮件
#   slack         → 发送到 Slack
#   logstash      → 发送到 Logstash
#   junit         → 生成 JUnit XML (CI/CD)
```

```yaml
# 自定义回调示例 — JUnit (CI/CD)
# ansible.cfg
[defaults]
callback_whitelist = junit
[callback_junit]
output_dir = ./test-results       # JUnit XML 输出目录

# CI/CD 中使用
# GitLab CI
test:
  script:
    - ansible-playbook site.yml
  artifacts:
    reports:
      junit: test-results/*.xml
```

## 8. 插件类型总览？

**回答：**

```
Ansible 插件类型:
  Connection     → 连接目标主机 (ssh, winrm, docker, local)
  Inventory      → 生成主机清单 (aws_ec2, azure_rm, vmware)
  Callback       → 控制输出和事件处理
  Lookup         → 从外部获取数据 (file, env, pipe, hashi_vault)
  Filter         → Jinja2 过滤器 (to_json, regex_replace, ipaddr)
  Test           → Jinja2 测试 (is defined, is file, is match)
  Action         → 在控制节点执行模块逻辑
  Cache          → 缓存Facts (jsonfile, redis, memcached)
  Vars           → 自动加载变量
  Strategy       → 控制执行策略 (linear, free, debug)
  Become         → 提权方式 (sudo, su, doas, runas)
  Shell          → 远程 Shell 类型 (sh, csh, powershell)
  Httpapi        → REST API 连接 (网络设备)
```

```yaml
# 查看已安装的插件
ansible-doc -t connection -l     # 列出连接插件
ansible-doc -t callback -l      # 列出回调插件
ansible-doc -t lookup -l        # 列出 lookup 插件
ansible-doc -t filter -l        # 列出过滤器插件
ansible-doc -t inventory -l     # 列出 inventory 插件

# 查看插件文档
ansible-doc -t lookup file
ansible-doc -t filter to_json
```

## 9. 动态 include 与高级复用？

**回答：**

```yaml
# 根据条件选择不同任务文件
- name: Include OS-specific tasks
  include_tasks: "{{ ansible_os_family | lower }}.yml"

# 根据变量选择 Role
- name: Apply database role
  include_role:
    name: "{{ db_engine }}"      # postgresql / mysql / mongodb
  vars:
    db_port: "{{ db_config.port }}"

# 循环 include
- name: Configure each app
  include_tasks: deploy_app.yml
  loop: "{{ applications }}"
  loop_control:
    loop_var: app
  vars:
    app_name: "{{ app.name }}"
    app_version: "{{ app.version }}"

# apply — 给 include 内的任务添加属性
- name: Include with tags
  include_tasks:
    file: security.yml
    apply:
      tags: [security]
      become: true

# 复用 Task 列表 — 参数化 include
# tasks/create_vhost.yml
- name: Deploy vhost config
  template:
    src: vhost.conf.j2
    dest: "/etc/nginx/conf.d/{{ vhost_name }}.conf"
  notify: reload nginx

# 调用
- include_tasks: tasks/create_vhost.yml
  vars:
    vhost_name: site1
    vhost_domain: site1.example.com

- include_tasks: tasks/create_vhost.yml
  vars:
    vhost_name: site2
    vhost_domain: site2.example.com
```

## 10. 高级特性面试高频题？

**回答：**

```
Q: block/rescue/always 和 ignore_errors 的区别？
A: block/rescue 可以执行回滚逻辑, ignore_errors 只是跳过失败
   block/rescue 类似 try/catch, 可以精确处理不同错误
   ignore_errors 无法区分错误类型

Q: serial 和 forks 的区别？
A: serial → 每批处理多少台主机 (顺序批次)
   forks  → 每批内并行连接多少台 (并行度)
   serial: 10 + forks: 5 → 每批 10 台, 但同时只连接 5 台

Q: import_tasks 和 include_tasks 在 handler 中的区别？
A: import_tasks 的 handler 可以直接 notify
   include_tasks 的 handler 必须 notify 完整的 include 任务名

Q: run_once 在 serial 中的行为？
A: run_once 在每个 serial 批次都执行一次！
   如果真的只要全局一次, 用 run_once + delegate_to: localhost

Q: 如何实现蓝绿部署？
A: 用 serial + delegate_to + 负载均衡器 API:
   1. serial: "50%"
   2. pre_tasks: 从 LB 移除
   3. tasks: 部署新版本
   4. post_tasks: 加回 LB + 健康检查

Q: 如何在所有主机间共享变量？
A: set_fact + delegate_to 或 hostvars:
   - set_fact:
       shared_data: "{{ local_data }}"
   - debug:
       msg: "{{ hostvars['web1'].shared_data }}"
```
